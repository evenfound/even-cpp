123 void QWebSocketDataProcessor::process(QIODevice *pIoDevice)
124 {
125  bool isDone = false;
126 
127  while (!isDone) {
128  QWebSocketFrame frame = QWebSocketFrame::readFrame(pIoDevice);
129  if (Q_LIKELY(frame.isValid())) {
130  if (frame.isControlFrame()) {
131  isDone = processControlFrame(frame);
132  } else {
133  //we have a dataframe; opcode can be OC_CONTINUE, OC_TEXT or OC_BINARY
134  if (Q_UNLIKELY(!m_isFragmented && frame.isContinuationFrame())) {
135  clear();
136  Q_EMIT errorEncountered(QWebSocketProtocol::CloseCodeProtocolError,
137  tr("Received Continuation frame, while there is " \
138  "nothing to continue."));
139  return;
140  }
141  if (Q_UNLIKELY(m_isFragmented && frame.isDataFrame() &&
142  !frame.isContinuationFrame())) {
143  clear();
144  Q_EMIT errorEncountered(QWebSocketProtocol::CloseCodeProtocolError,
145  tr("All data frames after the initial data frame " \
146  "must have opcode 0 (continuation)."));
147  return;
148  }
149  if (!frame.isContinuationFrame()) {
150  m_opCode = frame.opCode();
151  m_isFragmented = !frame.isFinalFrame();
152  }
153  quint64 messageLength = (quint64)(m_opCode == QWebSocketProtocol::OpCodeText)
154  ? m_textMessage.length()
155  : m_binaryMessage.length();
156  if (Q_UNLIKELY((messageLength + quint64(frame.payload().length())) >
157  MAX_MESSAGE_SIZE_IN_BYTES)) {
158  clear();
159  Q_EMIT errorEncountered(QWebSocketProtocol::CloseCodeTooMuchData,
160  tr("Received message is too big."));
161  return;
162  }
163 
164  if (m_opCode == QWebSocketProtocol::OpCodeText) {
165  QString frameTxt = m_pTextCodec->toUnicode(frame.payload().constData(),
166  frame.payload().size(),
167  m_pConverterState);
168  bool failed = (m_pConverterState->invalidChars != 0)
169  || (frame.isFinalFrame() && (m_pConverterState->remainingChars != 0));
170  if (Q_UNLIKELY(failed)) {
171  clear();
172  Q_EMIT errorEncountered(QWebSocketProtocol::CloseCodeWrongDatatype,
173  tr("Invalid UTF-8 code encountered."));
174  return;
175  } else {
176  m_textMessage.append(frameTxt);
177  Q_EMIT textFrameReceived(frameTxt, frame.isFinalFrame());
178  }
179  } else {
180  m_binaryMessage.append(frame.payload());
181  Q_EMIT binaryFrameReceived(frame.payload(), frame.isFinalFrame());
182  }
183 
184  if (frame.isFinalFrame()) {
185  if (m_opCode == QWebSocketProtocol::OpCodeText)
186  Q_EMIT textMessageReceived(m_textMessage);
187  else
188  Q_EMIT binaryMessageReceived(m_binaryMessage);
189  clear();
190  isDone = true;
191  }
192  }
193  } else {
194  Q_EMIT errorEncountered(frame.closeCode(), frame.closeReason());
195  clear();
196  isDone = true;
197  }
198  }
199 }

204 void QWebSocketDataProcessor::clear()
205 {
206  m_processingState = PS_READ_HEADER;
207  m_isFinalFrame = false;
208  m_isFragmented = false;
209  m_opCode = QWebSocketProtocol::OpCodeClose;

223  QTextCodec::IgnoreHeader);

229 bool QWebSocketDataProcessor::processControlFrame(const QWebSocketFrame &frame)
230 {
231  bool mustStopProcessing = true; //control frames never expect additional frames to be processed
232  switch (frame.opCode()) {
233  case QWebSocketProtocol::OpCodePing:
234  Q_EMIT pingReceived(frame.payload());
235  break;
236 
237  case QWebSocketProtocol::OpCodePong:
238  Q_EMIT pongReceived(frame.payload());
239  break;
240 
241  case QWebSocketProtocol::OpCodeClose:
242  {
243  quint16 closeCode = QWebSocketProtocol::CloseCodeNormal;
244  QString closeReason;
245  QByteArray payload = frame.payload();
246  if (Q_UNLIKELY(payload.size() == 1)) {
247  //size is either 0 (no close code and no reason)
248  //or >= 2 (at least a close code of 2 bytes)
249  closeCode = QWebSocketProtocol::CloseCodeProtocolError;
250  closeReason = tr("Payload of close frame is too small.");
251  } else if (Q_LIKELY(payload.size() > 1)) {
252  //close frame can have a close code and reason
253  closeCode = qFromBigEndian<quint16>(reinterpret_cast<const uchar *>(payload.constData()));
254  if (Q_UNLIKELY(!QWebSocketProtocol::isCloseCodeValid(closeCode))) {
255  closeCode = QWebSocketProtocol::CloseCodeProtocolError;
256  closeReason = tr("Invalid close code %1 detected.").arg(closeCode);
257  } else {
258  if (payload.size() > 2) {
259  QTextCodec *tc = QTextCodec::codecForName(QByteArrayLiteral("UTF-8"));
260  QTextCodec::ConverterState state(QTextCodec::ConvertInvalidToNull);
261  closeReason = tc->toUnicode(payload.constData() + 2, payload.size() - 2, &state);
262  const bool failed = (state.invalidChars != 0) || (state.remainingChars != 0);
263  if (Q_UNLIKELY(failed)) {
264  closeCode = QWebSocketProtocol::CloseCodeWrongDatatype;
265  closeReason = tr("Invalid UTF-8 code encountered.");
266  }
267  }
268  }
269  }
270  Q_EMIT closeReceived(static_cast<QWebSocketProtocol::CloseCode>(closeCode), closeReason);
271  break;
272  }
273 
274  case QWebSocketProtocol::OpCodeContinue:
275  case QWebSocketProtocol::OpCodeBinary:
276  case QWebSocketProtocol::OpCodeText:
277  case QWebSocketProtocol::OpCodeReserved3:
278  case QWebSocketProtocol::OpCodeReserved4:
279  case QWebSocketProtocol::OpCodeReserved5:
280  case QWebSocketProtocol::OpCodeReserved6:
281  case QWebSocketProtocol::OpCodeReserved7:
282  case QWebSocketProtocol::OpCodeReservedC:
283  case QWebSocketProtocol::OpCodeReservedB:
284  case QWebSocketProtocol::OpCodeReservedD:
285  case QWebSocketProtocol::OpCodeReservedE:
286  case QWebSocketProtocol::OpCodeReservedF:
287  //do nothing
288  //case statements added to make C++ compiler happy
289  break;
290 
291  default:
292  Q_EMIT errorEncountered(QWebSocketProtocol::CloseCodeProtocolError,
293  tr("Invalid opcode detected: %1").arg(int(frame.opCode())));
294  //do nothing
295  break;
296  }
297  return mustStopProcessing;
298 }
