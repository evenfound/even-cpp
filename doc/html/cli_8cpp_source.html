76  }

188 void Command::do_read_file(std::istream& in,
189  std::function<void (uint8_t[], size_t)> consumer_fn,
190  size_t buf_size) const
191  {
192  // Avoid an infinite loop on --buf-size=0
193  std::vector<uint8_t> buf(buf_size == 0 ? 4096 : buf_size);
194 
195  while(in.good())
196  {
197  in.read(reinterpret_cast<char*>(buf.data()), buf.size());
198  const size_t got = static_cast<size_t>(in.gcount());
199  consumer_fn(buf.data(), got);
200  }
201  }

213 std::string Command::get_passphrase(const std::string& prompt)
214  {
215  error_output() << prompt << ": " << std::flush;
216  std::string pass;
217 
218 #if defined(BOTAN_TARGET_OS_HAS_POSIX1)
219 
220  struct termios old_flags;
221  int stdin_fd = fileno(stdin);
222  ::tcgetattr(stdin_fd, &old_flags);
223  struct termios noecho_flags = old_flags;
224  noecho_flags.c_lflag &= ~ECHO;
225  noecho_flags.c_lflag |= ECHONL;
226 
227  if(::tcsetattr(stdin_fd, TCSANOW, &noecho_flags) != 0)
228  throw CLI_Error("Clearing terminal echo bit failed");
229 
230  std::getline(std::cin, pass);
231 
232  if(::tcsetattr(stdin_fd, TCSANOW, &old_flags) != 0)
233  throw CLI_Error("Restoring terminal echo bit failed");
234 #else
235 
236  // TODO equivalent for Windows ...
237  std::getline(std::cin, pass);
238 
239 #endif
240 
241  return pass;
242  }
