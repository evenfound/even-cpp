46  * This slavishly emulates the behavior of the reference
47  * implementations RNG, in order to ensure that from the same
48  * random seed we compute the exact same result.

81  NEWHOPE_RNG(const std::vector<uint8_t>& seed)
82  {
83  if(seed.size() != 64 && seed.size() != 32)

86  }
