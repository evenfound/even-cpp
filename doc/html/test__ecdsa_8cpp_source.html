166  Test::Result run_one_test(const std::string&, const VarMap& vars) override
167  {
168  Test::Result result("ECDSA invalid keys");
169 
170  const std::string group_id = vars.get_req_str("Group");
171  Botan::EC_Group group(Botan::OIDS::lookup(group_id));
172  const Botan::BigInt x = vars.get_req_bn("InvalidKeyX");
173  const Botan::BigInt y = vars.get_req_bn("InvalidKeyY");
174 
175  std::unique_ptr<Botan::PointGFp> public_point;
176 
177  try
178  {
179  public_point.reset(new Botan::PointGFp(group.point(x, y)));
180  }
181  catch(Botan::Invalid_Argument&)
182  {
183  // PointGFp() performs a range check on x, y in [0, p−1],
184  // which is also part of the EC public key checks, e.g.,
185  // in NIST SP800-56A rev2, sec. 5.6.2.3.2
186  result.test_success("public key fails check");
187  return result;
188  }
189 
190  std::unique_ptr<Botan::Public_Key> key(new Botan::ECDSA_PublicKey(group, *public_point));
191  result.test_eq("public key fails check", key->check_key(Test::rng(), false), false);
192  return result;
193  }
194  };
