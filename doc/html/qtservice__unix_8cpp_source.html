69 static QString encodeName(const QString &name, bool allowUpper = false)
70 {
71  QString n = name.toLower();
72  QString legal = QLatin1String("abcdefghijklmnopqrstuvwxyz1234567890");
73  if (allowUpper)
74  legal += QLatin1String("ABCDEFGHIJKLMNOPQRSTUVWXYZ");
75  int pos = 0;
76  while (pos < n.size()) {
77  if (legal.indexOf(n[pos]) == -1)
78  n.remove(pos, 1);
79  else
80  ++pos;
81  }
82  return n;
83 }

101 static bool sendCmd(const QString &serviceName, const QString &cmd)
102 {
103  bool retValue = false;
104  QtUnixSocket sock;
105  if (sock.connectTo(socketPath(serviceName))) {
106  sock.write(QString(cmd+"\r\n").toLatin1().constData());
107  sock.flush();
108  sock.waitForReadyRead(-1);
109  QString reply = sock.readAll();
110  if (reply == QLatin1String("true"))
111  retValue = true;
112  sock.close();
113  }
114  return retValue;
115 }

117 static QString absPath(const QString &path)
118 {
119  QString ret;
120  if (path[0] != QChar('/')) { // Not an absolute path
121  int slashpos;
122  if ((slashpos = path.lastIndexOf('/')) != -1) { // Relative path
123  QDir dir = QDir::current();
124  dir.cd(path.left(slashpos));
125  ret = dir.absolutePath();
126  } else { // Need to search $PATH
127  char *envPath = ::getenv("PATH");
128  if (envPath) {
129  QStringList envPaths = QString::fromLocal8Bit(envPath).split(':');
130  for (int i = 0; i < envPaths.size(); ++i) {
131  if (QFile::exists(envPaths.at(i) + QLatin1String("/") + QString(path))) {
132  QDir dir(envPaths.at(i));
133  ret = dir.absolutePath();
134  break;
135  }
136  }
137  }
138  }
139  } else {
140  QFileInfo fi(path);
141  ret = fi.absolutePath();
142  }
143  return ret;
144 }

321 void QtServiceSysPrivate::slotReady()
322 {
323  QTcpSocket *s = (QTcpSocket *)sender();
324  cache[s] += QString(s->readAll());
325  QString cmd = getCommand(s);
326  while (!cmd.isEmpty()) {
327  bool retValue = false;
328  if (cmd == QLatin1String("terminate")) {
329  if (!(serviceFlags & QtServiceBase::CannotBeStopped)) {
330  QtServiceBase::instance()->stop();
331  QCoreApplication::instance()->quit();
332  retValue = true;
333  }
334  } else if (cmd == QLatin1String("pause")) {
335  if (serviceFlags & QtServiceBase::CanBeSuspended) {
336  QtServiceBase::instance()->pause();
337  retValue = true;
338  }
339  } else if (cmd == QLatin1String("resume")) {
340  if (serviceFlags & QtServiceBase::CanBeSuspended) {
341  QtServiceBase::instance()->resume();
342  retValue = true;
343  }
344  } else if (cmd == QLatin1String("alive")) {
345  retValue = true;
346  } else if (cmd.length() > 4 && cmd.left(4) == QLatin1String("num:")) {
347  cmd = cmd.mid(4);
348  QtServiceBase::instance()->processCommand(cmd.toInt());
349  retValue = true;
350  }
351  QString retString;
352  if (retValue)
353  retString = QLatin1String("true");
354  else
355  retString = QLatin1String("false");
356  s->write(retString.toLatin1().constData());
357  s->flush();
358  cmd = getCommand(s);
359  }
360 }

445  int, uint, const QByteArray &)
