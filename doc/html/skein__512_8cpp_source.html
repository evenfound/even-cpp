70  {

100 void Skein_512::ubi_512(const uint8_t msg[], size_t msg_len)
101  {
102  secure_vector<uint64_t> M(8);
103 
104  do
105  {
106  const size_t to_proc = std::min<size_t>(msg_len, 64);
107  m_T[0] += to_proc;
108 
109  load_le(M.data(), msg, to_proc / 8);
110 
111  if(to_proc % 8)
112  {
113  for(size_t j = 0; j != to_proc % 8; ++j)
114  M[to_proc/8] |= static_cast<uint64_t>(msg[8*(to_proc/8)+j]) << (8*j);
115  }
116 
117  m_threefish->skein_feedfwd(M, m_T);
118 
119  // clear first flag if set
120  m_T[1] &= ~(static_cast<uint64_t>(1) << 62);
121 
122  msg_len -= to_proc;
123  msg += to_proc;
124  } while(msg_len);
125  }
