20 int ipfs_bitswap_network_send_message(const struct BitswapContext* context, struct Libp2pPeer* peer, const struct BitswapMessage* message) {
21  libp2p_logger_debug("bitswap_network", "Sending bitswap message to %s.\n", libp2p_peer_id_to_string(peer));
22  // get a connection to the peer
23  if (peer->connection_type != CONNECTION_TYPE_CONNECTED || peer->sessionContext == NULL) {
24  libp2p_peer_connect(context->ipfsNode->dialer, peer, context->ipfsNode->peerstore, context->ipfsNode->repo->config->datastore, 10);
25  if(peer->connection_type != CONNECTION_TYPE_CONNECTED)
26  return 0;
27  }
28  // protobuf the message
29  size_t buf_size = ipfs_bitswap_message_protobuf_encode_size(message);
30  uint8_t* buf = (uint8_t*) malloc(buf_size + 20);
31  if (buf == NULL)
32  return 0;
33  if (!ipfs_bitswap_message_protobuf_encode(message, &buf[20], buf_size, &buf_size)) {
34  free(buf);
35  return 0;
36  }
37  // tack on the protocol header
38  memcpy(buf, "/ipfs/bitswap/1.1.0\n", 20);
39  buf_size += 20;
40  // send it
41  struct StreamMessage outgoing;
42  outgoing.data = buf;
43  outgoing.data_size = buf_size;
44  int bytes_written = peer->sessionContext->default_stream->write(peer->sessionContext, &outgoing);
45  if (bytes_written <= 0) {
46  free(buf);
47  return 0;
48  }
49  free(buf);
50  return 1;
51 }

92  struct BitswapContext* bitswapContext = (struct BitswapContext*)node->exchange->exchangeContext;
93  // strip off the protocol header
94  int start = -1;
95  for(int i = 0; i < bytes_length; i++) {
96  if (bytes[i] == '\n') {
97  start = i+1;
98  break;
99  }
100  }
101  if (start == -1)
102  return 0;
103  // un-protobuf the message
104  struct BitswapMessage* message = NULL;
105  if (!ipfs_bitswap_message_protobuf_decode(&bytes[start], bytes_length - start, &message))
106  return 0;
107  // process the message
108  // payload - what we want
109  if (message->payload != NULL) {
110  for(int i = 0; i < message->payload->total; i++) {
111  struct Block* blk = (struct Block*)libp2p_utils_vector_get(message->payload, i);
112  // we need a copy of the block so it survives the destruction of the message
113  node->exchange->HasBlock(node->exchange, ipfs_block_copy(blk));
114  }
115  }
116  // wantlist - what they want
117  if (message->wantlist != NULL && message->wantlist->entries != NULL && message->wantlist->entries->total > 0) {
118  // get the peer
119  if (sessionContext->remote_peer_id == NULL) {
120  ipfs_bitswap_message_free(message);
121  return 0;
122  }
123  struct Libp2pPeer* peer = libp2p_peerstore_get_or_add_peer_by_id(node->peerstore, (unsigned char*)sessionContext->remote_peer_id, strlen(sessionContext->remote_peer_id));
124  if (peer == NULL) {
125  libp2p_logger_error("bitswap_network", "Unable to find or add peer %s of length %d to peerstore.\n", sessionContext->remote_peer_id, strlen(sessionContext->remote_peer_id));
126  ipfs_bitswap_message_free(message);
127  return 0;
128  }
129  // find the queue (adds it if it is not there)
130  struct PeerRequest* peerRequest = ipfs_peer_request_queue_find_peer(bitswapContext->peerRequestQueue, peer);
131  for(int i = 0; i < message->wantlist->entries->total; i++) {
132  struct WantlistEntry* entry = (struct WantlistEntry*) libp2p_utils_vector_get(message->wantlist->entries, i);
133  // turn the "block" back into a cid
134  struct Cid* cid = NULL;
135  if (!ipfs_cid_protobuf_decode(entry->block, entry->block_size, &cid) || cid->hash_length == 0) {
136  libp2p_logger_error("bitswap_network", "Message had invalid CID\n");
137  ipfs_cid_free(cid);
138  ipfs_bitswap_message_free(message);
139  return 0;
140  }
141  ipfs_bitswap_network_adjust_cid_queue(peerRequest->cids_they_want, cid, entry->cancel);
142  }
143  }
144  ipfs_bitswap_message_free(message);
145  return 1;
146 }
