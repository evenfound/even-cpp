54 size_t PKCS7_Padding::unpad(const uint8_t input[], size_t input_length) const
55  {
56  if(input_length <= 2)
57  return input_length;
58 
59  CT::poison(input, input_length);
60  size_t bad_input = 0;
61  const uint8_t last_byte = input[input_length-1];
62 
63  bad_input |= CT::expand_mask<size_t>(last_byte > input_length);
64 
65  const size_t pad_pos = input_length - last_byte;
66 
67  for(size_t i = 0; i != input_length - 1; ++i)
68  {
69  const uint8_t in_range = CT::expand_mask<uint8_t>(i >= pad_pos);
70  bad_input |= in_range & (~CT::is_equal(input[i], last_byte));
71  }
72 
73  CT::unpoison(input, input_length);
74  return CT::conditional_return(bad_input, input_length, pad_pos);

133 * Unpad with One and Zeros Method
134 */
135 size_t OneAndZeros_Padding::unpad(const uint8_t input[], size_t input_length) const
136  {
137  if(input_length <= 2)
138  return input_length;
139 
140  CT::poison(input, input_length);
141 
142  uint8_t bad_input = 0;
143  uint8_t seen_one = 0;
144  size_t pad_pos = input_length - 1;
145  size_t i = input_length;
146 
147  while(i)
148  {
149  seen_one |= CT::is_equal<uint8_t>(input[i-1], 0x80);
150  pad_pos -= CT::select<uint8_t>(~seen_one, 1, 0);
151  bad_input |= ~CT::is_zero<uint8_t>(input[i-1]) & ~seen_one;
152  i--;
153  }
154  bad_input |= ~seen_one;
155 
156  CT::unpoison(input, input_length);
157  return CT::conditional_return(bad_input, input_length, pad_pos);
158  }

179  {

182

184

186  uint8_t bad_input = 0;
187  bad_input |= CT::is_zero(last_byte) | CT::expand_mask<uint8_t>(last_byte > input_length);
188 
189  const size_t pad_pos = input_length - last_byte;
190  size_t i = input_length - 1;
191  while(i)
192  {
193  const uint8_t in_range = CT::expand_mask<uint8_t>(i > pad_pos);
194  bad_input |= (~CT::is_equal<uint8_t>(input[i-1], input[i]-1)) & in_range;
195  --i;
196  }
197 
198  CT::unpoison(input, input_length);
199  return CT::conditional_return(bad_input, input_length, pad_pos);
