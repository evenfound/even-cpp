220  if (entry != NULL) {

222  // do we have something in the queue?

224  return 1;

226  return 1;

232  if (retVal < 0) {

234  libp2p_peer_handle_connection_error(request->peer);
235  return 0;
236  }
237  if (retVal > 0) {
238  libp2p_logger_debug("peer_request_queue", "We have something to read. %d bytes.\n", retVal);

368 int ipfs_bitswap_peer_request_process_entry(const struct BitswapContext* context, struct PeerRequest* request) {
369  // determine if we have enough information to continue
370  if (request == NULL)
371  return 0;
372  if (request->peer == NULL)
373  return 0;
374  if (!request->peer->is_local) {
375  if (request->peer->connection_type != CONNECTION_TYPE_CONNECTED)
376  if (request->peer->addr_head == NULL || request->peer->addr_head->item == NULL)
377  return 0;
378  }
379  // determine if we're connected
380  int connected = request->peer->is_local || request->peer->connection_type == CONNECTION_TYPE_CONNECTED;
381  int need_to_connect = ipfs_bitswap_peer_request_we_want_cids(request->cids_we_want) || ipfs_bitswap_peer_request_cids_waiting(request->cids_they_want) || request->blocks_we_want_to_send->total != 0;
382 
383  // determine if we need to connect
384  if (need_to_connect) {
385  if (!connected) {
386  // connect
387  connected = libp2p_peer_connect(context->ipfsNode->dialer, request->peer, context->ipfsNode->peerstore, context->ipfsNode->repo->config->datastore, 0);
388  }
389  if (connected) {
390  // build a message
391  struct BitswapMessage* msg = ipfs_bitswap_message_new();
392  // see if we can fulfill any of their requests. If so, fill in msg->payload
393  ipfs_bitswap_peer_request_get_blocks_they_want(context, request);
394  ipfs_bitswap_message_add_blocks(msg, request->blocks_we_want_to_send, request->cids_they_want);
395  // add requests that we would like
396  ipfs_bitswap_message_add_wantlist_items(msg, request->cids_we_want);
397  // send message
398  if (ipfs_bitswap_network_send_message(context, request->peer, msg)) {
399  ipfs_bitswap_message_free(msg);
400  return 1;
401  }
402  ipfs_bitswap_message_free(msg);
403  }
404  }
405  return 0;
406 }
