20 #define SHA2_32_F(A, B, C, D, E, F, G, H, M1, M2, M3, M4, magic) do { \
21  uint32_t A_rho = rotr<2>(A) ^ rotr<13>(A) ^ rotr<22>(A); \
22  uint32_t E_rho = rotr<6>(E) ^ rotr<11>(E) ^ rotr<25>(E); \
23  uint32_t M2_sigma = rotr<17>(M2) ^ rotr<19>(M2) ^ (M2 >> 10); \
24  uint32_t M4_sigma = rotr<7>(M4) ^ rotr<18>(M4) ^ (M4 >> 3); \
25  H += magic + E_rho + ((E & F) ^ (~E & G)) + M1; \
26  D += H; \
27  H += A_rho + ((A & B) | ((A | B) & C)); \
28  M1 += M2_sigma + M3 + M4_sigma; \
29  } while(0);
