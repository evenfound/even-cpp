92  void consume_input(const uint8_t buf[], size_t buf_len)
93  {
94  m_req_buf.append(reinterpret_cast<const char*>(buf), buf_len);
95 
96  std::istringstream strm(m_req_buf);
97 
98  std::string http_version;
99  std::string verb;
100  std::string location;
101  std::map<std::string, std::string> headers;
102 
103  strm >> verb >> location >> http_version;
104 
105  if(verb.empty() || location.empty())
106  return;
107 
108  while(true)
109  {
110  std::string header_line;
111  std::getline(strm, header_line);
112 
113  if(header_line == "\r")
114  {
115  continue;
116  }
117 
118  auto delim = header_line.find(": ");
119  if(delim == std::string::npos)
120  {
121  break;
122  }
123 
124  const std::string hdr_name = header_line.substr(0, delim);
125  const std::string hdr_val = header_line.substr(delim + 2, std::string::npos);
126 
127  headers[hdr_name] = hdr_val;
128 
129  if(headers.size() > 1024)
130  throw std::runtime_error("That's an awful lot of headers");
131  }
132 
133  if(verb != "" && location != "")
134  {
135  Request req(verb, location, headers);
136  m_cb.handle_http_request(req);
137  m_req_buf.clear();
138  }
139  else
140  printf("ignoring\n");
141  }
142  private:
143  Callbacks& m_cb;
144  std::string m_req_buf;
145  };
