308  struct Stream* parent_stream = libp2p_yamux_get_parent_stream(stream_context);
309  // this is the normal situation (not dead code).
310  struct StreamMessage* incoming = NULL;
311  // we need a lock
312  if (parent_stream->read(parent_stream->stream_context, &incoming, yamux_default_timeout)) {
313  libp2p_logger_debug("yamux", "read: successfully read %d bytes from network.\n", incoming->data_size);
314  // This could be a data frame with the actual data coming later. Yuck.
315  // JMJ in the case of an incomplete buffer, the next read should be the data. This must
316  // be true, as the next data does not have a frame. We should only read the bytes we need.
317  int moreToRead = yamux_more_to_read(incoming);
318  if (moreToRead > 0) {
319  uint8_t buffer[moreToRead];
320  if (parent_stream->read_raw(parent_stream->stream_context, buffer, moreToRead, timeout_secs) == moreToRead) {
321  // we have the bytes we need
322  uint8_t* new_buffer = (uint8_t*) malloc(incoming->data_size + moreToRead);
323  memcpy(new_buffer, incoming->data, incoming->data_size);
324  memcpy(&new_buffer[incoming->data_size], buffer, moreToRead);
325  incoming->data_size += moreToRead;
326  free(incoming->data);
327  incoming->data = new_buffer;
328  } else {
329  // we didn't get the bytes we needed
330  return 0;
331  }
332  }
333  // parse the frame. This is where the work happens.
334  if (yamux_decode(ctx, incoming->data, incoming->data_size, message) >= 0) {
335  libp2p_stream_message_free(incoming);
336  // The message may not have anything in it. If so, return 0, as if nothing was done. Everything has been handled
337  if (*message != NULL && (*message)->data_size == 0) {
338  libp2p_stream_message_free(*message);
339  *message = NULL;
340  return 0;
341  }
342  return 1;
343  }
344  libp2p_logger_error("yamux", "yamux_decode returned error.\n");
345  libp2p_stream_message_free(incoming);
346  } else {
347  // read failed
348  }

351  return 0;

404 int libp2p_yamux_write(void* stream_context, struct StreamMessage* message) {
405  if (stream_context == NULL)
406  return 0;
407  // look at the first byte of the context to determine if this is a YamuxContext (we're negotiating)
408  // or a YamuxChannelContext (we're talking to an established channel)
409  struct YamuxContext* ctx = NULL;
410  struct YamuxChannelContext* channel = NULL;
411  char proto = ((uint8_t*)stream_context)[0];
412  if (proto == YAMUX_CHANNEL_CONTEXT) {
413  channel = (struct YamuxChannelContext*)stream_context;
414  ctx = channel->yamux_context;
415  } else if (proto == YAMUX_CONTEXT) {
416  ctx = (struct YamuxContext*)stream_context;
417  }
418 
419  if (ctx == NULL && channel == NULL)
420  return 0;
421 
422  if (ctx->state != yamux_stream_est) {
423  struct Stream* parent_stream = ctx->stream->parent_stream;
424  return parent_stream->write(parent_stream->stream_context, message);
425  }
426 
427  struct StreamMessage* outgoing_message = libp2p_yamux_prepare_to_send(message);
428  // now convert fame for network use
429  struct yamux_frame* frame = (struct yamux_frame*)outgoing_message->data;
430  // set a few more flags
431  frame->flags = get_flags(stream_context);
432  if (channel == NULL) {
433  // if we don't yet have a channel, set the id to the next available
434  frame->streamid = libp2p_yamux_get_next_id(ctx);
435  } else {
436  frame->streamid = channel->channel;
437  }
438  encode_frame(frame);
439 
440  int retVal = 0;
441  if (channel != NULL && channel->channel != 0) {
442  // we have an established channel. Use it.
443  libp2p_logger_debug("yamux", "About to write %d bytes to yamux channel %d.\n", outgoing_message->data_size, channel->channel);
444  struct Stream* parent_stream = libp2p_yamux_get_parent_stream(stream_context);
445  retVal = parent_stream->write(parent_stream->stream_context, outgoing_message);
446  } else if (ctx != NULL) {
447  libp2p_logger_debug("yamux", "About to write %d bytes to stream.\n", outgoing_message->data_size);
448  retVal = ctx->stream->parent_stream->write(ctx->stream->parent_stream->stream_context, outgoing_message);
449  }
450  libp2p_stream_message_free(outgoing_message);
451 
452  return retVal;
453 }
