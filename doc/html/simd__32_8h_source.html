519  SIMD_4x32 bswap() const
520  {
521 #if defined(BOTAN_SIMD_USE_SSE2)
522 
523  __m128i T = m_sse;
524  T = _mm_shufflehi_epi16(T, _MM_SHUFFLE(2, 3, 0, 1));
525  T = _mm_shufflelo_epi16(T, _MM_SHUFFLE(2, 3, 0, 1));
526  return SIMD_4x32(_mm_or_si128(_mm_srli_epi16(T, 8), _mm_slli_epi16(T, 8)));
527 
528 #elif defined(BOTAN_SIMD_USE_ALTIVEC)
529 
530  union {
531  __vector unsigned int V;
532  uint32_t R[4];
533  } vec;
534 
535  vec.V = m_vmx;
536  bswap_4(vec.R);
537  return SIMD_4x32(vec.R[0], vec.R[1], vec.R[2], vec.R[3]);
538 
539 #elif defined(BOTAN_SIMD_USE_NEON)
540 
541  //return SIMD_4x32(vrev64q_u32(m_neon));
542 
543  // FIXME this is really slow
544  SIMD_4x32 ror8 = this->rotr<8>();
545  SIMD_4x32 rol8 = this->rotl<8>();
546 
547  const SIMD_4x32 mask1 = SIMD_4x32::splat(0xFF00FF00);
548  const SIMD_4x32 mask2 = SIMD_4x32::splat(0x00FF00FF);
549  return (ror8 & mask1) | (rol8 & mask2);
550 #else
551  // scalar
552  return SIMD_4x32(reverse_bytes(m_scalar[0]),
553  reverse_bytes(m_scalar[1]),
554  reverse_bytes(m_scalar[2]),
555  reverse_bytes(m_scalar[3]));

562  static void transpose(SIMD_4x32& B0, SIMD_4x32& B1,
563  SIMD_4x32& B2, SIMD_4x32& B3)
564  {
565 #if defined(BOTAN_SIMD_USE_SSE2)
566  const __m128i T0 = _mm_unpacklo_epi32(B0.m_sse, B1.m_sse);
567  const __m128i T1 = _mm_unpacklo_epi32(B2.m_sse, B3.m_sse);
568  const __m128i T2 = _mm_unpackhi_epi32(B0.m_sse, B1.m_sse);
569  const __m128i T3 = _mm_unpackhi_epi32(B2.m_sse, B3.m_sse);
570 
571  B0.m_sse = _mm_unpacklo_epi64(T0, T1);
572  B1.m_sse = _mm_unpackhi_epi64(T0, T1);
573  B2.m_sse = _mm_unpacklo_epi64(T2, T3);
574  B3.m_sse = _mm_unpackhi_epi64(T2, T3);
575 #elif defined(BOTAN_SIMD_USE_ALTIVEC)
576  const __vector unsigned int T0 = vec_mergeh(B0.m_vmx, B2.m_vmx);
577  const __vector unsigned int T1 = vec_mergeh(B1.m_vmx, B3.m_vmx);
578  const __vector unsigned int T2 = vec_mergel(B0.m_vmx, B2.m_vmx);
579  const __vector unsigned int T3 = vec_mergel(B1.m_vmx, B3.m_vmx);
580 
581  B0.m_vmx = vec_mergeh(T0, T1);
582  B1.m_vmx = vec_mergel(T0, T1);
583  B2.m_vmx = vec_mergeh(T2, T3);
584  B3.m_vmx = vec_mergel(T2, T3);
585 #elif defined(BOTAN_SIMD_USE_NEON)
586 
587 #if defined(BOTAN_TARGET_ARCH_IS_ARM32)
588 
589  const uint32x4x2_t T0 = vzipq_u32(B0.m_neon, B2.m_neon);
590  const uint32x4x2_t T1 = vzipq_u32(B1.m_neon, B3.m_neon);
591  const uint32x4x2_t O0 = vzipq_u32(T0.val[0], T1.val[0]);
592  const uint32x4x2_t O1 = vzipq_u32(T0.val[1], T1.val[1]);
593 
594  B0.m_neon = O0.val[0];
595  B1.m_neon = O0.val[1];
596  B2.m_neon = O1.val[0];
597  B3.m_neon = O1.val[1];
598 
599 #elif defined(BOTAN_TARGET_ARCH_IS_ARM64)
600  const uint32x4_t T0 = vzip1q_u32(B0.m_neon, B2.m_neon);
601  const uint32x4_t T2 = vzip2q_u32(B0.m_neon, B2.m_neon);
602 
603  const uint32x4_t T1 = vzip1q_u32(B1.m_neon, B3.m_neon);
604  const uint32x4_t T3 = vzip2q_u32(B1.m_neon, B3.m_neon);
605 
606  B0.m_neon = vzip1q_u32(T0, T1);
607  B1.m_neon = vzip2q_u32(T0, T1);
608 
609  B2.m_neon = vzip1q_u32(T2, T3);
