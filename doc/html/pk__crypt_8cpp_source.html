43  {
44  throw CLI_Error("Unable to load public key");
45  }
46 
47  if(key->algo_name() != "RSA")
48  {
49  throw CLI_Usage_Error("This function requires an RSA key");
50  }

52  const std::string OAEP_HASH = "SHA-256";
53  const std::string aead_algo = get_arg("aead");
54 
55  std::unique_ptr<Botan::AEAD_Mode> aead =
56  Botan::AEAD_Mode::create(aead_algo, Botan::ENCRYPTION);
57 
58  if(!aead)
59  throw CLI_Usage_Error("The AEAD '" + aead_algo + "' is not available");
60 
61  const Botan::OID aead_oid = Botan::OIDS::lookup(aead_algo);
62  if(aead_oid.empty())
63  throw CLI_Usage_Error("No OID defined for AEAD '" + aead_algo + "'");
64 
65  Botan::secure_vector<uint8_t> data;
66  auto insert_fn = [&](const uint8_t b[], size_t l)
67  {
68  data.insert(data.end(), b, b + l);
69  };
70  this->read_file(get_arg("datafile"), insert_fn);
71 
72  const Botan::AlgorithmIdentifier hash_id(OAEP_HASH, Botan::AlgorithmIdentifier::USE_EMPTY_PARAM);
73  const Botan::AlgorithmIdentifier pk_alg_id("RSA/OAEP", hash_id.BER_encode());
74 
75  Botan::PK_Encryptor_EME enc(*key, rng(), "OAEP(" + OAEP_HASH + ")");
76 
77  const Botan::secure_vector<uint8_t> file_key = rng().random_vec(aead->key_spec().maximum_keylength());
78 
79  const std::vector<uint8_t> encrypted_key = enc.encrypt(file_key, rng());
80 
81  const Botan::secure_vector<uint8_t> nonce = rng().random_vec(aead->default_nonce_length());
82  aead->set_key(file_key);
83  aead->set_associated_data_vec(encrypted_key);
84  aead->start(nonce);
85 
86  aead->finish(data);
87 
88  std::vector<uint8_t> buf;
89  Botan::DER_Encoder der(buf);
90 
91  der.start_cons(Botan::SEQUENCE)
92  .encode(pk_alg_id)
93  .encode(encrypted_key, Botan::OCTET_STRING)
94  .encode(aead_oid)
95  .encode(nonce, Botan::OCTET_STRING)
96  .encode(data, Botan::OCTET_STRING)
97  .end_cons();
