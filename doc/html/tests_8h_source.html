451  template<typename Alloc>
452  static std::vector<uint8_t, Alloc>
453  mutate_vec(const std::vector<uint8_t, Alloc>& v,
454  bool maybe_resize = false,
455  size_t min_offset = 0)
456  {
457  auto& rng = Test::rng();
458 
459  std::vector<uint8_t, Alloc> r = v;
460 
461  if(maybe_resize && (r.empty() || rng.next_byte() < 32))
462  {
463  // TODO: occasionally truncate, insert at random index
464  const size_t add = 1 + (rng.next_byte() % 16);
465  r.resize(r.size() + add);
466  rng.randomize(&r[r.size() - add], add);
467  }
468 
469  if(r.size() > min_offset)
470  {
471  const size_t offset = std::max<size_t>(min_offset, rng.next_byte() % r.size());
472  const uint8_t perturb = rng.next_nonzero_byte();
473  r[offset] ^= perturb;
474  }
475 
476  return r;
477  }
