19 def parse_version_file(version_path):
20  version_file = open(version_path)
21  key_and_val = re.compile(r"([a-z_]+) = ([a-zA-Z0-9:\-\']+)")
22 
23  results = {}
24  for line in version_file.readlines():
25  if not line or line[0] == '#':
26  continue
27  match = key_and_val.match(line)
28  if match:
29  key = match.group(1)
30  val = match.group(2)
31 
32  if val == 'None':
33  val = None
34  elif val.startswith("'") and val.endswith("'"):
35  val = val[1:len(val)-1]
36  else:
37  val = int(val)
38 
39  results[key] = val
40  return results
