109  {

111 
112  std::vector<std::string> kdfs = { "KDF2(SHA-512)", "KDF1-18033(SHA-512)" };

114 
115  const size_t mac_key_len = 16;
116 
117  std::unique_ptr<Botan::KDF> kdf;
118  std::unique_ptr<Botan::MAC> mac;
119 
120  Botan::DH_PrivateKey alice(Test::rng(), Botan::DL_Group("modp/ietf/2048"));
121  Botan::DH_PrivateKey bob(Test::rng(), Botan::DL_Group("modp/ietf/2048"));
122 
123  for(const auto& kfunc : kdfs)
124  {
125  kdf = Botan::KDF::create(kfunc);
126 
127  if(!kdf)
128  {
129  result.test_note("Skipping due to missing KDF: " + kfunc);
130  continue;
131  }
132 
133  for(const auto& mfunc : macs)
134  {
135  mac = Botan::MAC::create(mfunc);
136 
137  if(!mac)
138  {
139  result.test_note("Skipping due to missing MAC: " + mfunc);
140  continue;
141  }
142 
143  Botan::DLIES_Encryptor encryptor(alice, Test::rng(), kdf->clone(), mac->clone(), mac_key_len);
144 
145  // negative test: other pub key not set
146  Botan::secure_vector<uint8_t> plaintext = Test::rng().random_vec(32);
147 
148  result.test_throws("encrypt not possible without setting other public key", [&encryptor, &plaintext]()
149  {
150  encryptor.encrypt(plaintext, Test::rng());
151  });
152 
153  encryptor.set_other_key(bob.public_value());
154  std::vector<uint8_t> ciphertext = encryptor.encrypt(plaintext, Test::rng());
155 
156  Botan::DLIES_Decryptor decryptor(bob, Test::rng(), kdf->clone(), mac->clone(), mac_key_len);
157 
158  // negative test: ciphertext too short
159  result.test_throws("ciphertext too short", [ &decryptor ]()
160  {
161  decryptor.decrypt(std::vector<uint8_t>(2));
162  });
163 
164  result.test_eq("decryption", decryptor.decrypt(ciphertext), plaintext);
165 
166  check_invalid_ciphertexts(result, decryptor, unlock(plaintext), ciphertext);
167  }
168  }

170  return result;
