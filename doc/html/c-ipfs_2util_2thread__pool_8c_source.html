207 void thpool_destroy(thpool_* thpool_p){
208  /* No need to destory if it's NULL */
209  if (thpool_p == NULL) return ;
210 
211  volatile int threads_total = thpool_p->num_threads_alive;
212 
213  /* End each thread 's infinite loop */
214  threads_keepalive = 0;
215 
216  /* Give one second to kill idle threads */
217  double TIMEOUT = 1.0;
218  time_t start, end;
219  double tpassed = 0.0;
220  time (&start);
221  while (tpassed < TIMEOUT && thpool_p->num_threads_alive){
222  bsem_post_all(thpool_p->jobqueue.has_jobs);
223  time (&end);
224  tpassed = difftime(end,start);
225  }
226

229  bsem_post_all(thpool_p->jobqueue.has_jobs);
230  sleep(1);

314 *
315 * In principle this is an endless loop. The only time this loop gets interuppted is once
316 * thpool_destroy() is invoked or the program exits.

319 * @return nothing

322 
323  /* Set thread name for profiling and debuging */
324  char thread_name[128] = {0};
325  sprintf(thread_name, "thread-pool-%d", thread_p->id);
326 
327 #if defined(__linux__)
328  /* Use prctl instead to prevent using _GNU_SOURCE flag and implicit declaration */
329  prctl(PR_SET_NAME, thread_name);
330 #elif defined(__APPLE__) && defined(__MACH__)
331  pthread_setname_np(thread_name);
332 #else
333  err("thread_do(): pthread_setname_np is not supported on this system");
334 #endif
335 
336  /* Assure all threads have been created before starting serving */
337  thpool_* thpool_p = thread_p->thpool_p;
338 
339  /* Register signal handler */
340  struct sigaction act;
341  sigemptyset(&act.sa_mask);
342  act.sa_flags = 0;
343  act.sa_handler = thread_hold;
344  if (sigaction(SIGUSR1, &act, NULL) == -1) {
345  err("thread_do(): cannot handle SIGUSR1");
346  }
347 
348  /* Mark thread as alive (initialized) */
349  pthread_mutex_lock(&thpool_p->thcount_lock);
350  thpool_p->num_threads_alive += 1;
351  pthread_mutex_unlock(&thpool_p->thcount_lock);
352 
353  while(threads_keepalive){
354 
355  bsem_wait(thpool_p->jobqueue.has_jobs);
356 
357  if (threads_keepalive){
358 
359  pthread_mutex_lock(&thpool_p->thcount_lock);
360  thpool_p->num_threads_working++;
361  pthread_mutex_unlock(&thpool_p->thcount_lock);
362 
363  /* Read job from queue and execute it */
364  void (*func_buff)(void*);
365  void* arg_buff;
366  job* job_p = jobqueue_pull(&thpool_p->jobqueue);
367  if (job_p) {
368  func_buff = job_p->function;
369  arg_buff = job_p->arg;
370  func_buff(arg_buff);
371  free(job_p);
372  }
373 
374  pthread_mutex_lock(&thpool_p->thcount_lock);
375  thpool_p->num_threads_working--;
376  if (!thpool_p->num_threads_working) {
377  pthread_cond_signal(&thpool_p->threads_all_idle);
378  }
379  pthread_mutex_unlock(&thpool_p->thcount_lock);
380 
381  }
382  }
383  pthread_mutex_lock(&thpool_p->thcount_lock);
384  thpool_p->num_threads_alive --;
385  pthread_mutex_unlock(&thpool_p->thcount_lock);
386 
387  return NULL;
388 }
