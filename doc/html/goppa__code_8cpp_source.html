40 secure_vector<gf2m> goppa_decode(const polyn_gf2m & syndrom_polyn,
41  const polyn_gf2m & g,
42  const std::vector<polyn_gf2m> & sqrtmod,
43  const std::vector<gf2m> & Linv)
44  {
45  gf2m a;
46  uint32_t code_length = Linv.size();
47  uint32_t t = g.get_degree();
48 
49  std::shared_ptr<GF2m_Field> sp_field = g.get_sp_field();
50 
51  std::pair<polyn_gf2m, polyn_gf2m> h_aux = polyn_gf2m::eea_with_coefficients( syndrom_polyn, g, 1);
52  polyn_gf2m & h = h_aux.first;
53  polyn_gf2m & aux = h_aux.second;
54  a = sp_field->gf_inv(aux.get_coef(0));
55  gf2m log_a = sp_field->gf_log(a);
56  for(int i = 0; i <= h.get_degree(); ++i)
57  {
58  h.set_coef(i,sp_field->gf_mul_zrz(log_a,h.get_coef(i)));
59  }
60 
61  // compute h(z) += z
62  h.add_to_coef( 1, 1);
63  // compute S square root of h (using sqrtmod)
64  polyn_gf2m S(t - 1, g.get_sp_field());
65 
66  for(uint32_t i=0;i<t;i++)
67  {
68  a = sp_field->gf_sqrt(h.get_coef(i));
69 
70  if(i & 1)
71  {
72  for(uint32_t j=0;j<t;j++)
73  {
74  S.add_to_coef( j, sp_field->gf_mul(a, sqrtmod[i/2].get_coef(j)));
75  }
76  }
77  else
78  {
79  S.add_to_coef( i/2, a);
80  }
81  } /* end for loop (i) */
82 
83 
84  S.get_degree();
85 
86  std::pair<polyn_gf2m, polyn_gf2m> v_u = polyn_gf2m::eea_with_coefficients(S, g, t/2+1);
87  polyn_gf2m & u = v_u.second;
88  polyn_gf2m & v = v_u.first;
89 
90  // sigma = u^2+z*v^2
91  polyn_gf2m sigma ( t , g.get_sp_field());
92 
93  const int u_deg = u.get_degree();
94  BOTAN_ASSERT(u_deg >= 0, "Valid degree");
95  for(int i = 0; i <= u_deg; ++i)
96  {
97  sigma.set_coef(2*i, sp_field->gf_square(u.get_coef(i)));
98  }
99 
100  const int v_deg = v.get_degree();
101  BOTAN_ASSERT(v_deg >= 0, "Valid degree");
102  for(int i = 0; i <= v_deg; ++i)
103  {
104  sigma.set_coef(2*i+1, sp_field->gf_square(v.get_coef(i)));
105  }
106 
107  secure_vector<gf2m> res = find_roots_gf2m_decomp(sigma, code_length);
108  size_t d = res.size();
109 
110  secure_vector<gf2m> result(d);
111  for(uint32_t i = 0; i < d; ++i)
112  {
113  gf2m current = res[i];
114 
115  gf2m tmp;
116  tmp = gray_to_lex(current);
117  if(tmp >= code_length) /* invalid root */
118  {
119  result[i] = i;
120  }
121  result[i] = Linv[tmp];
122  }
123 
124  return result;
125  }

164 secure_vector<uint8_t> mceliece_decrypt(
165  secure_vector<gf2m> & error_pos,
166  const uint8_t *ciphertext, uint32_t ciphertext_len,
167  const McEliece_PrivateKey & key)
168  {
169 
170  uint32_t dimension = key.get_dimension();
171  uint32_t codimension = key.get_codimension();
172  uint32_t t = key.get_goppa_polyn().get_degree();
173  polyn_gf2m syndrome_polyn(key.get_goppa_polyn().get_sp_field()); // init as zero polyn
174  const unsigned unused_pt_bits = dimension % 8;
175  const uint8_t unused_pt_bits_mask = (1 << unused_pt_bits) - 1;
176 
177  if(ciphertext_len != (key.get_code_length()+7)/8)
178  {
179  throw Invalid_Argument("wrong size of McEliece ciphertext");
180  }
181  uint32_t cleartext_len = (key.get_message_word_bit_length()+7)/8;
182 
183  if(cleartext_len != bit_size_to_byte_size(dimension))
184  {
185  throw Invalid_Argument("mce-decryption: wrong length of cleartext buffer");
186  }
187 
188  secure_vector<uint32_t> syndrome_vec(bit_size_to_32bit_size(codimension));
189  matrix_arr_mul(key.get_H_coeffs(),
190  key.get_code_length(),
191  bit_size_to_32bit_size(codimension),
192  ciphertext,
193  syndrome_vec.data(), syndrome_vec.size());
194 
195  secure_vector<uint8_t> syndrome_byte_vec(bit_size_to_byte_size(codimension));
196  uint32_t syndrome_byte_vec_size = syndrome_byte_vec.size();
197  for(uint32_t i = 0; i < syndrome_byte_vec_size; i++)
198  {
199  syndrome_byte_vec[i] = syndrome_vec[i/4] >> (8* (i % 4));
200  }
201 
202  syndrome_polyn = polyn_gf2m(t-1, syndrome_byte_vec.data(), bit_size_to_byte_size(codimension), key.get_goppa_polyn().get_sp_field());
203 
204  syndrome_polyn.get_degree();
205  error_pos = goppa_decode(syndrome_polyn, key.get_goppa_polyn(), key.get_sqrtmod(), key.get_Linv());
206 
207  uint32_t nb_err = error_pos.size();
208 
209  secure_vector<uint8_t> cleartext(cleartext_len);
210  copy_mem(cleartext.data(), ciphertext, cleartext_len);
211 
212  for(uint32_t i = 0; i < nb_err; i++)
213  {
214  gf2m current = error_pos[i];
215 
216  if(current >= cleartext_len * 8)
217  {
218  // an invalid position, this shouldn't happen
219  continue;
220  }
221  cleartext[current / 8] ^= (1 << (current % 8));
222  }
223 
224  if(unused_pt_bits)
225  {
226  cleartext[cleartext_len - 1] &= unused_pt_bits_mask;
227  }
228 
229  return cleartext;
230  }
