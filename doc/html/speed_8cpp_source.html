193  std::string print()
194  {
195  const size_t name_padding = 35;
196  const size_t op_name_padding = 16;
197  const size_t op_padding = 16;
198 
199  std::ostringstream result_ss;
200  result_ss << std::fixed;
201 
202  if(m_bps_entries.size() > 0)
203  {
204  result_ss << "\n";
205 
206  // add table header
207  result_ss << std::setw(name_padding) << std::left << "algo"
208  << std::setw(op_name_padding) << std::left << "operation";
209 
210  for(const Timer& t : m_bps_entries.begin()->second)
211  {
212  result_ss << std::setw(op_padding) << std::right << (std::to_string(t.buf_size()) + " bytes");
213  }
214  result_ss << "\n";
215 
216  // add table entries
217  for(const auto& entry : m_bps_entries)
218  {
219  if(entry.second.empty())
220  continue;
221 
222  result_ss << std::setw(name_padding) << std::left << (entry.first.second)
223  << std::setw(op_name_padding) << std::left << (entry.first.first);
224 
225  for(const Timer& t : entry.second)
226  {
227 
228  if(t.events() == 0)
229  {
230  result_ss << std::setw(op_padding) << std::right << "N/A";
231  }
232  else
233  {
234  result_ss << std::setw(op_padding) << std::right
235  << std::setprecision(2) << (t.bytes_per_second() / 1000.0);
236  }
237  }
238 
239  result_ss << "\n";
240  }
241 
242  result_ss << "\n[results are the number of 1000s bytes processed per second]\n";
243  }
244 
245  if(m_ops_entries.size() > 0)
246  {
247  result_ss << std::setprecision(6) << "\n";
248 
249  // sort entries
250  std::sort(m_ops_entries.begin(), m_ops_entries.end());
251 
252  // add table header
253  result_ss << std::setw(name_padding) << std::left << "algo"
254  << std::setw(op_name_padding) << std::left << "operation"
255  << std::setw(op_padding) << std::right << "sec/op"
256  << std::setw(op_padding) << std::right << "op/sec"
257  << "\n";
258 
259  // add table entries
260  for(const Timer& entry : m_ops_entries)
261  {
262  result_ss << std::setw(name_padding) << std::left << entry.get_name()
263  << std::setw(op_name_padding) << std::left << entry.doing()
264  << std::setw(op_padding) << std::right << entry.seconds_per_event()
265  << std::setw(op_padding) << std::right << entry.events_per_second()
266  << "\n";
267  }
268  }
269 
270  return result_ss.str();
271  }

278 std::vector<size_t> unique_buffer_sizes(const std::string& cmdline_arg)
279  {
280  const size_t MAX_BUF_SIZE = 64*1024*1024;
281 
282  std::set<size_t> buf;
283  for(std::string size_str : Botan::split_on(cmdline_arg, ','))
284  {
285  size_t x = 0;
286  try
287  {
288  size_t converted = 0;
289  x = static_cast<size_t>(std::stoul(size_str, &converted, 0));
290 
291  if(converted != size_str.size())
292  throw CLI_Usage_Error("Invalid integer");
293  }
294  catch(std::exception&)
295  {
296  throw CLI_Usage_Error("Invalid integer value '" + size_str + "' for option buf-size");
297  }
298 
299  if(x == 0)
300  throw CLI_Usage_Error("Cannot have a zero-sized buffer");
301 
302  if(x > MAX_BUF_SIZE)
303  throw CLI_Usage_Error("Specified buffer size is too large");
304 
305  buf.insert(x);
306  }
307 
308  return std::vector<size_t>(buf.begin(), buf.end());
309  }
