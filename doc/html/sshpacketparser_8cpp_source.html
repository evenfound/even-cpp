121 SshNameList SshPacketParser::asNameList(const QByteArray &data, quint32 *offset)
122 {
123  const quint32 length = asUint32(data, offset);
124  const int listEndPos = *offset + length;
125  if (data.size() < listEndPos)
126  throw SshPacketParseException();
127  SshNameList names(length + 4);
128  int nextNameOffset = *offset;
129  int nextCommaOffset = data.indexOf(',', nextNameOffset);
130  while (nextNameOffset > 0 && nextNameOffset < listEndPos) {
131  const int stringEndPos = nextCommaOffset == -1
132  || nextCommaOffset > listEndPos ? listEndPos : nextCommaOffset;
133  names.names << QByteArray(data.constData() + nextNameOffset,
134  stringEndPos - nextNameOffset);
135  nextNameOffset = nextCommaOffset + 1;
136  nextCommaOffset = data.indexOf(',', nextNameOffset);
137  }
138  *offset += length;
139  return names;
140 }
