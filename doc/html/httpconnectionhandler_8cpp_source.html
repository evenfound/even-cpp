284 void HttpConnectionHandler::readHttp()
285 {
286  // The loop adds support for HTTP pipelinig
287  while (socket->bytesAvailable())
288  {
289  #ifdef SUPERVERBOSE
290  qDebug("HttpConnectionHandler (%p): read input",this);
291  #endif
292 
293  // Create new HttpRequest object if necessary
294  if (!currentRequest)
295  {
296  currentRequest=new HttpRequest(settings);
297  }
298 
299  // Collect data for the request object
300  while (socket->bytesAvailable() && currentRequest->getStatus()!=HttpRequest::complete && currentRequest->getStatus()!=HttpRequest::abort)
301  {
302  currentRequest->readFromSocket(socket);
303  if (currentRequest->getStatus()==HttpRequest::waitForBody)
304  {
305  // Restart timer for read timeout, otherwise it would
306  // expire during large file uploads.
307  int readTimeout=settings->value("readTimeout",10000).toInt();
308  readTimer.start(readTimeout);
309  }
310  }
311 
312  // If the request is aborted, return error message and close the connection
313  if (currentRequest->getStatus()==HttpRequest::abort)
314  {
315  socket->write("HTTP/1.1 413 entity too large\r\nConnection: close\r\n\r\n413 Entity too large\r\n");
316  socket->flush();
317  socket->disconnectFromHost();
318  delete currentRequest;
319  currentRequest=0;
320  return;
321  }
322 
323  // If the request is complete, let the request mapper dispatch it
324  if (currentRequest->getStatus()==HttpRequest::complete)
325  {
326  readTimer.stop();
327  qDebug("HttpConnectionHandler (%p): received request",this);
328 
329  // Copy the Connection:close header to the response
330  HttpResponse response(socket);
331  bool closeConnection=QString::compare(currentRequest->getHeader("Connection"),"close",Qt::CaseInsensitive)==0;
332  if (closeConnection)
333  {
334  response.setHeader("Connection","close");
335  }
336 
337  // In case of HTTP 1.0 protocol add the Connection:close header.
338  // This ensures that the HttpResponse does not activate chunked mode, which is not spported by HTTP 1.0.
339  else
340  {
341  bool http1_0=QString::compare(currentRequest->getVersion(),"HTTP/1.0",Qt::CaseInsensitive)==0;
342  if (http1_0)
343  {
344  closeConnection=true;
345  response.setHeader("Connection","close");
346  }
347  }
348 
349  // Call the request mapper
350  try
351  {
352  requestHandler->service(*currentRequest, response);
353  }
354  catch (...)
355  {
356  qCritical("HttpConnectionHandler (%p): An uncatched exception occured in the request handler",this);
357  }
358 
359  // Finalize sending the response if not already done
360  if (!response.hasSentLastPart())
361  {
362  response.write(QByteArray(),true);
363  }
364 
365  qDebug("HttpConnectionHandler (%p): finished request",this);
366 
367  // Find out whether the connection must be closed
368  if (!closeConnection)
369  {
370  // Maybe the request handler or mapper added a Connection:close header in the meantime
371  bool closeResponse=QString::compare(response.getHeaders().value("Connection"),"close",Qt::CaseInsensitive)==0;
372  if (closeResponse==true)
373  {
374  closeConnection=true;
375  }
376  else
377  {
378  // If we have no Content-Length header and did not use chunked mode, then we have to close the
379  // connection to tell the HTTP client that the end of the response has been reached.
380  bool hasContentLength=response.getHeaders().contains("Content-Length");
381  if (!hasContentLength)
382  {
383  bool hasChunkedMode=QString::compare(response.getHeaders().value("Transfer-Encoding"),"chunked",Qt::CaseInsensitive)==0;
384  if (!hasChunkedMode)
385  {
386  closeConnection=true;
387  }
388  }
389  }
390  }
391 
392  // Close the connection or prepare for the next request on the same connection.
393  if (closeConnection)
394  {
395  socket->flush();
396  socket->disconnectFromHost();
397  }
398  else
399  {
400  // Start timer for next request
401  int readTimeout=settings->value("readTimeout",10000).toInt();
402  readTimer.start(readTimeout);
403  }
404  delete currentRequest;
405  currentRequest=0;
406  }
407  }
