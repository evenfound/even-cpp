300 QWebSocket *QWebSocketPrivate::upgradeFrom(QTcpSocket *pTcpSocket,
301  const QWebSocketHandshakeRequest &request,
302  const QWebSocketHandshakeResponse &response,
303  QObject *parent)
304 {
305  QWebSocket *pWebSocket = new QWebSocket(pTcpSocket, response.acceptedVersion(), parent);
306  if (Q_LIKELY(pWebSocket)) {
307  QNetworkRequest netRequest(request.requestUrl());
308  const auto headers = request.headers();
309  for (auto it = headers.begin(), end = headers.end(); it != end; ++it)
310  netRequest.setRawHeader(it.key().toLatin1(), it.value().toLatin1());
311 #ifndef QT_NO_SSL
312  if (QSslSocket *sslSock = qobject_cast<QSslSocket *>(pTcpSocket))
313  pWebSocket->setSslConfiguration(sslSock->sslConfiguration());
314 #endif
315  pWebSocket->d_func()->setExtension(response.acceptedExtension());
316  pWebSocket->d_func()->setOrigin(request.origin());
317  pWebSocket->d_func()->setRequest(netRequest);
318  pWebSocket->d_func()->setProtocol(response.acceptedProtocol());
319  pWebSocket->d_func()->setResourceName(request.requestUrl().toString(QUrl::RemoveUserInfo));
320  //a server should not send masked frames
321  pWebSocket->d_func()->enableMasking(false);
322  }
323 
324  return pWebSocket;
325 }

330 void QWebSocketPrivate::close(QWebSocketProtocol::CloseCode closeCode, QString reason)
331 {
332  if (Q_UNLIKELY(!m_pSocket))
333  return;
334  if (!m_isClosingHandshakeSent) {
335  Q_Q(QWebSocket);
336  m_closeCode = closeCode;
337  m_closeReason = reason;
338  const quint16 code = qToBigEndian<quint16>(closeCode);
339  QByteArray payload;
340  payload.append(static_cast<const char *>(static_cast<const void *>(&code)), 2);
341  if (!reason.isEmpty())
342  payload.append(reason.toUtf8());
343  quint32 maskingKey = 0;
344  if (m_mustMask) {
345  maskingKey = generateMaskingKey();
346  QWebSocketProtocol::mask(payload.data(), payload.size(), maskingKey);
347  }
348  QByteArray frame = getFrameHeader(QWebSocketProtocol::OpCodeClose,
349  payload.size(), maskingKey, true);
350  frame.append(payload);
351  m_pSocket->write(frame);
352  m_pSocket->flush();
353 
354  m_isClosingHandshakeSent = true;
355 
356  Q_EMIT q->aboutToClose();
357  }
358  m_pSocket->close();
359 }

364 void QWebSocketPrivate::open(const QNetworkRequest &request, bool mask)
365 {
366  //just delete the old socket for the moment;
367  //later, we can add more 'intelligent' handling by looking at the URL
368 
369  Q_Q(QWebSocket);
370  QUrl url = request.url();
371  if (!url.isValid() || url.toString().contains(QStringLiteral("\r\n"))) {
372  setErrorString(QWebSocket::tr("Invalid URL."));
373  Q_EMIT q->error(QAbstractSocket::ConnectionRefusedError);
374  return;
375  }
376  if (m_pSocket) {
377  releaseConnections(m_pSocket);
378  m_pSocket->deleteLater();
379  m_pSocket = Q_NULLPTR;
380  }
381  //if (m_url != url)
382  if (Q_LIKELY(!m_pSocket)) {
383  m_dataProcessor.clear();
384  m_isClosingHandshakeReceived = false;
385  m_isClosingHandshakeSent = false;
386 
387  setRequest(request);
388  QString resourceName = url.path(QUrl::FullyEncoded);
389  // Check for encoded \r\n
390  if (resourceName.contains(QStringLiteral("%0D%0A"))) {
391  setRequest(QNetworkRequest()); //clear request
392  setErrorString(QWebSocket::tr("Invalid resource name."));
393  Q_EMIT q->error(QAbstractSocket::ConnectionRefusedError);
394  return;
395  }
396  if (!url.query().isEmpty()) {
397  if (!resourceName.endsWith(QChar::fromLatin1('?'))) {
398  resourceName.append(QChar::fromLatin1('?'));
399  }
400  resourceName.append(url.query(QUrl::FullyEncoded));
401  }
402  if (resourceName.isEmpty())
403  resourceName = QStringLiteral("/");
404  setResourceName(resourceName);
405  enableMasking(mask);
406 
407  #ifndef QT_NO_SSL
408  if (url.scheme() == QStringLiteral("wss")) {
409  if (!QSslSocket::supportsSsl()) {
410  const QString message =
411  QWebSocket::tr("SSL Sockets are not supported on this platform.");
412  setErrorString(message);
413  Q_EMIT q->error(QAbstractSocket::UnsupportedSocketOperationError);
414  } else {
415  QSslSocket *sslSocket = new QSslSocket(q_ptr);
416  m_pSocket = sslSocket;
417  if (Q_LIKELY(m_pSocket)) {
418  m_pSocket->setSocketOption(QAbstractSocket::LowDelayOption, 1);
419  m_pSocket->setSocketOption(QAbstractSocket::KeepAliveOption, 1);
420  m_pSocket->setReadBufferSize(m_readBufferSize);
421  m_pSocket->setPauseMode(m_pauseMode);
422 
423  makeConnections(m_pSocket);
424  setSocketState(QAbstractSocket::ConnectingState);
425 
426  sslSocket->setSslConfiguration(m_configuration.m_sslConfiguration);
427  if (Q_UNLIKELY(m_configuration.m_ignoreSslErrors))
428  sslSocket->ignoreSslErrors();
429  else
430  sslSocket->ignoreSslErrors(m_configuration.m_ignoredSslErrors);
431  #ifndef QT_NO_NETWORKPROXY
432  sslSocket->setProxy(m_configuration.m_proxy);
433  #endif
434  sslSocket->connectToHostEncrypted(url.host(), url.port(443));
435  } else {
436  const QString message = QWebSocket::tr("Out of memory.");
437  setErrorString(message);
438  Q_EMIT q->error(QAbstractSocket::SocketResourceError);
439  }
440  }
441  } else
442  #endif
443  if (url.scheme() == QStringLiteral("ws")) {
444  m_pSocket = new QTcpSocket(q_ptr);
445  if (Q_LIKELY(m_pSocket)) {
446  m_pSocket->setSocketOption(QAbstractSocket::LowDelayOption, 1);
447  m_pSocket->setSocketOption(QAbstractSocket::KeepAliveOption, 1);
448  m_pSocket->setReadBufferSize(m_readBufferSize);
449  m_pSocket->setPauseMode(m_pauseMode);
450 
451  makeConnections(m_pSocket);
452  setSocketState(QAbstractSocket::ConnectingState);
453  #ifndef QT_NO_NETWORKPROXY
454  m_pSocket->setProxy(m_configuration.m_proxy);
455  #endif
456  m_pSocket->connectToHost(url.host(), url.port(80));
457  } else {
458  const QString message = QWebSocket::tr("Out of memory.");
459  setErrorString(message);
460  Q_EMIT q->error(QAbstractSocket::SocketResourceError);
461  }
462  } else {
463  const QString message =
464  QWebSocket::tr("Unsupported WebSocket scheme: %1").arg(url.scheme());
465  setErrorString(message);
466  Q_EMIT q->error(QAbstractSocket::UnsupportedSocketOperationError);
467  }
468  }

474 void QWebSocketPrivate::ping(const QByteArray &payload)
475 {
476  QByteArray payloadTruncated = payload.left(125);

482  maskingKey, true);
483  if (m_mustMask)
484  QWebSocketProtocol::mask(&payloadTruncated, maskingKey);
485  pingFrame.append(payloadTruncated);
486  qint64 ret = writeFrame(pingFrame);
487  Q_UNUSED(ret);

578  QObject::connect(pTcpSocket, &QAbstractSocket::bytesWritten, q, &QWebSocket::bytesWritten);
579

582  this, &QWebSocketPrivate::socketDestroyed);
583 
584  //catch signals
585  QObjectPrivate::connect(pTcpSocket, &QAbstractSocket::stateChanged, this,
586  &QWebSocketPrivate::processStateChanged);

707 QByteArray QWebSocketPrivate::getFrameHeader(QWebSocketProtocol::OpCode opCode,
708  quint64 payloadLength, quint32 maskingKey,
709  bool lastFrame)
710 {
711  QByteArray header;
712  bool ok = payloadLength <= 0x7FFFFFFFFFFFFFFFULL;
713 
714  if (Q_LIKELY(ok)) {
715  //FIN, RSV1-3, opcode (RSV-1, RSV-2 and RSV-3 are zero)
716  quint8 byte = static_cast<quint8>((opCode & 0x0F) | (lastFrame ? 0x80 : 0x00));
717  header.append(static_cast<char>(byte));
718 
719  byte = 0x00;
720  if (maskingKey != 0)
721  byte |= 0x80;
722  if (payloadLength <= 125) {
723  byte |= static_cast<quint8>(payloadLength);
724  header.append(static_cast<char>(byte));
725  } else if (payloadLength <= 0xFFFFU) {
726  byte |= 126;
727  header.append(static_cast<char>(byte));
728  quint16 swapped = qToBigEndian<quint16>(static_cast<quint16>(payloadLength));
729  header.append(static_cast<const char *>(static_cast<const void *>(&swapped)), 2);
730  } else if (payloadLength <= 0x7FFFFFFFFFFFFFFFULL) {
731  byte |= 127;
732  header.append(static_cast<char>(byte));
733  quint64 swapped = qToBigEndian<quint64>(payloadLength);
734  header.append(static_cast<const char *>(static_cast<const void *>(&swapped)), 8);
735  }
736 
737  if (maskingKey != 0) {
738  const quint32 mask = qToBigEndian<quint32>(maskingKey);
739  header.append(static_cast<const char *>(static_cast<const void *>(&mask)),
740  sizeof(quint32));
741  }
742  } else {
743  setErrorString(QStringLiteral("WebSocket::getHeader: payload too big!"));
744  Q_EMIT q_ptr->error(QAbstractSocket::DatagramTooLargeError);
745  }
746 
747  return header;
748 }

753 qint64 QWebSocketPrivate::doWriteFrames(const QByteArray &data, bool isBinary)
754 {
755  qint64 payloadWritten = 0;
756  if (Q_UNLIKELY(!m_pSocket) || (state() != QAbstractSocket::ConnectedState))
757  return payloadWritten;
758 
759  Q_Q(QWebSocket);
760  const QWebSocketProtocol::OpCode firstOpCode = isBinary ?
761  QWebSocketProtocol::OpCodeBinary : QWebSocketProtocol::OpCodeText;
762 
763  int numFrames = data.size() / FRAME_SIZE_IN_BYTES;
764  QByteArray tmpData(data);
765  tmpData.detach();
766  char *payload = tmpData.data();
767  quint64 sizeLeft = quint64(data.size()) % FRAME_SIZE_IN_BYTES;
768  if (Q_LIKELY(sizeLeft))
769  ++numFrames;
770 
771  //catch the case where the payload is zero bytes;
772  //in this case, we still need to send a frame
773  if (Q_UNLIKELY(numFrames == 0))
774  numFrames = 1;
775  quint64 currentPosition = 0;
776  quint64 bytesLeft = data.size();
777 
778  for (int i = 0; i < numFrames; ++i) {
779  quint32 maskingKey = 0;
780  if (m_mustMask)
781  maskingKey = generateMaskingKey();
782 
783  const bool isLastFrame = (i == (numFrames - 1));
784  const bool isFirstFrame = (i == 0);
785 
786  const quint64 size = qMin(bytesLeft, FRAME_SIZE_IN_BYTES);
787  const QWebSocketProtocol::OpCode opcode = isFirstFrame ? firstOpCode
788  : QWebSocketProtocol::OpCodeContinue;
789 
790  //write header
791  m_pSocket->write(getFrameHeader(opcode, size, maskingKey, isLastFrame));
792 
793  //write payload
794  if (Q_LIKELY(size > 0)) {
795  char *currentData = payload + currentPosition;
796  if (m_mustMask)
797  QWebSocketProtocol::mask(currentData, size, maskingKey);
798  qint64 written = m_pSocket->write(currentData, static_cast<qint64>(size));
799  if (Q_LIKELY(written > 0)) {
800  payloadWritten += written;
801  } else {
802  m_pSocket->flush();
803  setErrorString(QWebSocket::tr("Error writing bytes to socket: %1.")
804  .arg(m_pSocket->errorString()));
805  Q_EMIT q->error(QAbstractSocket::NetworkError);
806  break;
807  }
808  }
809  currentPosition += size;
810  bytesLeft -= size;
811  }
812  if (Q_UNLIKELY(payloadWritten != data.size())) {
813  setErrorString(QWebSocket::tr("Bytes written %1 != %2.")
814  .arg(payloadWritten).arg(data.size()));
815  Q_EMIT q->error(QAbstractSocket::NetworkError);
816  }
817  return payloadWritten;
818 }

898 // this function is a copy of QHttpNetworkReplyPrivate::parseStatus
899 static bool parseStatusLine(const QByteArray &status, int *majorVersion, int *minorVersion,
900  int *statusCode, QString *reasonPhrase)
901 {
902  // from RFC 2616:
903  // Status-Line = HTTP-Version SP Status-Code SP Reason-Phrase CRLF
904  // HTTP-Version = "HTTP" "/" 1*DIGIT "." 1*DIGIT
905  // that makes: 'HTTP/n.n xxx Message'
906  // byte count: 0123456789012
907 
908  static const int minLength = 11;
909  static const int dotPos = 6;
910  static const int spacePos = 8;
911  static const char httpMagic[] = "HTTP/";
912 
913  if (status.length() < minLength
914  || !status.startsWith(httpMagic)
915  || status.at(dotPos) != '.'
916  || status.at(spacePos) != ' ') {
917  // I don't know how to parse this status line
918  return false;
919  }
920 
921  // optimize for the valid case: defer checking until the end
922  *majorVersion = status.at(dotPos - 1) - '0';
923  *minorVersion = status.at(dotPos + 1) - '0';
924 
925  int i = spacePos;
926  int j = status.indexOf(' ', i + 1); // j == -1 || at(j) == ' ' so j+1 == 0 && j+1 <= length()
927  const QByteArray code = status.mid(i + 1, j - i - 1);
928 
929  bool ok;
930  *statusCode = code.toInt(&ok);
931  *reasonPhrase = QString::fromLatin1(status.constData() + j + 1);
932 
933  return ok && uint(*majorVersion) <= 9 && uint(* minorVersion) <= 9;
934 }

937 //called on the client for a server handshake response
941 void QWebSocketPrivate::processHandshake(QTcpSocket *pSocket)
942 {
943  Q_Q(QWebSocket);
944  if (Q_UNLIKELY(!pSocket))
945  return;
946  // Reset handshake on a new connection.
947  if (m_handshakeState == AllDoneState)
948  m_handshakeState = NothingDoneState;
949 
950  QString errorDescription;
951 
952  switch (m_handshakeState) {
953  case NothingDoneState:
954  m_headers.clear();
955  m_handshakeState = ReadingStatusState;
956  // no break
957  case ReadingStatusState:
958  if (!pSocket->canReadLine())
959  return;
960  m_statusLine = pSocket->readLine();
961  if (Q_UNLIKELY(!parseStatusLine(m_statusLine, &m_httpMajorVersion, &m_httpMinorVersion, &m_httpStatusCode, &m_httpStatusMessage))) {
962  errorDescription = QWebSocket::tr("Invalid statusline in response: %1.").arg(QString::fromLatin1(m_statusLine));
963  break;
964  }
965  m_handshakeState = ReadingHeaderState;
966  // no break
967  case ReadingHeaderState:
968  while (pSocket->canReadLine()) {
969  QString headerLine = readLine(pSocket);
970  const QStringList headerField = headerLine.split(QStringLiteral(": "),
971  QString::SkipEmptyParts);
972  if (headerField.size() == 2) {
973  m_headers.insertMulti(headerField[0].toLower(), headerField[1]);
974  }
975  if (headerField.isEmpty()) {
976  m_handshakeState = ParsingHeaderState;
977  break;
978  }
979  }
980 
981  if (m_handshakeState != ParsingHeaderState) {
982  if (pSocket->state() != QAbstractSocket::ConnectedState) {
983  errorDescription = QWebSocket::tr("QWebSocketPrivate::processHandshake: Connection closed while reading header.");
984  break;
985  }
986  return;
987  }
988  // no break
989  case ParsingHeaderState: {
990  const QString acceptKey = m_headers.value(QStringLiteral("sec-websocket-accept"), QString());
991  const QString upgrade = m_headers.value(QStringLiteral("upgrade"), QString());
992  const QString connection = m_headers.value(QStringLiteral("connection"), QString());
993 // unused for the moment
994 // const QString extensions = m_headers.value(QStringLiteral("sec-websocket-extensions"),
995 // QString());
996 // const QString protocol = m_headers.value(QStringLiteral("sec-websocket-protocol"),
997 // QString());
998  const QString version = m_headers.value(QStringLiteral("sec-websocket-version"), QString());
999 
1000  bool ok = false;
1001  if (Q_LIKELY(m_httpStatusCode == 101)) {
1002  //HTTP/x.y 101 Switching Protocols
1003  //TODO: do not check the httpStatusText right now
1004  ok = !(acceptKey.isEmpty() ||
1005  (m_httpMajorVersion < 1 || m_httpMinorVersion < 1) ||
1006  (upgrade.toLower() != QStringLiteral("websocket")) ||
1007  (connection.toLower() != QStringLiteral("upgrade")));
1008  if (ok) {
1009  const QString accept = calculateAcceptKey(m_key);
1010  ok = (accept == acceptKey);
1011  if (!ok)
1012  errorDescription =
1013  QWebSocket::tr("Accept-Key received from server %1 does not match the client key %2.")
1014  .arg(acceptKey).arg(accept);
1015  } else {
1016  errorDescription =
1017  QWebSocket::tr("QWebSocketPrivate::processHandshake: Invalid statusline in response: %1.")
1018  .arg(QString::fromLatin1(m_statusLine));
1019  }
1020  } else if (m_httpStatusCode == 400) {
1021  //HTTP/1.1 400 Bad Request
1022  if (!version.isEmpty()) {
1023  const QStringList versions = version.split(QStringLiteral(", "),
1024  QString::SkipEmptyParts);
1025  if (!versions.contains(QString::number(QWebSocketProtocol::currentVersion()))) {
1026  //if needed to switch protocol version, then we are finished here
1027  //because we cannot handle other protocols than the RFC one (v13)
1028  errorDescription =
1029  QWebSocket::tr("Handshake: Server requests a version that we don't support: %1.")
1030  .arg(versions.join(QStringLiteral(", ")));
1031  } else {
1032  //we tried v13, but something different went wrong
1033  errorDescription =
1034  QWebSocket::tr("QWebSocketPrivate::processHandshake: Unknown error condition encountered. Aborting connection.");
1035  }
1036  } else {
1037  errorDescription =
1038  QWebSocket::tr("QWebSocketPrivate::processHandshake: Unknown error condition encountered. Aborting connection.");
1039  }
1040  } else {
1041  errorDescription =
1042  QWebSocket::tr("QWebSocketPrivate::processHandshake: Unhandled http status code: %1 (%2).")
1043  .arg(m_httpStatusCode).arg(m_httpStatusMessage);
1044  }
1045  if (ok)
1046  m_handshakeState = AllDoneState;
1047  break;
1048  }
1049  case AllDoneState:
1050  Q_UNREACHABLE();
1051  break;
1052  }
1053 
1054  if (m_handshakeState == AllDoneState) {
1055  // handshake succeeded
1056  setSocketState(QAbstractSocket::ConnectedState);
1057  Q_EMIT q->connected();
1058  } else {
1059  // handshake failed
1060  m_handshakeState = AllDoneState;
1061  setErrorString(errorDescription);
1062  Q_EMIT q->error(QAbstractSocket::ConnectionRefusedError);
1063  }
1064 }
