51  {
52  Test::Result result("BER nested indefinite length");
53 
54  // OSS-Fuzz #4353
55 
56  Botan::ASN1_Pretty_Printer printer;
57 
58  size_t max_eoc_allowed = 0;
59 
60  for(size_t len = 1; len < 1024; ++len)
61  {
62  std::vector<uint8_t> buf(4*len);
63 
64  /*
65  This constructs a len deep sequence of SEQUENCES each with
66  an indefinite length
67  */
68  for(size_t i = 0; i != 2*len; i += 2)
69  {
70  buf[i ] = 0x30;
71  buf[i+1] = 0x80;
72  }
73  // remainder of values left as zeros (EOC markers)
74 
75  try
76  {
77  printer.print(buf);
78  }
79  catch(Botan::BER_Decoding_Error&)
80  {
81  max_eoc_allowed = len - 1;
82  break;
83  }
84  }
