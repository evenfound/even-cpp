2 Package dnslink implements a dns link resolver. dnslink is a basic
3 standard for placing traversable links in dns itself. See dnslink.info
4 
5 A dnslink is a path link in a dns TXT record, like this:
6 
7  dnslink=/ipfs/QmR7tiySn6vFHcEjBeZNtYGAFh735PJHfEMdVEycj9jAPy
8 
9 For example:
10 
11  > dig TXT ipfs.io
12  ipfs.io. 120 IN TXT dnslink=/ipfs/QmR7tiySn6vFHcEjBeZNtYGAFh735PJHfEMdVEycj9jAPy
13 
14 This package eases resolving and working with thse dns links. For example:
15 
16  import (
17  dnslink "github.com/jbenet/go-dnslink"
18  )
19 
20  link, err := dnslink.Resolve("ipfs.io")
21  // link = "/ipfs/QmR7tiySn6vFHcEjBeZNtYGAFh735PJHfEMdVEycj9jAPy"
22 
23 It even supports recursive resolution. Suppose you have three domains with
24 dnslink records like these:
25 
26  > dig TXT foo.com
27  foo.com. 120 IN TXT dnslink=/dns/bar.com/f/o/o
28  > dig TXT bar.com
29  bar.com. 120 IN TXT dnslink=/dns/long.test.baz.it/b/a/r
30  > dig TXT long.test.baz.it
31  long.test.baz.it. 120 IN TXT dnslink=/b/a/z
32 
33 Expect these resolutions:
34 
35  dnslink.ResolveN("long.test.baz.it", 0) // "/dns/long.test.baz.it"
36  dnslink.Resolve("long.test.baz.it") // "/b/a/z"
37 
38  dnslink.ResolveN("bar.com", 1) // "/dns/long.test.baz.it/b/a/r"
39  dnslink.Resolve("bar.com") // "/b/a/z/b/a/r"
40 
41  dnslink.ResolveN("foo.com", 1) // "/dns/bar.com/f/o/o/"
42  dnslink.ResolveN("foo.com", 2) // "/dns/long.test.baz.it/b/a/r/f/o/o/"
43  dnslink.Resolve("foo.com") // "/b/a/z/b/a/r/f/o/o"
44 
45 */

241  int ipfs_dnslink_resolv_lookupTXT(char ***txt, char *domain)
242  {
243  char buf[4096], *p;
244  int responseLength;
245  int i, l, n = 0;
246  ns_msg query_parse_msg;
247  ns_rr query_parse_rr;
248  u_char responseByte[4096];
249 
250  // Use res_query from libresolv to retrieve TXT record from DNS server.
251  if ((responseLength = res_query(domain,ns_c_in,ns_t_txt,responseByte,sizeof(responseByte))) < 0 ||
252  ns_initparse(responseByte,responseLength,&query_parse_msg) < 0) {
253  return ErrResolveFailed;
254  } else {
255  l = sizeof (buf);
256  buf[--l] = '\0';
257  p = buf;
258  // save every TXT record to buffer separating with a \0
259  for (i=0 ; i < ns_msg_count(query_parse_msg,ns_s_an) ; i++) {
260  if (ns_parserr(&query_parse_msg,ns_s_an,i,&query_parse_rr)) {
261  return ErrResolveFailed;
262  } else {
263  const unsigned char *rdata = ns_rr_rdata(query_parse_rr);
264  memcpy(p, rdata+1, *rdata); // first byte is record length
265  p += *rdata; // update pointer
266  *p++ = '\0'; // mark end-of-record and update pointer to next record.
267  n++; // update record count
268  }
269  }
270  // allocate array for all records + NULL pointer terminator.
271  *txt = calloc(n+1, sizeof(void*));
272  if (!*txt) {
273  return ErrAllocFailed;
274  }
275  l = p - buf; // length of all records in buffer.
276  p = malloc(l); // allocate memory that will be used as string data at *txt array.
277  if (!p) {
278  free(*txt);
279  *txt = NULL;
280  return ErrAllocFailed;
281  }
282  memcpy(p, buf, l); // transfer from buffer to allocated memory.
283  for (i = 0 ; i < n ; i++) {
284  *txt[i] = p; // save position of current record at *txt array.
285  p = memchr(p, '\0', l - (p - *txt[0])) + 1; // find next record position after next \0
286  }
287  }
288  return 0;
289  }

331 // ipfs_dnslink_parse_txt parses a TXT record value for a dnslink value.
332 // The TXT record must follow the dnslink format:
333 // TXT dnslink=<path>
334 // TXT dnslink=/foo/bar/baz
335 // ipfs_dnslink_parse_txt will return ErrInvalidDNSLink if parsing fails.
336 int ipfs_dnslink_parse_txt (char **path, char *txt)
337 {
338  char **parts;
339 
340  if (!path || !txt) {
341  return ErrInvalidParam;
342  }
343  parts = ipfs_path_split_n (txt, "=", 2);
344  if (!parts) {
345  return ErrAllocFailed;
346  }
347  if (ipfs_path_segments_length (parts) == 2 && strcmp(parts[0], "dnslink")==0 && memcmp(parts[1], "/", 1)==0) {
348  *path = ipfs_path_clean_path(parts[1]);
349  if (path) {
350  ipfs_path_free_segments (&parts);
351  return 0;
352  }
353  }
354  ipfs_path_free_segments (&parts);
355  *path = NULL;
356  return ErrInvalidDNSLink;
357 }
