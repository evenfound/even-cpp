90  Test::Result run_one_test(const std::string& hash_algo, const VarMap& vars) override
91  {
92  Test::Result result("TOTP " + hash_algo);
93 
94  std::unique_ptr<Botan::HashFunction> hash_test = Botan::HashFunction::create(hash_algo);
95  if(!hash_test)
96  return {result};
97 
98  const std::vector<uint8_t> key = vars.get_req_bin("Key");
99  const size_t otp = vars.get_req_sz("OTP");
100  const size_t digits = vars.get_req_sz("Digits");
101  const size_t timestep = vars.get_req_sz("Timestep");
102  const std::string timestamp = vars.get_req_str("Timestamp");
103 
104  Botan::TOTP totp(key, hash_algo, digits, timestep);
105 
106  std::chrono::system_clock::time_point time = from_timestring(timestamp);
107  std::chrono::system_clock::time_point later_time = time + std::chrono::seconds(timestep);
108  std::chrono::system_clock::time_point too_late = time + std::chrono::seconds(2*timestep);
109 
110  result.test_eq("TOTP generate", totp.generate_totp(time), otp);
111 
112  result.test_eq("TOTP verify valid", totp.verify_totp(otp, time, 0), true);
113  result.test_eq("TOTP verify invalid", totp.verify_totp(otp ^ 1, time, 0), false);
114  result.test_eq("TOTP verify time slip", totp.verify_totp(otp, later_time, 0), false);
115  result.test_eq("TOTP verify time slip allowed", totp.verify_totp(otp, later_time, 1), true);
116  result.test_eq("TOTP verify time slip out of range", totp.verify_totp(otp, too_late, 1), false);
117 
118  return result;
119  }
120 
121  private:
122  std::chrono::system_clock::time_point from_timestring(const std::string& time_str)
123  {
124  if(time_str.size() != 19)
125  throw std::invalid_argument("Invalid TOTP timestamp string " + time_str);
126  // YYYY-MM-DDTHH:MM:SS
127  // 0123456789012345678
128  const uint32_t year = Botan::to_u32bit(time_str.substr(0, 4));
129  const uint32_t month = Botan::to_u32bit(time_str.substr(5, 2));
130  const uint32_t day = Botan::to_u32bit(time_str.substr(8, 2));
131  const uint32_t hour = Botan::to_u32bit(time_str.substr(11, 2));
132  const uint32_t minute = Botan::to_u32bit(time_str.substr(14, 2));
133  const uint32_t second = Botan::to_u32bit(time_str.substr(17, 2));
134  return Botan::calendar_point(year, month, day, hour, minute, second).to_std_timepoint();
135  }
