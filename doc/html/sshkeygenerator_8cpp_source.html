60 bool SshKeyGenerator::generateKeys(KeyType type, PrivateKeyFormat format, int keySize,
61  EncryptionMode encryptionMode)
62 {
63  m_type = type;
64  m_encryptionMode = encryptionMode;
65 
66  try {
67  AutoSeeded_RNG rng;
68  KeyPtr key;
69  if (m_type == Rsa)
70  key = KeyPtr(new RSA_PrivateKey(rng, keySize));
71  else
72  key = KeyPtr(new DSA_PrivateKey(rng, DL_Group(rng, DL_Group::DSA_Kosherizer, keySize)));
73  switch (format) {
74  case Pkcs8:
75  generatePkcs8KeyStrings(key, rng);
76  break;
77  case OpenSsl:
78  generateOpenSslKeyStrings(key);
79  break;
80  case Mixed:
81  default:
82  generatePkcs8KeyString(key, true, rng);
83  generateOpenSslPublicKeyString(key);
84  }
85  return true;
86  } catch (Botan::Exception &e) {
87  m_error = tr("Error generating key: %1").arg(QString::fromAscii(e.what()));
88  return false;
89  }
90 }

98 void SshKeyGenerator::generatePkcs8KeyString(const KeyPtr &key, bool privateKey,
99  Botan::RandomNumberGenerator &rng)
100 {
101  Pipe pipe;
102  pipe.start_msg();
103  QByteArray *keyData;
104  if (privateKey) {
105  QString password;
106  if (m_encryptionMode == DoOfferEncryption)
107  password = getPassword();
108  if (!password.isEmpty())
109  pipe.write(PKCS8::PEM_encode(*key, rng, password.toLocal8Bit().data()));
110  else
111  pipe.write(PKCS8::PEM_encode(*key));
112  keyData = &m_privateKey;
113  } else {
114  pipe.write(X509::PEM_encode(*key));
115  keyData = &m_publicKey;
116  }
117  pipe.end_msg();
118  keyData->resize(pipe.remaining(pipe.message_count() - 1));
119  pipe.read(convertByteArray(*keyData), keyData->size(),
120  pipe.message_count() - 1);
121 }

129 void SshKeyGenerator::generateOpenSslPublicKeyString(const KeyPtr &key)
130 {
131  QList<BigInt> params;
132  QByteArray keyId;
133  if (m_type == Rsa) {
134  const QSharedPointer<RSA_PrivateKey> rsaKey = key.dynamicCast<RSA_PrivateKey>();
135  params << rsaKey->get_e() << rsaKey->get_n();
136  keyId = SshCapabilities::PubKeyRsa;
137  } else {
138  const QSharedPointer<DSA_PrivateKey> dsaKey = key.dynamicCast<DSA_PrivateKey>();
139  params << dsaKey->group_p() << dsaKey->group_q() << dsaKey->group_g() << dsaKey->get_y();
140  keyId = SshCapabilities::PubKeyDss;
141  }
142 
143  QByteArray publicKeyBlob = AbstractSshPacket::encodeString(keyId);
144  foreach (const BigInt &b, params)
145  publicKeyBlob += AbstractSshPacket::encodeMpInt(b);
146  publicKeyBlob = publicKeyBlob.toBase64();
147  const QByteArray id = "QtCreator/"
148  + QDateTime::currentDateTime().toString(Qt::ISODate).toUtf8();
149  m_publicKey = keyId + ' ' + publicKeyBlob + ' ' + id;
150 }

152 void SshKeyGenerator::generateOpenSslPrivateKeyString(const KeyPtr &key)
153 {
154  QList<BigInt> params;
155  QByteArray keyId;
156  const char *label;
157  if (m_type == Rsa) {
158  const QSharedPointer<RSA_PrivateKey> rsaKey
159  = key.dynamicCast<RSA_PrivateKey>();
160  params << rsaKey->get_n() << rsaKey->get_e() << rsaKey->get_d() << rsaKey->get_p()
161  << rsaKey->get_q();
162  keyId = SshCapabilities::PubKeyRsa;
163  label = "RSA PRIVATE KEY";
164  } else {
165  const QSharedPointer<DSA_PrivateKey> dsaKey = key.dynamicCast<DSA_PrivateKey>();
166  params << dsaKey->group_p() << dsaKey->group_q() << dsaKey->group_g() << dsaKey->get_y()
167  << dsaKey->get_x();
168  keyId = SshCapabilities::PubKeyDss;
169  label = "DSA PRIVATE KEY";
170  }
171 
172  DER_Encoder encoder;
173  encoder.start_cons(SEQUENCE).encode(size_t(0));
174  foreach (const BigInt &b, params)
175  encoder.encode(b);
176  encoder.end_cons();
177  m_privateKey = QByteArray(PEM_Code::encode (encoder.get_contents(), label).c_str());
178 }
