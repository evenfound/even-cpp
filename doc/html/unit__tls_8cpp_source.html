490 void TLS_Handshake_Test::go()
491  {
492  m_results.start_timer();
493 
494  Botan::RandomNumberGenerator& rng = Test::rng();
495 
496  const std::vector<std::string> protocols_offered = { "test/1", "test/2" };
497 
498  // Choose random application data to send
499  const size_t c_len = 1 + ((static_cast<size_t>(rng.next_byte()) << 4) ^ rng.next_byte());
500  std::vector<uint8_t> client_msg(c_len);
501  Test::rng().randomize(client_msg.data(), client_msg.size());
502  bool client_has_written = false;
503 
504  const size_t s_len = 1 + ((static_cast<size_t>(rng.next_byte()) << 4) ^ rng.next_byte());
505  std::vector<uint8_t> server_msg(s_len);
506  Test::rng().randomize(server_msg.data(), server_msg.size());
507  bool server_has_written = false;
508 
509  std::unique_ptr<Botan::TLS::Client> client;
510  client.reset(
511  new Botan::TLS::Client(*m_client_cb, m_client_sessions, m_creds,
512  m_client_policy, m_rng,
513  Botan::TLS::Server_Information("server.example.com"),
514  m_offer_version,
515  protocols_offered));
516 
517  size_t rounds = 0;
518 
519  bool client_handshake_completed = false;
520  bool server_handshake_completed = false;
521 
522  while(true)
523  {
524  ++rounds;
525 
526  if(rounds > 25)
527  {
528  m_results.test_failure("Still here after many rounds, deadlock?");
529  break;
530  }
531 
532  if(client_handshake_completed == false && client->is_active())
533  client_handshake_completed = true;
534 
535  if(server_handshake_completed == false && m_server->is_active())
536  server_handshake_completed = true;
537 
538  if(client->is_closed() || m_server->is_closed())
539  {
540  break;
541  }
542 
543  if(client->is_active() && client_has_written == false)
544  {
545  m_results.test_eq("client ALPN protocol", client->application_protocol(), "test/3");
546 
547  size_t sent_so_far = 0;
548  while(sent_so_far != client_msg.size())
549  {
550  const size_t left = client_msg.size() - sent_so_far;
551  const size_t rnd12 = (rng.next_byte() << 4) ^ rng.next_byte();
552  const size_t sending = std::min(left, rnd12);
553 
554  client->send(&client_msg[sent_so_far], sending);
555  sent_so_far += sending;
556  }
557  client->send_warning_alert(Botan::TLS::Alert::NO_RENEGOTIATION);
558  client_has_written = true;
559  }
560 
561  if(m_server->is_active() && server_has_written == false)
562  {
563  m_results.test_eq("server ALPN protocol", m_server->application_protocol(), "test/3");
564 
565  size_t sent_so_far = 0;
566  while(sent_so_far != server_msg.size())
567  {
568  const size_t left = server_msg.size() - sent_so_far;
569  const size_t rnd12 = (rng.next_byte() << 4) ^ rng.next_byte();
570  const size_t sending = std::min(left, rnd12);
571 
572  m_server->send(&server_msg[sent_so_far], sending);
573  sent_so_far += sending;
574  }
575 
576  m_server->send_warning_alert(Botan::TLS::Alert::NO_RENEGOTIATION);
577  server_has_written = true;
578  }
579 
580  if(m_c2s.size() > 0)
581  {
