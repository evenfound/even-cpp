119 QString QWebSocketHandshakeResponse::calculateAcceptKey(const QString &key) const
120 {
121  //the UID comes from RFC6455
122  const QString tmpKey = key % QStringLiteral("258EAFA5-E914-47DA-95CA-C5AB0DC85B11");
123  const QByteArray hash = QCryptographicHash::hash(tmpKey.toLatin1(), QCryptographicHash::Sha1);
124  return QString::fromLatin1(hash.toBase64());

130 QString QWebSocketHandshakeResponse::getHandshakeResponse(
131  const QWebSocketHandshakeRequest &request,
132  const QString &serverName,
133  bool isOriginAllowed,
134  const QList<QWebSocketProtocol::Version> &supportedVersions,
135  const QList<QString> &supportedProtocols,
136  const QList<QString> &supportedExtensions)
137 {
138  QStringList response;
139  m_canUpgrade = false;
140 
141  if (!isOriginAllowed) {
142  if (!m_canUpgrade) {
143  m_error = QWebSocketProtocol::CloseCodePolicyViolated;
144  m_errorString = tr("Access forbidden.");
145  response << QStringLiteral("HTTP/1.1 403 Access Forbidden");
146  }
147  } else {
148  if (request.isValid()) {
149  const QString acceptKey = calculateAcceptKey(request.key());
150  const QList<QString> matchingProtocols =
151  supportedProtocols.toSet().intersect(request.protocols().toSet()).toList();
152  //TODO: extensions must be kept in the order in which they arrive
153  //cannot use set.intersect() to get the supported extensions
154  const QList<QString> matchingExtensions =
155  supportedExtensions.toSet().intersect(request.extensions().toSet()).toList();
156  QList<QWebSocketProtocol::Version> matchingVersions =
157  request.versions().toSet().intersect(supportedVersions.toSet()).toList();
158  std::sort(matchingVersions.begin(), matchingVersions.end(),
159  std::greater<QWebSocketProtocol::Version>()); //sort in descending order
160 
161  if (Q_UNLIKELY(matchingVersions.isEmpty())) {
162  m_error = QWebSocketProtocol::CloseCodeProtocolError;
163  m_errorString = tr("Unsupported version requested.");
164  m_canUpgrade = false;
165  } else {
166  response << QStringLiteral("HTTP/1.1 101 Switching Protocols") <<
167  QStringLiteral("Upgrade: websocket") <<
168  QStringLiteral("Connection: Upgrade") <<
169  QStringLiteral("Sec-WebSocket-Accept: ") % acceptKey;
170  if (!matchingProtocols.isEmpty()) {
171  m_acceptedProtocol = matchingProtocols.first();
172  response << QStringLiteral("Sec-WebSocket-Protocol: ") % m_acceptedProtocol;
173  }
174  if (!matchingExtensions.isEmpty()) {
175  m_acceptedExtension = matchingExtensions.first();
176  response << QStringLiteral("Sec-WebSocket-Extensions: ") % m_acceptedExtension;
177  }
178  QString origin = request.origin().trimmed();
179  if (origin.contains(QStringLiteral("\r\n")) ||
180  serverName.contains(QStringLiteral("\r\n"))) {
181  m_error = QWebSocketProtocol::CloseCodeAbnormalDisconnection;
182  m_errorString = tr("One of the headers contains a newline. " \
183  "Possible attack detected.");
184  m_canUpgrade = false;
185  } else {
186  if (origin.isEmpty())
187  origin = QStringLiteral("*");
188  QDateTime datetime = QDateTime::currentDateTimeUtc();
189  response << QStringLiteral("Server: ") % serverName <<
190  QStringLiteral("Access-Control-Allow-Credentials: false") <<
191  QStringLiteral("Access-Control-Allow-Methods: GET") <<
192  QStringLiteral("Access-Control-Allow-Headers: content-type") <<
193  QStringLiteral("Access-Control-Allow-Origin: ") % origin <<
194  QStringLiteral("Date: ") % QLocale::c()
195  .toString(datetime, QStringLiteral("ddd, dd MMM yyyy hh:mm:ss 'GMT'"));
196 
197 
198  m_acceptedVersion = QWebSocketProtocol::currentVersion();
199  m_canUpgrade = true;
200  }
201  }
202  } else {
203  m_error = QWebSocketProtocol::CloseCodeProtocolError;
204  m_errorString = tr("Bad handshake request received.");
205  m_canUpgrade = false;
206  }
207  if (Q_UNLIKELY(!m_canUpgrade)) {
208  response << QStringLiteral("HTTP/1.1 400 Bad Request");
209  QStringList versions;
210  for (QWebSocketProtocol::Version version : supportedVersions)
211  versions << QString::number(static_cast<int>(version));
212  response << QStringLiteral("Sec-WebSocket-Version: ")
213  % versions.join(QStringLiteral(", "));
214  }
215  }
216  response << QStringLiteral("\r\n"); //append empty line at end of header
217  return response.join(QStringLiteral("\r\n"));
218 }
