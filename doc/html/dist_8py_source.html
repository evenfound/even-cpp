73 def datestamp(tag):
74  ts = maybe_gpg(run_git(['show', '--no-patch', '--format=%ai', tag]))

76  ts_matcher = re.compile(r'^(\d{4})-(\d{2})-(\d{2}) (\d{2}):(\d{2}):(\d{2}) .*')
77 
78  logging.debug('Git returned timestamp of %s for tag %s' % (ts, tag))

162 def rewrite_version_file(version_file, target_version, snapshot_branch, rev_id, rel_date):
163

167  version_file_name = os.path.basename(version_file)

170 
171  version_re = re.compile('release_(major|minor|patch) = ([0-9]+)')

173  def content_rewriter():
174  for line in contents:
175 
176  if not snapshot_branch:
177  # Verify the version set in the source matches the tag
178  match = version_re.match(line)
179  if match:
180  name_to_idx = {
181  'major': 0,
182  'minor': 1,
183  'patch': 2
184  }
185  version_parts = target_version.split('.')
186  assert len(version_parts) == 3
187  in_tag = int(version_parts[name_to_idx[match.group(1)]])
188  in_file = int(match.group(2))
189 
190  if in_tag != in_file:
191  raise Exception('Version number part "%s" in %s does not match tag %s' %
192  (match.group(1), version_file_name, target_version))
193 
194  if line == 'release_vc_rev = None\n':
195  yield 'release_vc_rev = \'git:%s\'\n' % (rev_id)
196  elif line == 'release_datestamp = 0\n':
197  yield 'release_datestamp = %d\n' % (rel_date)
198  elif line == "release_type = \'unreleased\'\n":
199  if target_version == snapshot_branch:
200  yield "release_type = 'snapshot:%s'\n" % (snapshot_branch)
201  else:

215  def write_mode(archive_type):
216  if archive_type == 'tgz':
217  return 'w:gz'
218  elif archive_type == 'tbz':
219  return 'w:bz2'
220  elif archive_type == 'tar':
221  return 'w'
222  else:
223  raise Exception("Unknown archive type '%s'" % (archive_type))
224 
225  # gzip format embeds the original filename, tarfile.py does the wrong
226  # thing unless the output name ends in .gz. So pass an explicit
227  # fileobj in that case, and supply a name in the form tarfile expects.
228  if archive_type == 'tgz':
229  archive = tarfile.open(output_basename + '.tar.gz',
230  write_mode(archive_type),
231  fileobj=open(output_archive, 'wb'))
232  else:
233  archive = tarfile.open(output_basename + '.tar',
234  write_mode(archive_type))
235 
236  for f in all_files:
237  tarinfo = archive.gettarinfo(f)
238  tarinfo.uid = 500
239  tarinfo.gid = 500
240  tarinfo.uname = "botan"
241  tarinfo.gname = "botan"
242  tarinfo.mtime = rel_epoch
243  archive.addfile(tarinfo, open(f, 'rb'))
244  archive.close()
245 
246  sha256 = hashlib.new('sha256')
247  sha256.update(open(output_archive, 'rb').read())
248  archive_hash = sha256.hexdigest().upper()
249 
250  logging.info('SHA-256(%s) = %s' % (output_archive, archive_hash))
251  if hash_file is not None:
252  hash_file.write("%s %s\n" % (archive_hash, output_archive))
253
