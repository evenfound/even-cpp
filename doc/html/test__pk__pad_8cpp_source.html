27  Test::Result run_one_test(const std::string& algo, const VarMap& vars) override
28  {
29  Test::Result result(algo + " Decoding");
30 
31  std::unique_ptr<Botan::EME> eme;
32 
33  try
34  {
35  eme.reset(Botan::get_eme(algo));
36  }
37  catch(Botan::Lookup_Error&)
38  {
39  result.note_missing(algo);
40  return result;
41  }
42 
43  const std::vector<uint8_t> ciphertext = vars.get_req_bin("RawCiphertext");
44  const std::vector<uint8_t> plaintext = vars.get_opt_bin("Plaintext");
45  const bool is_valid = vars.get_req_bool("ValidInput");
46 
47  if(is_valid == false)
48  {
49  result.test_eq("Plaintext value is empty for invalid EME inputs", plaintext.size(), 0);
50  }
51 
52  uint8_t valid_mask = 0;
53  Botan::secure_vector<uint8_t> decoded =
54  eme->unpad(valid_mask, ciphertext.data(), ciphertext.size());
55 
56  result.confirm("EME valid_mask has expected value", valid_mask == 0x00 || valid_mask == 0xFF);
57  result.test_eq("EME decoding valid/invalid matches", valid_mask == 0xFF, is_valid);
58 
59  if(is_valid && valid_mask == 0xFF)
60  {
61  result.test_eq("EME decoded plaintext correct", decoded, plaintext);
62  }
63 
64  // TODO: also test that encoding is accepted
65 
66  return result;
67  }
