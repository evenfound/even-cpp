278  * @return the number of bytes sent

280 ssize_t yamux_stream_write(struct YamuxChannelContext* channel_ctx, uint32_t data_length, void* data_)
281 {
282  // validate parameters
283  if (channel_ctx == NULL || data_ == NULL || data_length == 0)
284  return -EINVAL;
285  /*
286  if (!((size_t)stream | (size_t)data_) || stream->state == yamux_stream_closed
287  || stream->state == yamux_stream_closing || stream->session->closed)
288  return -EINVAL;
289  */
290 
291  // gather details
292  char* data = (char*)data_;
293  char* data_end = data + data_length;
294  uint32_t ws = channel_ctx->window_size;
295  uint32_t id = channel_ctx->channel;
296 
297  char sendd[ws + sizeof(struct yamux_frame)];
298 
299  // Send the data, breaking it up into pieces if it is too large
300  while (data < data_end) {
301  uint32_t dr = (uint32_t)(data_end - data); // length of the data for this round
302  uint32_t adv = MIN(dr, ws); // the size of the data we will send this round
303 
304  struct yamux_frame f = (struct yamux_frame){
305  .version = YAMUX_VERSION ,
306  .type = yamux_frame_data,
307  .flags = get_flags(channel_ctx),
308  .streamid = id,
309  .length = adv
310  };
311 
312  encode_frame(&f);
313  // put the frame into the buffer
314  memcpy(sendd, &f, sizeof(struct yamux_frame));
315  // put the data into the buffer
316  memcpy(sendd + sizeof(struct yamux_frame), data, (size_t)adv);
317 
318  // send the buffer through the network
319  struct StreamMessage outgoing;
320  outgoing.data = (uint8_t*)sendd;
321  outgoing.data_size = adv + sizeof(struct yamux_frame);
322  if (!channel_ctx->yamux_context->stream->parent_stream->write(channel_ctx->yamux_context->stream->parent_stream->stream_context, &outgoing))
323  return adv;
324 
325  // prepare to loop again
326  data += adv;
327  }
328 
329  return data_end - (char*)data_;
