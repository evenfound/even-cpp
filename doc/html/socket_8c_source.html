120 int socket_connect4_with_timeout(int s, uint32_t ip, uint16_t port, int timeout_secs)
121 {
122  struct sockaddr_in sa;
123  long args; // fctl args with O_NONBLOCK
124  long orig_args; // fctl args;
125 
126  memset(&sa, 0, sizeof sa);
127  sa.sin_family = AF_INET;
128  sa.sin_port = htons(port);
129  sa.sin_addr.s_addr = ip;
130 
131  // set to non blocking
132  orig_args = fcntl(s, F_GETFL, NULL);
133  if (orig_args < 0) {
134  // unable to get flags
135  libp2p_logger_error("socket", "Unable to get socket flags on connect.\n");
136  return -1;
137  }
138  args = orig_args;
139  args |= O_NONBLOCK;
140  if (args != orig_args) {
141  libp2p_logger_debug("socket", "Setting socket to non-blocking on connect.\n");
142  if (fcntl(s, F_SETFL, args) < 0) {
143  // unable to set flags
144  return -1;
145  }
146  } else {
147  libp2p_logger_debug("socket", "Socket already non-blocking during connect.\n");
148  }
149 
150  // connect
151  int retVal = connect(s, (struct sockaddr *) &sa, sizeof sa);
152  int tried_secs = 0;
153  while (retVal == -1 && errno == EINPROGRESS && tried_secs < timeout_secs) {
154  libp2p_logger_debug("socket", "Socket connect unsuccessful. Waiting to try again.\n");
155  // wait for timeout
156  sleep(1);
157  tried_secs++;
158  retVal = connect(s, (struct sockaddr *) &sa, sizeof sa);
159  if (retVal == -1 && errno == EALREADY) {
160  libp2p_logger_debug("socket", "Socket connect completed.\n");
161  retVal = 0;
162  break;
163  }
164  }
165 
166  if (retVal == -1) {
167  libp2p_logger_debug("socket", "Socket connect failed with error %d.\n", errno);
168  }
169 
170  if (args != orig_args) {
171  // set back to blocking
172  libp2p_logger_debug("socket", "Setting socket back to blocking.\n");
173  args = fcntl(s, F_GETFL, NULL);
174  args &= (~O_NONBLOCK);
175  fcntl(s, F_SETFL, args);
176  }

178 }

247  // determine if we're big or little endian
248  int big_endian = 1;
249  if (*(char*)&big_endian == 1) {
250  big_endian = 0;
251  }
252  // convert to int32_t
253  int32_t conv = htonl(size);
254  // swap bytes if this machine is little endian
255  if (!big_endian) {
256  uint32_t b0, b1, b2, b3;
257  b0 = (conv & 0x000000ff) << 24u;
258  b1 = (conv & 0x0000ff00) << 8u;
259  b2 = (conv & 0x00ff0000) >> 8u;
260  b3 = (conv & 0xff000000) >> 24u;
261  conv = b0 | b1 | b2 | b3;
262  }
263 
264  // send to socket
265  char* data = (char*)&conv;
266  int left = sizeof(conv);
267  ssize_t rc;
268  int retries_left = 100;
269  do {
270  rc = send(s, data, left, flags);
271  if (rc < 0) {
272  if ((errno == EAGAIN) || (errno == EWOULDBLOCK)) {
273  retries_left--;
274  if (retries_left <= 0)
275  break;
276  } else {
277  return -1;
278  }
279  } else {
280  data += rc;
281  left -= rc;
282  }
283  } while (left > 0);
284  return sizeof(conv) - left;
