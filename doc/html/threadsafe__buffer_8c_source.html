65 size_t threadsafe_buffer_read(struct ThreadsafeBufferContext* context, uint8_t* results, size_t results_size) {
66  size_t bytes_read = 0;
67  if (context == NULL)
68  return 0;
69  pthread_mutex_lock(&context->lock);
70  // do the read
71  if (context->buffer != NULL && context->buffer_size > 0) {
72  bytes_read = results_size < context->buffer_size ? results_size : context->buffer_size;
73  libp2p_logger_debug("threadsafe_buffer", "read: We want to read %d bytes, and have %d in the buffer. Therefore, we will read %d.\n", results_size, context->buffer_size, bytes_read);
74  memcpy(results, context->buffer, bytes_read);
75  }
76  // adjust the size
77  if (bytes_read > 0) {
78  if (context->buffer_size - bytes_read > 0) {
79  // more remains
80  size_t new_buffer_size = context->buffer_size - bytes_read;
81  uint8_t* new_buffer = (uint8_t*) malloc(new_buffer_size);
82  memcpy(new_buffer, &context->buffer[bytes_read], new_buffer_size);
83  free(context->buffer);
84  context->buffer = new_buffer;
85  context->buffer_size = new_buffer_size;
86  } else {
87  // everything has been read
88  free(context->buffer);
89  context->buffer = NULL;
90  context->buffer_size = 0;
91  }
92  }
93  pthread_mutex_unlock(&context->lock);
94  return bytes_read;
95 }
