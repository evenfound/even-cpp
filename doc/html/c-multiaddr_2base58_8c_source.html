42  uint32_t c;
43  size_t i, j;
44  uint8_t bytesleft = binsz % 4;
45  uint32_t zeromask = bytesleft ? (0xffffffff << (bytesleft * 8)) : 0;
46  unsigned zerocount = 0;
47  size_t b58sz;
48 
49  b58sz = strlen(b58);
50 
51  memset(outi, 0, outisz * sizeof(*outi));
52 
53  // Leading zeros, just count
54  for (i = 0; i < b58sz && !b58digits_map[b58u[i]]; ++i) {
55  ++zerocount;
56  }
57 
58  for (; i < b58sz; ++i) {
59  if (b58u[i] & 0x80) {
60  // High-bit set on invalid digit
61  return 0;
62  }
63  if (b58digits_map[b58u[i]] == -1) {
64  // Invalid base58 digit
65  return 0;
66  }
67  c = (unsigned)b58digits_map[b58u[i]];
68  for (j = outisz; j--;) {
69  t = ((uint64_t)outi[j]) * 58 + c;
70  c = (t & 0x3f00000000) >> 32;
71  outi[j] = t & 0xffffffff;
72  }
73  if (c) {
74  // Output number too big (carry to the next int32)
75  memset(outi, 0, outisz * sizeof(*outi));
76  return 0;
77  }
78  if (outi[0] & zeromask) {
79  // Output number too big (last int32 filled too far)
80  memset(outi, 0, outisz * sizeof(*outi));
81  return 0;
82  }
83  }
84 
85  j = 0;
86  switch (bytesleft) {
87  case 3:
88  *(binu++) = (outi[0] & 0xff0000) >> 16;
89  case 2:
90  *(binu++) = (outi[0] & 0xff00) >> 8;
91  case 1:
92  *(binu++) = (outi[0] & 0xff);
93  ++j;
94  default:
95  break;
96  }
97 
98  for (; j < outisz; ++j) {
99  *(binu++) = (outi[j] >> 0x18) & 0xff;
100  *(binu++) = (outi[j] >> 0x10) & 0xff;
101  *(binu++) = (outi[j] >> 8) & 0xff;
102  *(binu++) = (outi[j] >> 0) & 0xff;
103  }
104

106  binu = *bin;
107  for (i = 0; i < binsz; ++i) {

109  break;
110  }

112  }
