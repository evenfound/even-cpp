22  const uintptr_t buf = reinterpret_cast<uintptr_t>(buf_ptr);
23  return (buf >= pool) && (buf + bufsize <= pool + poolsize);

64 void* Memory_Pool::allocate(size_t req)
65  {
66  const size_t alignment = (static_cast<size_t>(1) << m_align_bit);
67 
68  if(req > m_pool_size)
69  return nullptr;
70  if(req < m_min_alloc || req > m_max_alloc)
71  return nullptr;
72 
73  lock_guard_type<mutex_type> lock(m_mutex);
74 
75  auto best_fit = m_freelist.end();
76 
77  for(auto i = m_freelist.begin(); i != m_freelist.end(); ++i)
78  {
79  // If we have a perfect fit, use it immediately
80  if(i->second == req && (i->first % alignment) == 0)
81  {
82  const size_t offset = i->first;
83  m_freelist.erase(i);
84  clear_mem(m_pool + offset, req);
85 
86  BOTAN_ASSERT((reinterpret_cast<uintptr_t>(m_pool) + offset) % alignment == 0,
87  "Returning correctly aligned pointer");
88 
89  return m_pool + offset;
90  }
91 
92 
93  if(((best_fit == m_freelist.end()) || (best_fit->second > i->second)) &&
94  (i->second >= (req + padding_for_alignment(i->first, alignment))))
95  {
96  best_fit = i;
97  }
98  }
99 
100  if(best_fit != m_freelist.end())
101  {
102  const size_t offset = best_fit->first;
103 
104  const size_t alignment_padding = padding_for_alignment(offset, alignment);
105 
106  best_fit->first += req + alignment_padding;
107  best_fit->second -= req + alignment_padding;
108 
109  // Need to realign, split the block
110  if(alignment_padding)
111  {
112  /*
113  If we used the entire block except for small piece used for
114  alignment at the beginning, so just update the entry already
115  in place (as it is in the correct location), rather than
116  deleting the empty range and inserting the new one in the
117  same location.
118  */
119  if(best_fit->second == 0)
120  {
121  best_fit->first = offset;
122  best_fit->second = alignment_padding;
123  }
124  else
125  m_freelist.insert(best_fit, std::make_pair(offset, alignment_padding));
126  }
127 
128  clear_mem(m_pool + offset + alignment_padding, req);
129 
130  BOTAN_ASSERT((reinterpret_cast<uintptr_t>(m_pool) + offset + alignment_padding) % alignment == 0,
131  "Returning correctly aligned pointer");
132 
133  return m_pool + offset + alignment_padding;
134  }
135 
136  return nullptr;
137  }

139 bool Memory_Pool::deallocate(void* p, size_t n) noexcept
140  {
141  if(!ptr_in_pool(m_pool, m_pool_size, p, n))
142  return false;
143 
144  std::memset(p, 0, n);
145 
146  lock_guard_type<mutex_type> lock(m_mutex);
147 
148  const size_t start = static_cast<uint8_t*>(p) - m_pool;
149 
150  auto comp = [](std::pair<size_t, size_t> x, std::pair<size_t, size_t> y){ return x.first < y.first; };
151 
152  auto i = std::lower_bound(m_freelist.begin(), m_freelist.end(),
153  std::make_pair(start, 0), comp);
154 
155  // try to merge with later block
156  if(i != m_freelist.end() && start + n == i->first)
157  {
158  i->first = start;
159  i->second += n;
160  n = 0;
161  }
162 
163  // try to merge with previous block
164  if(i != m_freelist.begin())
165  {
166  auto prev = std::prev(i);
167 
168  if(prev->first + prev->second == start)
169  {
170  if(n)
171  {
172  prev->second += n;
173  n = 0;
174  }
175  else
176  {
177  // merge adjoining
178  prev->second += i->second;
179  m_freelist.erase(i);
180  }
181  }
182  }
183 
184  if(n != 0) // no merge possible?
185  m_freelist.insert(i, std::make_pair(start, n));
186 
187  return true;
188  }
