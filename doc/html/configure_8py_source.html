59 def parse_version_file(version_path):
60  version_file = open(version_path)
61  key_and_val = re.compile(r"([a-z_]+) = ([a-zA-Z0-9:\-\']+)")
62 
63  results = {}
64  for line in version_file.readlines():
65  if not line or line[0] == '#':
66  continue
67  match = key_and_val.match(line)
68  if match:
69  key = match.group(1)
70  val = match.group(2)
71 
72  if val == 'None':
73  val = None
74  elif val.startswith("'") and val.endswith("'"):
75  val = val[1:len(val)-1]
76  else:
77  val = int(val)
78 
79  results[key] = val
80  return results

681 def lex_me_harder(infofile, allowed_groups, allowed_maps, name_val_pairs):
682  """
683  Generic lexer function for info.txt and src/build-data files
684  """
685  out = LexResult()
686 
687  # Format as a nameable Python variable
688  def py_var(group):
689  return group.replace(':', '_')
690 
691  lexer = shlex.shlex(open(infofile), infofile, posix=True)
692  lexer.wordchars += ':.<>/,-!?+*' # handle various funky chars in info.txt
693 
694  groups = allowed_groups + allowed_maps
695  for group in groups:
696  out.__dict__[py_var(group)] = []
697  for (key, val) in name_val_pairs.items():
698  out.__dict__[key] = val
699 
700  def lexed_tokens(): # Convert to an iterator
701  while True:
702  token = lexer.get_token()
703  if token != lexer.eof:
704  yield token
705  else:
706  return
707 
708  for token in lexed_tokens():
709  match = re.match('<(.*)>', token)
710 
711  # Check for a grouping
712  if match is not None:
713  group = match.group(1)
714 
715  if group not in groups:
716  raise LexerError('Unknown group "%s"' % (group),
717  infofile, lexer.lineno)
718 
719  end_marker = '</' + group + '>'
720 
721  token = lexer.get_token()
722  while token != end_marker:
723  out.__dict__[py_var(group)].append(token)
724  token = lexer.get_token()
725  if token is None:
726  raise LexerError('Group "%s" not terminated' % (group),
727  infofile, lexer.lineno)
728 
729  elif token in name_val_pairs.keys():
730  if isinstance(out.__dict__[token], list):
731  out.__dict__[token].append(lexer.get_token())
732  else:
733  out.__dict__[token] = lexer.get_token()
734 
735  else: # No match -> error
736  raise LexerError('Bad token "%s"' % (token), infofile, lexer.lineno)
737 
738  for group in allowed_maps:
739  out.__dict__[group] = parse_lex_dict(out.__dict__[group])
740 
741  return out
