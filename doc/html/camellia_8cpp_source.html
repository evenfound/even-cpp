616 inline uint64_t FL(uint64_t v, uint64_t K)
617  {
618  uint32_t x1 = static_cast<uint32_t>(v >> 32);
619  uint32_t x2 = static_cast<uint32_t>(v & 0xFFFFFFFF);
620 
621  const uint32_t k1 = static_cast<uint32_t>(K >> 32);
622  const uint32_t k2 = static_cast<uint32_t>(K & 0xFFFFFFFF);
623 
624  x2 ^= rotl<1>(x1 & k1);
625  x1 ^= (x2 | k2);
626 
627  return ((static_cast<uint64_t>(x1) << 32) | x2);
628  }

630 inline uint64_t FLINV(uint64_t v, uint64_t K)
631  {
632  uint32_t x1 = static_cast<uint32_t>(v >> 32);
633  uint32_t x2 = static_cast<uint32_t>(v & 0xFFFFFFFF);
634 
635  const uint32_t k1 = static_cast<uint32_t>(K >> 32);
636  const uint32_t k2 = static_cast<uint32_t>(K & 0xFFFFFFFF);
637 
638  x1 ^= (x2 | k2);
639  x2 ^= rotl<1>(x1 & k1);
640 
641  return ((static_cast<uint64_t>(x1) << 32) | x2);
642  }

644 /*
645 * Camellia Encryption
646 */
647 void encrypt(const uint8_t in[], uint8_t out[], size_t blocks,
648  const secure_vector<uint64_t>& SK, const size_t rounds)
649  {
650  BOTAN_PARALLEL_FOR(size_t i = 0; i < blocks; ++i)
651  {
652  uint64_t D1, D2;
653  load_be(in + 16*i, D1, D2);
654 
655  const uint64_t* K = SK.data();
656 
657  D1 ^= *K++;
658  D2 ^= *K++;
659 
660  D2 ^= F_SLOW(D1, *K++);
661  D1 ^= F_SLOW(D2, *K++);
662 
663  for(size_t r = 1; r != rounds - 1; ++r)
664  {
665  if(r % 3 == 0)
666  {
667  D1 = FL (D1, *K++);
668  D2 = FLINV(D2, *K++);
669  }
670 
671  D2 ^= F(D1, *K++);
672  D1 ^= F(D2, *K++);
673  }
674 
675  D2 ^= F_SLOW(D1, *K++);
676  D1 ^= F_SLOW(D2, *K++);
677 
678  D2 ^= *K++;
679  D1 ^= *K++;
680 
681  store_be(out + 16*i, D2, D1);
682  }
683  }

685 /*
686 * Camellia Decryption
687 */
688 void decrypt(const uint8_t in[], uint8_t out[], size_t blocks,
689  const secure_vector<uint64_t>& SK, const size_t rounds)
690  {
691  BOTAN_PARALLEL_FOR(size_t i = 0; i < blocks; ++i)
692  {
693  uint64_t D1, D2;
694  load_be(in + 16*i, D1, D2);
695 
696  const uint64_t* K = &SK[SK.size()-1];
697 
698  D2 ^= *K--;
699  D1 ^= *K--;
700 
701  D2 ^= F_SLOW(D1, *K--);
702  D1 ^= F_SLOW(D2, *K--);
703 
704  for(size_t r = 1; r != rounds - 1; ++r)
705  {
706  if(r % 3 == 0)
707  {
708  D1 = FL (D1, *K--);
709  D2 = FLINV(D2, *K--);
710  }
711 
712  D2 ^= F(D1, *K--);
713  D1 ^= F(D2, *K--);
714  }
715 
716  D2 ^= F_SLOW(D1, *K--);
717  D1 ^= F_SLOW(D2, *K--);
718 
719  D1 ^= *K--;
720  D2 ^= *K;
721 
722  store_be(out + 16*i, D2, D1);
723  }
724  }
