174 int ipfs_bitswap_get_block(struct Exchange* exchange, struct Cid* cid, struct Block** block) {
175  struct BitswapContext* bitswapContext = (struct BitswapContext*)exchange->exchangeContext;
176  if (bitswapContext != NULL) {
177  // check locally first
178  if (bitswapContext->ipfsNode->blockstore->Get(bitswapContext->ipfsNode->blockstore->blockstoreContext, cid, block))
179  return 1;
180  // now ask the network
181  //NOTE: this timeout should be configurable
182  int timeout = 60;
183  int waitSecs = 1;
184  int timeTaken = 0;
185  struct WantListSession *wantlist_session = ipfs_bitswap_wantlist_session_new();
186  wantlist_session->type = WANTLIST_SESSION_TYPE_LOCAL;
187  wantlist_session->context = (void*)bitswapContext->ipfsNode;
188  struct WantListQueueEntry* want_entry = ipfs_bitswap_want_manager_add(bitswapContext, cid, wantlist_session);
189  if (want_entry != NULL) {
190  // loop waiting for it to fill
191  while(1) {
192  if (want_entry->block != NULL) {
193  *block = ipfs_block_copy(want_entry->block);
194  // error or not, we no longer need the block (decrement reference count)
195  ipfs_bitswap_want_manager_remove(bitswapContext, cid);
196  if (*block == NULL) {
197  return 0;
198  }
199  return 1;
200  }
201  //TODO: This is a busy-loop. Find another way.
202  timeTaken += waitSecs;
203  if (timeTaken >= timeout) {
204  // It took too long. Stop looking.
205  ipfs_bitswap_want_manager_remove(bitswapContext, cid);
206  break;
207  }
208  sleep(waitSecs);
209  }
210  }
211  }
212  return 0;
213 }
