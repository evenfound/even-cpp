134  std::vector<Botan::BigInt> factorize(const Botan::BigInt& n_in,
135  Botan::RandomNumberGenerator& rng)
136  {
137  Botan::BigInt n = n_in;
138  std::vector<Botan::BigInt> factors = remove_small_factors(n);

166  /*
167  * Pollard's Rho algorithm, as described in the MIT algorithms book.
168  * Uses Brent's cycle finding
169  */
170  Botan::BigInt rho(const Botan::BigInt& n, Botan::RandomNumberGenerator& rng)
171  {
172  auto monty_n = std::make_shared<Botan::Montgomery_Params>(n);
173 
174  const Botan::Montgomery_Int one(monty_n, monty_n->R1(), false);
175 
176  Botan::Montgomery_Int x(monty_n, Botan::BigInt::random_integer(rng, 2, n - 3), false);
177  Botan::Montgomery_Int y = x;
178  Botan::Montgomery_Int z = one;
179  Botan::Montgomery_Int t(monty_n);
180  Botan::BigInt d;
181 
182  Botan::secure_vector<Botan::word> ws;
183 
184  size_t i = 1, k = 2;
185 
186  while(true)
187  {
188  i++;
189 
190  if(i >= 0xFFFF0000) // bad seed? too slow? bail out
191  {
192  break;
193  }
194 
195  x.square_this(ws); // x = x^2
196  x.add(one, ws);
197 
198  t = y;
199  t.sub(x, ws);
200 
201  z.mul_by(t, ws);
202 
203  if(i == k || i % 128 == 0)
204  {
205  d = Botan::gcd(z.value(), n);
206  z = one;
207 
208  if(d == n)
209  {
210  // TODO Should rewind here
211  break;
212  }
213 
214  if(d != 1)
215  return d;
216  }
217 
218  if(i == k)
219  {
220  y = x;
221  k = 2 * k;
222  }
223  }
224 
225  // failed
226  return 0;
227  }

229  // Remove (and return) any small (< 2^16) factors
230  std::vector<Botan::BigInt> remove_small_factors(Botan::BigInt& n)
231  {
232  std::vector<Botan::BigInt> factors;
233 
234  while(n.is_even())
235  {
236  factors.push_back(2);
237  n /= 2;
238  }
239 
240  for(size_t j = 0; j != Botan::PRIME_TABLE_SIZE; j++)
241  {
242  uint16_t prime = Botan::PRIMES[j];
243  if(n < prime)
244  {
245  break;
246  }
247 
248  Botan::BigInt x = Botan::gcd(n, prime);
249 
250  if(x != 1)
251  {
252  n /= x;
253 
254  while(x != 1)
255  {
256  x /= prime;
257  factors.push_back(prime);
258  }
259  }
260  }
261 
262  return factors;
263  }
