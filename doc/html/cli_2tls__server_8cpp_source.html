50  void go() override
51  {
52  const std::string server_crt = get_arg("cert");
53  const std::string server_key = get_arg("key");
54  const int port = get_arg_sz("port");
55  const size_t max_clients = get_arg_sz("max-clients");
56  const std::string transport = get_arg("type");
57  const std::string dump_traces_to = get_arg("dump-traces");
58 
59  if(transport != "tcp" && transport != "udp")
60  {
61  throw CLI_Usage_Error("Invalid transport type '" + transport + "' for TLS");
62  }
63 
64  m_is_tcp = (transport == "tcp");
65 
66  std::unique_ptr<Botan::TLS::Policy> policy;
67  const std::string policy_file = get_arg("policy");
68  if(policy_file.size() > 0)
69  {
70  std::ifstream policy_stream(policy_file);
71  if(!policy_stream.good())
72  {
73  error_output() << "Failed reading policy file\n";
74  return;
75  }
76  policy.reset(new Botan::TLS::Text_Policy(policy_stream));
77  }
78 
79  if(!policy)
80  {
81  policy.reset(new Botan::TLS::Policy);
82  }
83 
84  Botan::TLS::Session_Manager_In_Memory session_manager(rng()); // TODO sqlite3
85 
86  Basic_Credentials_Manager creds(rng(), server_crt, server_key);
87 
88  output() << "Listening for new connections on " << transport << " port " << port << std::endl;
89 
90  int server_fd = make_server_socket(port);
91  size_t clients_served = 0;
92 
93  while(true)
94  {
95  if(max_clients > 0 && clients_served >= max_clients)
96  break;
97 
98  if(m_is_tcp)
99  {
100  m_socket = ::accept(server_fd, nullptr, nullptr);
101  }
102  else
103  {
104  struct sockaddr_in from;
105  socklen_t from_len = sizeof(sockaddr_in);
106 
107  if(::recvfrom(server_fd, nullptr, 0, MSG_PEEK, reinterpret_cast<struct sockaddr*>(&from), &from_len) != 0)
108  {
109  throw CLI_Error("Could not peek next packet");
110  }
111 
112  if(::connect(server_fd, reinterpret_cast<struct sockaddr*>(&from), from_len) != 0)
113  {
114  throw CLI_Error("Could not connect UDP socket");
115  }
116  m_socket = server_fd;
117  }
118 
119  clients_served++;
120 
121  Botan::TLS::Server server(
122  *this,
123  session_manager,
124  creds,
125  *policy,
126  rng(),
127  m_is_tcp == false);
128 
129  std::unique_ptr<std::ostream> dump_stream;
130 
131  if(!dump_traces_to.empty())
132  {
133  uint64_t timestamp = Botan::OS::get_high_resolution_clock();
134  const std::string dump_file =
135  dump_traces_to + "/tls_" + std::to_string(timestamp) + ".bin";
136  dump_stream.reset(new std::ofstream(dump_file.c_str()));
137  }
138 
139  try
140  {
141  while(!server.is_closed())
142  {
143  try
144  {
145  uint8_t buf[4 * 1024] = { 0 };
146  ssize_t got = ::read(m_socket, buf, sizeof(buf));
147 
148  if(got == -1)
149  {
150  error_output() << "Error in socket read - " << std::strerror(errno) << std::endl;
151  break;
152  }
153 
154  if(got == 0)
155  {
156  error_output() << "EOF on socket" << std::endl;
157  break;
158  }
159 
160  if(dump_stream)
161  {
162  dump_stream->write(reinterpret_cast<const char*>(buf), got);
163  }
164 
165  server.received_data(buf, got);
166 
167  while(server.is_active() && !m_pending_output.empty())
168  {
169  std::string output = m_pending_output.front();
170  m_pending_output.pop_front();
171  server.send(output);
172 
173  if(output == "quit\n")
174  {
175  server.close();
176  }
177  }
178  }
179  catch(std::exception& e)
180  {
181  error_output() << "Connection problem: " << e.what() << std::endl;
182  if(m_is_tcp)
183  {
184  ::close(m_socket);
185  m_socket = -1;
186  }
187  }
188  }
189  }
190  catch(Botan::Exception& e)
191  {
192  error_output() << "Connection failed: " << e.what() << "\n";
193  }
194 
195  if(m_is_tcp)
196  {
197  ::close(m_socket);
198  m_socket = -1;
199  }
200  }
201 
202  ::close(server_fd);
203  }
204  private:
205  int make_server_socket(uint16_t port)
206  {
207  const int type = m_is_tcp ? SOCK_STREAM : SOCK_DGRAM;
208 
209  int fd = ::socket(PF_INET, type, 0);
210  if(fd == -1)
211  {
212  throw CLI_Error("Unable to acquire socket");
213  }
214 
215  sockaddr_in socket_info;
216  ::memset(&socket_info, 0, sizeof(socket_info));
217  socket_info.sin_family = AF_INET;
218  socket_info.sin_port = htons(port);
219 
220  // FIXME: support limiting listeners
221  socket_info.sin_addr.s_addr = INADDR_ANY;
222 
223  if(::bind(fd, reinterpret_cast<struct sockaddr*>(&socket_info), sizeof(struct sockaddr)) != 0)
224  {
225  ::close(fd);
226  throw CLI_Error("server bind failed");
227  }
228 
229  if(m_is_tcp)
230  {
231  if(::listen(fd, 100) != 0)
232  {
233  ::close(fd);
234  throw CLI_Error("listen failed");
235  }
236  }
237  return fd;
238  }

272  void tls_emit_data(const uint8_t buf[], size_t length) override
273  {
274  if(m_is_tcp)
275  {
276  ssize_t sent = ::send(m_socket, buf, length, MSG_NOSIGNAL);
277 
278  if(sent == -1)
279  {
280  error_output() << "Error writing to socket - " << std::strerror(errno) << std::endl;
281  }
282  else if(sent != static_cast<ssize_t>(length))
283  {
284  error_output() << "Packet of length " << length << " truncated to " << sent << std::endl;
285  }
286  }
287  else
288  {
289  while(length)
290  {
291  ssize_t sent = ::send(m_socket, buf, length, MSG_NOSIGNAL);
292 
293  if(sent == -1)
294  {
295  if(errno == EINTR)
296  {
297  sent = 0;
298  }
299  else
300  {
301  throw CLI_Error("Socket write failed");
302  }
303  }
304 
305  buf += sent;
306  length -= sent;
307  }
308  }
