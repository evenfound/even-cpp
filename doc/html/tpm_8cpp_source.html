407  secure_vector<uint8_t> sign(RandomNumberGenerator&) override
408  {
409  /*
410  * v1.2 TPMs will only sign with PKCS #1 v1.5 padding. SHA-1 is built
411  * in, all other hash inputs (TSS_HASH_OTHER) are treated as the
412  * concatenation of the hash OID and hash value and signed with just the
413  * 01FFFF... prefix. Even when using SHA-1 we compute the hash locally
414  * since it is going to be much faster than pushing data over the LPC bus.
415  */
416  secure_vector<uint8_t> msg_hash = m_hash->final();
417 
418  std::vector<uint8_t> id_and_msg;
419  id_and_msg.reserve(m_hash_id.size() + msg_hash.size());
420  id_and_msg.insert(id_and_msg.end(), m_hash_id.begin(), m_hash_id.end());
421  id_and_msg.insert(id_and_msg.end(), msg_hash.begin(), msg_hash.end());
422 
423  TSS_HCONTEXT ctx = m_key.ctx().handle();
424  TSS_HHASH tpm_hash;
425  TSPI_CHECK_SUCCESS(::Tspi_Context_CreateObject(ctx, TSS_OBJECT_TYPE_HASH, TSS_HASH_OTHER, &tpm_hash));
426  TSPI_CHECK_SUCCESS(::Tspi_Hash_SetHashValue(tpm_hash, id_and_msg.size(), id_and_msg.data()));
427 
428  BYTE* sig_bytes = nullptr;
429  UINT32 sig_len = 0;
430  TSPI_CHECK_SUCCESS(::Tspi_Hash_Sign(tpm_hash, m_key.handle(), &sig_len, &sig_bytes));
431  secure_vector<uint8_t> sig(sig_bytes, sig_bytes + sig_len);
432 
433  // TODO: RAII for Context_FreeMemory
434  TSPI_CHECK_SUCCESS(::Tspi_Context_FreeMemory(ctx, sig_bytes));
435 
436  // TODO: RAII for Context_CloseObject
437  TSPI_CHECK_SUCCESS(::Tspi_Context_CloseObject(ctx, tpm_hash));
438 
439  return sig;
440  }
441 
442  private:
443  const TPM_PrivateKey& m_key;
444  std::unique_ptr<HashFunction> m_hash;
445  std::vector<uint8_t> m_hash_id;
446  };
