285 ticks ECDSA_Timing_Test::measure_critical_function(std::vector<uint8_t> input)
286  {
287  const Botan::BigInt k(input.data(), input.size());
288  const Botan::BigInt msg(5); // fixed message to minimize noise
289 
290  ticks start = get_ticks();
291 
292  //The following ECDSA operations involve and should not leak any information about k.
293 
294  const Botan::BigInt k_inv = m_group.inverse_mod_order(k);
295  const Botan::PointGFp k_times_P = m_group.blinded_base_point_multiply(k, Timing_Test::timing_test_rng(), m_ws);
296  const Botan::BigInt r = m_group.mod_order(k_times_P.get_affine_x());
297  const Botan::BigInt s = m_group.multiply_mod_order(k_inv, mul_add(m_x, r, msg));
298 
299  BOTAN_UNUSED(r, s);
300 
301  ticks end = get_ticks();
302 
303  return (end - start);
304  }

400 std::vector<std::vector<ticks>> Timing_Test::execute_evaluation(
401  const std::vector<std::string>& raw_inputs,
402  size_t warmup_runs, size_t measurement_runs)
403  {
404  std::vector<std::vector<ticks>> all_results(raw_inputs.size());
405  std::vector<std::vector<uint8_t>> inputs(raw_inputs.size());
406 
407  for(auto& result : all_results)
408  {
409  result.reserve(measurement_runs);
410  }
411 
412  for(size_t i = 0; i != inputs.size(); ++i)
413  {
414  inputs[i] = prepare_input(raw_inputs[i]);
415  }
416 
417  // arbitrary upper bounds of 1 and 10 million resp
418  if(warmup_runs > 1000000 || measurement_runs > 100000000)
419  {
420  throw CLI_Error("Requested execution counts too large, rejecting");
421  }
422 
423  size_t total_runs = 0;
424  while(total_runs < (warmup_runs + measurement_runs))
425  {
426  std::vector<ticks> results(inputs.size());
427 
428  for(size_t i = 0; i != inputs.size(); ++i)
429  {
430  results[i] = measure_critical_function(inputs[i]);
431  }
432 
433  total_runs++;
434 
435  if(total_runs >= warmup_runs)
436  {
437  for(size_t i = 0; i != results.size(); ++i)
438  {
439  all_results[i].push_back(results[i]);
440  }
441  }
442  }
443 
444  return all_results;
445  }
