392  int pow[256];

395  /*
396  * compute pow and log tables over GF(2^8)
397  */
398  for( i = 0, x = 1; i < 256; i++ )

401  log[x] = i;
402  x = ( x ^ XTIME( x ) ) & 0xFF;

405  /*
406  * calculate the round constants
407  */
