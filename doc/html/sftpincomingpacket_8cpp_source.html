43 void SftpIncomingPacket::consumeData(QByteArray &newData)
44 {
45 #ifdef CREATOR_SSH_DEBUG
46  qDebug("%s: current data size = %d, new data size = %d", Q_FUNC_INFO,
47  m_data.size(), newData.size());
48 #endif
49 
50  if (isComplete() || dataSize() + newData.size() < sizeof m_length)
51  return;
52 
53  if (dataSize() < sizeof m_length) {
54  moveFirstBytes(m_data, newData, sizeof m_length - m_data.size());
55  m_length = SshPacketParser::asUint32(m_data, static_cast<quint32>(0));
56  if (m_length < static_cast<quint32>(TypeOffset + 1)
57  || m_length > MaxPacketSize) {
58  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_PROTOCOL_ERROR,
59  "Invalid length field in SFTP packet.");
60  }
61  }
62 
63  moveFirstBytes(m_data, newData,
64  qMin<quint32>(m_length - dataSize() + 4, newData.size()));
65 }

192 SftpFileAttributes SftpIncomingPacket::asFileAttributes(quint32 &offset) const
193 {
194  SftpFileAttributes attributes;
195  const quint32 flags = SshPacketParser::asUint32(m_data, &offset);
196  attributes.sizePresent = flags & SSH_FILEXFER_ATTR_SIZE;
197  attributes.timesPresent = flags & SSH_FILEXFER_ATTR_ACMODTIME;
198  attributes.uidAndGidPresent = flags & SSH_FILEXFER_ATTR_UIDGID;
199  attributes.permissionsPresent = flags & SSH_FILEXFER_ATTR_PERMISSIONS;
200  if (attributes.sizePresent)
201  attributes.size = SshPacketParser::asUint64(m_data, &offset);
202  if (attributes.uidAndGidPresent) {
203  attributes.uid = SshPacketParser::asUint32(m_data, &offset);
204  attributes.gid = SshPacketParser::asUint32(m_data, &offset);
205  }
206  if (attributes.permissionsPresent)
207  attributes.permissions = SshPacketParser::asUint32(m_data, &offset);
208  if (attributes.timesPresent) {
209  attributes.atime = SshPacketParser::asUint32(m_data, &offset);
210  attributes.mtime = SshPacketParser::asUint32(m_data, &offset);
211  }
212  if (flags & SSH_FILEXFER_ATTR_EXTENDED) {
213  const quint32 count = SshPacketParser::asUint32(m_data, &offset);
214  for (quint32 i = 0; i < count; ++i) {
215  SshPacketParser::asString(m_data, &offset);
216  SshPacketParser::asString(m_data, &offset);
217  }
218  }
219  return attributes;
220 }
