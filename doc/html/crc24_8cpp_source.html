194 /*
195 * Update a CRC24 Checksum
196 *
197 * Implementation uses Slicing-by-N algorithm described in
198 * "Novel Table Lookup-Based Algorithms for High-Performance
199 * CRC Generation", by M.Kounavis.
200 *
201 * This algorithm uses 4 precomputed look-up tables. First
202 * table T0 is computed same way as in a method proposed
203 * by D. Sarwate (1988). Then T_1, T2 and T3 are computed
204 * in following way:
205 *
206 * T1[j] = (T0[j] >> 8) ^ T0[ T0[j] & 0xFF ]
207 * T2[j] = (T1[j] >> 8) ^ T0[ T1[j] & 0xFF ]
208 * T3[j] = (T2[j] >> 8) ^ T0[ T2[j] & 0xFF ]
209 *
210 */
211 void CRC24::add_data(const uint8_t input[], size_t length)
212  {
213  uint32_t d[4];
214  uint32_t tmp = m_crc;
215 
216  // Input is word aligned if WA & input == 0
217  static const uint8_t WA = (BOTAN_MP_WORD_BITS/8) - 1;
218 
219  // Ensure input is word aligned before processing in parallel
220  for(;length && (reinterpret_cast<uintptr_t>(input) & WA); length--)
221  tmp = process8(tmp, *input++);
222 
223  while(length >= 16)
224  {
225  load_le(d, input, 4);
226  tmp = process32(tmp, d[0]);
227  tmp = process32(tmp, d[1]);
228  tmp = process32(tmp, d[2]);
229  tmp = process32(tmp, d[3]);
230 
231  input += 16;
232  length -= 16;
233  }
234 
235  while(length--)
236  tmp = process8(tmp, *input++);
237 
238  m_crc = tmp & 0xffffff;
239  }
