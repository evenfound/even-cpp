260  struct MultistreamContext* multistream_context = (struct MultistreamContext*)stream_context;
261  struct Stream* parent_stream = multistream_context->stream->parent_stream;
262 
263  // find out the length
264  uint8_t varint[12];
265  memset(varint, 0, 12);
266  size_t num_bytes_requested = 0;
267  size_t varint_length = 0;
268  for(int i = 0; i < 12; i++) {
269  if (parent_stream->read_raw(parent_stream->stream_context, &varint[i], 1, timeout_secs) == -1) {
270  libp2p_logger_debug("multistream", "read->read_raw returned false.\n");
271  return 0;
272  }
273  if (varint[i] >> 7 == 0) {
274  num_bytes_requested = varint_decode(&varint[0], i+1, &varint_length);
275  break;
276  }
277  }
278 
279  if (num_bytes_requested <= 0)
280  return 0;
281

283  *results = libp2p_stream_message_new();
284  struct StreamMessage* rslts = *results;
285  rslts->data_size = num_bytes_requested;
286  rslts->data = (uint8_t*) malloc(num_bytes_requested);
287  if (rslts->data == NULL) {
288  libp2p_logger_error("multistream", "read: Attempted allocation of stream message failed.\n");
289  libp2p_stream_message_free(rslts);
290  rslts = NULL;
291  }
292  // now get the data from the parent stream
293  int bytes_read = parent_stream->read_raw(parent_stream->stream_context, rslts->data, rslts->data_size, timeout_secs);
294  if (bytes_read != num_bytes_requested) {
295  libp2p_logger_error("multistream", "read: Was supposed to read %d bytes, but read_raw returned %d.\n", num_bytes_requested, bytes_read);
296  // problem reading from the parent stream
297  libp2p_stream_message_free(*results);
298  *results = NULL;
299  return 0;
300  }
