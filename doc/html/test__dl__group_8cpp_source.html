95  Test::Result test_dl_generate(Botan::RandomNumberGenerator& rng)

162  Test::Result test_dl_named(Botan::RandomNumberGenerator& rng)

179  "modp/srp/6144",
180  "modp/srp/8192",

182  "dsa/jce/1024",
183  "dsa/botan/2048",
184  "dsa/botan/3072",
185 
186  "ffdhe/ietf/2048",
187  "ffdhe/ietf/3072",
188  "ffdhe/ietf/4096",
189  "ffdhe/ietf/6144",
190  "ffdhe/ietf/8192",
191  };
192 
193  Test::Result result("DL_Group named");
194  result.start_timer();
195 
196  for(std::string name : dl_named)
197  {
198  // Confirm we can load every group we expect
199  Botan::DL_Group group(name);
200 
201  result.test_ne("DL_Group p is set", group.get_p(), 0);
202  result.test_ne("DL_Group g is set", group.get_g(), 0);
203 
204  const size_t strength = group.estimated_strength();
205 
206  // 8192 bit ~~ 2**202 strength
207  result.confirm("Plausible strength", strength >= 80 && strength < 210);
208 
209  if(name.find("modp/srp/") == std::string::npos)
210  {
211  result.test_ne("DL_Group q is set", group.get_q(), 0);
212  }
213  else
214  {
215  result.test_eq("DL_Group q is not set for SRP groups", group.get_q(), 0);
216  }
217 
218  if(group.p_bits() < 2048 || Test::run_long_tests())
219  {
220  result.test_eq(name + " verifies", group.verify_group(rng, false), true);
221  }
222 
223  }
224  result.end_timer();
225 
226  return result;
227  }
