44 Parameters ArgumentsCollector::collect(bool &success) const
45 {
46  Parameters parameters;
47  try {
48  bool authTypeGiven = false;
49  bool portGiven = false;
50  bool timeoutGiven = false;
51  bool smallFileCountGiven = false;
52  bool bigFileSizeGiven = false;
53  bool proxySettingGiven = false;
54  int pos;
55  int port;
56  for (pos = 1; pos < m_arguments.count() - 1; ++pos) {
57  if (checkAndSetStringArg(pos, parameters.sshParams.host, "-h")
58  || checkAndSetStringArg(pos, parameters.sshParams.userName, "-u"))
59  continue;
60  if (checkAndSetIntArg(pos, port, portGiven, "-p")
61  || checkAndSetIntArg(pos, parameters.sshParams.timeout, timeoutGiven, "-t")
62  || checkAndSetIntArg(pos, parameters.smallFileCount, smallFileCountGiven, "-c")
63  || checkAndSetIntArg(pos, parameters.bigFileSize, bigFileSizeGiven, "-s"))
64  continue;
65  if (checkAndSetStringArg(pos, parameters.sshParams.password, "-pwd")) {
66  if (!parameters.sshParams.privateKeyFile.isEmpty())
67  throw ArgumentErrorException(QLatin1String("-pwd and -k are mutually exclusive."));
68  parameters.sshParams.authenticationType
69  = SshConnectionParameters::AuthenticationByPassword;
70  authTypeGiven = true;
71  continue;
72  }
73  if (checkAndSetStringArg(pos, parameters.sshParams.privateKeyFile, "-k")) {
74  if (!parameters.sshParams.password.isEmpty())
75  throw ArgumentErrorException(QLatin1String("-pwd and -k are mutually exclusive."));
76  parameters.sshParams.authenticationType
77  = SshConnectionParameters::AuthenticationByKey;
78  authTypeGiven = true;
79  continue;
80  }
81  if (!checkForNoProxy(pos, parameters.sshParams.proxyType, proxySettingGiven))
82  throw ArgumentErrorException(QLatin1String("unknown option ") + m_arguments.at(pos));
83  }
84 
85  Q_ASSERT(pos <= m_arguments.count());
86  if (pos == m_arguments.count() - 1) {
87  if (!checkForNoProxy(pos, parameters.sshParams.proxyType, proxySettingGiven))
88  throw ArgumentErrorException(QLatin1String("unknown option ") + m_arguments.at(pos));
89  }
90 
91  if (!authTypeGiven)
92  throw ArgumentErrorException(QLatin1String("No authentication argument given."));
93  if (parameters.sshParams.host.isEmpty())
94  throw ArgumentErrorException(QLatin1String("No host given."));
95  if (parameters.sshParams.userName.isEmpty())
96  throw ArgumentErrorException(QLatin1String("No user name given."));
97 
98  parameters.sshParams.port = portGiven ? port : 22;
99  if (!timeoutGiven)
100  parameters.sshParams.timeout = 30;
101  if (!smallFileCountGiven)
102  parameters.smallFileCount = 1000;
103  if (!bigFileSizeGiven)
104  parameters.bigFileSize = 1024;
105  success = true;
106  } catch (ArgumentErrorException &ex) {
107  cerr << "Error: " << qPrintable(ex.error) << endl;
108  printUsage();
109  success = false;
110  }
111  return parameters;
112 }

123 bool ArgumentsCollector::checkAndSetStringArg(int &pos, QString &arg, const char *opt) const
124 {
125  if (m_arguments.at(pos) == QLatin1String(opt)) {
126  if (!arg.isEmpty()) {
127  throw ArgumentErrorException(QLatin1String("option ") + opt
128  + QLatin1String(" was given twice."));
129  }
130  arg = m_arguments.at(++pos);
131  if (arg.isEmpty() && QLatin1String(opt) != QLatin1String("-pwd"))
132  throw ArgumentErrorException(QLatin1String("empty argument not allowed here."));
133  return true;
134  }
135  return false;
136 }

138 bool ArgumentsCollector::checkAndSetIntArg(int &pos, int &val,
139  bool &alreadyGiven, const char *opt) const
140 {
141  if (m_arguments.at(pos) == QLatin1String(opt)) {
142  if (alreadyGiven) {
143  throw ArgumentErrorException(QLatin1String("option ") + opt
144  + QLatin1String(" was given twice."));
145  }
146  bool isNumber;
147  val = m_arguments.at(++pos).toInt(&isNumber);
148  if (!isNumber) {
149  throw ArgumentErrorException(QLatin1String("option ") + opt
150  + QLatin1String(" needs integer argument"));
151  }
152  alreadyGiven = true;
153  return true;
154  }
155  return false;
156 }
