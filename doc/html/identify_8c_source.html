276  goto exit;

278  break;

281  goto exit;

283  goto exit;

288  goto exit;
289  pos += bytes_read;
290  break;
291  case (5): // ProtocolVersion

293  goto exit;
294  pos += bytes_read;
295  break;
296  case (6): // AgentVersion

298  goto exit;
299  pos += bytes_read;
300  break;
301  default: // XXX_unrecognized
302  // create a comma-separated string of each unrecognized item.
303  if (!protobuf_decode_string(&in[pos], in_size - pos, (char**)&(item), &bytes_read))
304  goto exit;

311  if (!(*out)->XXX_unrecognized)

317  break;
318  }
319  }

321  retVal = 1;
