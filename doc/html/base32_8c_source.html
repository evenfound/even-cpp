96 /* Base32 encode IN array of size INLEN into OUT array of size OUTLEN.
97  If OUTLEN is less than BASE32_LENGTH(INLEN), write as many bytes as
98  possible. If OUTLEN is larger than BASE32_LENGTH(INLEN), also zero
99  terminate the output buffer. */
100 void
101 base32_encode (const char *restrict in, size_t inlen,
102  char *restrict out, size_t outlen)
103 {
104  static const char b32str[32] =
105  "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567";
106 
107  while (inlen && outlen)
108  {
109  *out++ = b32str[(to_uchar (in[0]) >> 3) & 0x1f];
110  if (!--outlen)
111  break;
112  *out++ = b32str[((to_uchar (in[0]) << 2)
113  + (--inlen ? to_uchar (in[1]) >> 6 : 0))
114  & 0x1f];
115  if (!--outlen)
116  break;
117  *out++ =
118  (inlen
119  ? b32str[(to_uchar (in[1]) >> 1) & 0x1f]
120  : '=');
121  if (!--outlen)
122  break;
123  *out++ =
124  (inlen
125  ? b32str[((to_uchar (in[1]) << 4)
126  + (--inlen ? to_uchar (in[2]) >> 4 : 0))
127  & 0x1f]
128  : '=');
129  if (!--outlen)
130  break;
131  *out++ =
132  (inlen
133  ? b32str[((to_uchar (in[2]) << 1)
134  + (--inlen ? to_uchar (in[3]) >> 7 : 0))
135  & 0x1f]
136  : '=');
137  if (!--outlen)
138  break;
139  *out++ =
140  (inlen
141  ? b32str[(to_uchar (in[3]) >> 2) & 0x1f]
142  : '=');
143  if (!--outlen)
144  break;
145  *out++ =
146  (inlen
147  ? b32str[((to_uchar (in[3]) << 3)
148  + (--inlen ? to_uchar (in[4]) >> 5 : 0))
149  & 0x1f]
150  : '=');
151  if (!--outlen)
152  break;
153  *out++ = inlen ? b32str[to_uchar (in[4]) & 0x1f] : '=';
154  if (!--outlen)
155  break;
156  if (inlen)
157  inlen--;
158  if (inlen)
159  in += 5;
160  }
161 
162  if (outlen)
163  *out = '\0';
164 }

166 /* Allocate a buffer and store zero terminated base32 encoded data
167  from array IN of size INLEN, returning BASE32_LENGTH(INLEN), i.e.,
168  the length of the encoded data, excluding the terminating zero. On
169  return, the OUT variable will hold a pointer to newly allocated
170  memory that must be deallocated by the caller. If output string
171  length would overflow, 0 is returned and OUT is set to NULL. If
172  memory allocation failed, OUT is set to NULL, and the return value
173  indicates length of the requested memory block, i.e.,
174  BASE32_LENGTH(inlen) + 1. */
175 size_t
176 base32_encode_alloc (const char *in, size_t inlen, char **out)
177 {
178  size_t outlen = 1 + BASE32_LENGTH (inlen);
179 
180  /* Check for overflow in outlen computation.
181  *
182  * If there is no overflow, outlen >= inlen.
183  *
184  * TODO Is this a sufficient check? (See the notes in base64.c.)
185  */
186  if (inlen > outlen)
187  {
188  *out = NULL;
189  return 0;
190  }
191 
192  *out = malloc (outlen);
193  if (!*out)
194  return outlen;
195 
196  base32_encode (in, inlen, *out, outlen);
197 
198  return outlen - 1;
199 }

334 /* If CTX->i is 0 or 8, there are eight or more bytes in [*IN..IN_END), and
335  none of those eight is a newline, then return *IN. Otherwise, copy up to
336  4 - CTX->i non-newline bytes from that range into CTX->buf, starting at
337  index CTX->i and setting CTX->i to reflect the number of bytes copied,
338  and return CTX->buf. In either case, advance *IN to point to the byte
339  after the last one processed, and set *N_NON_NEWLINE to the number of
340  verified non-newline bytes accessible through the returned pointer. */
341 static char *
342 get_8 (struct base32_decode_context *ctx,
343  char const *restrict *in, char const *restrict in_end,
344  size_t *n_non_newline)
345 {
346  if (ctx->i == 8)
347  ctx->i = 0;
348 
349  if (ctx->i == 0)
350  {
351  char const *t = *in;
352  if (8 <= in_end - *in && memchr (t, '\n', 8) == NULL)
353  {
354  /* This is the common case: no newline. */
355  *in += 8;
356  *n_non_newline = 8;
357  return (char *) t;
358  }
359  }
360 
361  {
362  /* Copy non-newline bytes into BUF. */
363  char const *p = *in;
364  while (p < in_end)
365  {
366  char c = *p++;
367  if (c != '\n')
368  {
369  ctx->buf[ctx->i++] = c;
370  if (ctx->i == 8)
371  break;
372  }
373  }
374 
375  *in = p;
376  *n_non_newline = ctx->i;
377  return ctx->buf;
378  }
379 }

389 /* Decode eight bytes of base32-encoded data, IN, of length INLEN
390  into the output buffer, *OUT, of size *OUTLEN bytes. Return true if
391  decoding is successful, false otherwise. If *OUTLEN is too small,
392  as many bytes as possible are written to *OUT. On return, advance
393  *OUT to point to the byte after the last one written, and decrement
394  *OUTLEN to reflect the number of bytes remaining in *OUT. */
395 static bool
396 decode_8 (char const *restrict in, size_t inlen,
397  char *restrict *outp, size_t *outleft)
398 {
399  char *out = *outp;
400  if (inlen < 8)
401  return false;
402 
403  if (!isbase32 (in[0]) || !isbase32 (in[1]) )
404  return false;
405 
406  if (*outleft)
407  {
408  *out++ = ((b32[to_uchar (in[0])] << 3)
409  | (b32[to_uchar (in[1])] >> 2));
410  --*outleft;
411  }
412 
413  if (in[2] == '=')
414  {
415  if (in[3] != '=' || in[4] != '=' || in[5] != '='
416  || in[6] != '=' || in[7] != '=')
417  return_false;
418  }
419  else
420  {
421  if (!isbase32 (in[2]) || !isbase32 (in[3]))
422  return_false;
423 
424  if (*outleft)
425  {
426  *out++ = ((b32[to_uchar (in[1])] << 6)
427  | (b32[to_uchar (in[2])] << 1)
428  | (b32[to_uchar (in[3])] >> 4));
429  --*outleft;
430  }
431 
432  if (in[4] == '=')
433  {
434  if (in[5] != '=' || in[6] != '=' || in[7] != '=')
435  return_false;
436  }
437  else
438  {
439  if (!isbase32 (in[4]))
440  return_false;
441 
442  if (*outleft)
443  {
444  *out++ = ((b32[to_uchar (in[3])] << 4)
445  | (b32[to_uchar (in[4])] >> 1));
446  --*outleft;
447  }
448 
449  if (in[5] == '=')
450  {
451  if (in[6] != '=' || in[7] != '=')
452  return_false;
453  }
454  else
455  {
456  if (!isbase32 (in[5]) || !isbase32 (in[6]))
457  return_false;
458 
459  if (*outleft)
460  {
461  *out++ = ((b32[to_uchar (in[4])] << 7)
462  | (b32[to_uchar (in[5])] << 2)
463  | (b32[to_uchar (in[6])] >> 3));
464  --*outleft;
465  }
466 
467  if (in[7] != '=')
468  {
469  if (!isbase32 (in[7]))
470  return_false;
471 
472  if (*outleft)
473  {
474  *out++ = ((b32[to_uchar (in[6])] << 5)
475  | (b32[to_uchar (in[7])]));
476  --*outleft;
477  }
478  }
479  }
480  }
481  }
482 
483  *outp = out;
484  return true;

487 /* Decode base32-encoded input array IN of length INLEN to output array
488  OUT that can hold *OUTLEN bytes. The input data may be interspersed
489  with newlines. Return true if decoding was successful, i.e. if the
490  input was valid base32 data, false otherwise. If *OUTLEN is too
491  small, as many bytes as possible will be written to OUT. On return,
492  *OUTLEN holds the length of decoded bytes in OUT. Note that as soon
493  as any non-alphabet, non-newline character is encountered, decoding
494  is stopped and false is returned. If INLEN is zero, then process
495  only whatever data is stored in CTX.
496 
497  Initially, CTX must have been initialized via base32_decode_ctx_init.
498  Subsequent calls to this function must reuse whatever state is recorded
499  in that buffer. It is necessary for when a octuple of base32 input
500  bytes spans two input buffers.
501 
502  If CTX is NULL then newlines are treated as garbage and the input
503  buffer is processed as a unit. */
504 
505 bool
506 base32_decode_ctx (struct base32_decode_context *ctx,
507  const char *restrict in, size_t inlen,
508  char *restrict out, size_t *outlen)
509 {
510  size_t outleft = *outlen;
511  bool ignore_newlines = ctx != NULL;
512  bool flush_ctx = false;
513  unsigned int ctx_i = 0;
514 
515  if (ignore_newlines)
516  {
517  ctx_i = ctx->i;
518  flush_ctx = inlen == 0;
519  }
520 
521 
522  while (true)
523  {
524  size_t outleft_save = outleft;
525  if (ctx_i == 0 && !flush_ctx)
526  {
527  while (true)
528  {
529  /* Save a copy of outleft, in case we need to re-parse this
530  block of four bytes. */
531  outleft_save = outleft;
532  if (!decode_8 (in, inlen, &out, &outleft))
533  break;
534 
535  in += 8;
536  inlen -= 8;
537  }
538  }
539 
540  if (inlen == 0 && !flush_ctx)
541  break;
542 
543  /* Handle the common case of 72-byte wrapped lines.
544  This also handles any other multiple-of-8-byte wrapping. */
545  if (inlen && *in == '\n' && ignore_newlines)
546  {
547  ++in;
548  --inlen;
549  continue;
550  }
551 
552  /* Restore OUT and OUTLEFT. */
553  out -= outleft_save - outleft;
554  outleft = outleft_save;
555 
556  {
557  char const *in_end = in + inlen;
558  char const *non_nl;
559 
560  if (ignore_newlines)
561  non_nl = get_8 (ctx, &in, in_end, &inlen);
562  else
563  non_nl = in; /* Might have nl in this case. */
564 
565  /* If the input is empty or consists solely of newlines (0 non-newlines),
566  then we're done. Likewise if there are fewer than 8 bytes when not
567  flushing context and not treating newlines as garbage. */
568  if (inlen == 0 || (inlen < 8 && !flush_ctx && ignore_newlines))
569  {
570  inlen = 0;
571  break;
572  }
573  if (!decode_8 (non_nl, inlen, &out, &outleft))
574  break;
575 
576  inlen = in_end - in;
577  }
578  }
579 
580  *outlen -= outleft;
581 
582  return inlen == 0;
583 }

585 /* Allocate an output buffer in *OUT, and decode the base32 encoded
586  data stored in IN of size INLEN to the *OUT buffer. On return, the
587  size of the decoded data is stored in *OUTLEN. OUTLEN may be NULL,
588  if the caller is not interested in the decoded length. *OUT may be
589  NULL to indicate an out of memory error, in which case *OUTLEN
590  contains the size of the memory block needed. The function returns
591  true on successful decoding and memory allocation errors. (Use the
592  *OUT and *OUTLEN parameters to differentiate between successful
593  decoding and memory error.) The function returns false if the
594  input was invalid, in which case *OUT is NULL and *OUTLEN is
595  undefined. */
596 bool
597 base32_decode_alloc_ctx (struct base32_decode_context *ctx,
598  const char *in, size_t inlen, char **out,
599  size_t *outlen)
600 {
601  /* This may allocate a few bytes too many, depending on input,
602  but it's not worth the extra CPU time to compute the exact size.
603  The exact size is 5 * inlen / 8, minus one or more bytes if the
604  input is padded with one or more "=".
605  Dividing before multiplying avoids the possibility of overflow. */
606  size_t needlen = 5 * (inlen / 8) + 5;
607 
608  *out = malloc (needlen);
609  if (!*out)
610  return true;
611 
612  if (!base32_decode_ctx (ctx, in, inlen, *out, &needlen))
613  {
614  free (*out);
615  *out = NULL;
616  return false;
617  }
618 
619  if (outlen)
620  *outlen = needlen;
621 
622  return true;
623 }
