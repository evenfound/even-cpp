63

66 
67  for(size_t i = 0; i != 8; ++i)
68  A[i] = input[i];
69

71

73  {
74  for(size_t i = n; i != 0; --i)
75  {
76  const uint32_t t = static_cast<uint32_t>((5 - j) * n + i);

173
