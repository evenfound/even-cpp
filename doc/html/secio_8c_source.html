662  * @returns the number of bytes read
663  */
664 int libp2p_secio_unencrypted_read(struct Stream* secio_stream, struct StreamMessage** msg, int timeout_secs) {
665  uint32_t buffer_size = 0;
666 
667  if (secio_stream == NULL) {
668  libp2p_logger_error("secio", "Attempted unencrypted read on invalid session.\n");
669  return 0;
670  }
671 
672  int socket_descriptor = libp2p_secio_get_socket_descriptor(secio_stream);
673 
674  if (socket_descriptor <= 0)
675  return 0;
676 
677  // first read the 4 byte integer
678  char* size = (char*)&buffer_size;
679  int left = 4;
680  int read = 0;
681  int read_this_time = 0;
682  int time_left = timeout_secs;
683  do {
684  read_this_time = socket_read(socket_descriptor, &size[read], 1, 0, timeout_secs);
685  if (read_this_time <= 0) {
686  if ( errno == EINPROGRESS ) {
687  time_left--;
688  sleep(1);
689  if (time_left == 0)
690  break;
691  continue;
692  }
693  if ( errno == EAGAIN || errno == EWOULDBLOCK ) {
694  // TODO: use epoll or select to wait for socket to be writable
695  libp2p_logger_debug("secio", "Attempted read, but got EAGAIN or EWOULDBLOCK. Code %d.\n", errno);
696  return 0;
697  } else {
698  // is this really an error?
699  if (errno != 0) {
700  libp2p_logger_error("secio", "Error in libp2p_secio_unencrypted_read: %s\n", strerror(errno));
701  return 0;
702  }
703  else {
704  libp2p_logger_error("secio", "Stream has been shut down from other end.\n");
705  libp2p_secio_set_socket_descriptor(secio_stream);
706  return 0;
707  }
708  }
709  } else {
710  left = left - read_this_time;
711  read += read_this_time;
712  }
713  } while (left > 0);
714  buffer_size = ntohl(buffer_size);
715  if (buffer_size == 0) {
716  libp2p_logger_error("secio", "unencrypted read buffer size is 0.\n");
717  return 0;
718  }
719 
720  // now read the number of bytes we've found, minus the 4 that we just read
721  left = buffer_size;
722  read = 0;
723  read_this_time = 0;
724  *msg = libp2p_stream_message_new();
725  struct StreamMessage* m = *msg;
726  if (m == NULL) {
727  libp2p_logger_error("secio", "Unable to allocate memory for the incoming message. Size: %ulld", left);
728  return 0;
729  }
730  m->data_size = left;
731  m->data = (uint8_t*) malloc(left);
732  unsigned char* ptr = m->data;
733  time_left = timeout_secs;
734  do {
735  read_this_time = socket_read(socket_descriptor, (char*)&ptr[read], left, 0, timeout_secs);
736  if (read_this_time < 0) {
737  if (errno == EINPROGRESS) {
738  sleep(1);
739  time_left--;
740  if (time_left == 0)
741  break;
742  continue;
743  }
744  read_this_time = 0;
745  if ( (errno == EAGAIN) || (errno == EWOULDBLOCK)) {
746  // TODO: use epoll or select to wait for socket to be writable
747  } else {
748  libp2p_logger_error("secio", "read from socket returned %d.\n", errno);
749  return 0;
750  }
751  } else if (read_this_time == 0) {
752  // socket_read returned 0, which it shouldn't
753  libp2p_logger_error("secio", "socket_read returned 0 trying to read from stream %d.\n", socket_descriptor);
754  return 0;
755  }
756  left = left - read_this_time;
757  read += read_this_time;
758  } while (left > 0);
759 
760  m->data_size = buffer_size;
761  return buffer_size;
762 }

875 int libp2p_secio_decrypt(struct SessionContext* session, const unsigned char* incoming, size_t incoming_size, struct StreamMessage** outgoing) {
