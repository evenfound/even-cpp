197 class PGP_S2K_Iter_Test final : public Test
198  {
199  public:
200  std::vector<Test::Result> run() override
201  {
202  Test::Result result("PGP_S2K iteration encoding");
203 
204  // The maximum representable iteration count
205  const size_t max_iter = 65011712;
206 
207  result.test_eq("Encoding of large value accepted",
208  Botan::RFC4880_encode_count(max_iter * 2), size_t(255));
209  result.test_eq("Encoding of small value accepted",
210  Botan::RFC4880_encode_count(0), size_t(0));
211 
212  for(size_t c = 0; c != 256; ++c)
213  {
214  const size_t dec = Botan::RFC4880_decode_count(static_cast<uint8_t>(c));
215  const size_t comp_dec = (16 + (c & 0x0F)) << ((c >> 4) + 6);
216  result.test_eq("Decoded value matches PGP formula", dec, comp_dec);
217  }
218 
219  uint8_t last_enc = 0;
220 
221  for(size_t i = 0; i <= max_iter; i += 64)
222  {
223  const uint8_t enc = Botan::RFC4880_encode_count(i);
224  result.test_lte("Encoded value non-decreasing", last_enc, enc);
225 
226  /*
227  The iteration count as encoded may not be exactly the
228  value requested, but should never be less
229  */
230  const size_t dec = Botan::RFC4880_decode_count(enc);
231  result.test_gte("Decoded value is >= requested", dec, i);
232 
233  last_enc = enc;
234  }
235 
236  return std::vector<Test::Result>{result};
237  }
