320 QWebSocket *QWebSocket::upgradeFrom(QTcpSocket *tcpSocket, const QString& serverName, bool isSecure, QObject *parent )
321 {
322  QWebSocket* wSocket = NULL;
323  qDebug() << "HTTP signature detected, using WebSocket handler";
324  QWebSocketHandshakeRequest request(tcpSocket->peerPort(), isSecure );
325  QTextStream textStream(tcpSocket);
326  request.readHandshake(textStream, MAX_HEADERLINE_LENGTH, MAX_HEADERLINES);
327 
328  if (request.isValid())
329  {
330 
331  QWebSocketCorsAuthenticator corsAuthenticator(request.origin());
332  //TODO Q_EMIT q->originAuthenticationRequired(&corsAuthenticator);
333  QList<QWebSocketProtocol::Version> supportedVersions;
334  supportedVersions << QWebSocketProtocol::currentVersion(); //we only support V13
335  QStringList supportedProtocols; //no protocols are currently supported
336  QStringList supportedExtensions; //no extensions are currently supported
337  QWebSocketHandshakeResponse response(request,
338  serverName,
339  corsAuthenticator.allowed(),
340  supportedVersions,
341  supportedProtocols,
342  supportedExtensions );
343 
344  if( response.isValid() )
345  {
346  QTextStream httpStream(tcpSocket);
347  httpStream << response;
348  httpStream.flush();
349 
350  if( response.canUpgrade() )
351  {
352  wSocket = QWebSocketPrivate::upgradeFrom( tcpSocket, request, response, parent);
353  }
354  else
355  {
356  qDebug() << "ERROR can't upgrade to websocket: " << response.errorString();
357  }
358  }
359  else
360  {
361  qDebug() << "ERROR: " << tr("Invalid response received.");
362  }
363  }
364 
365  return wSocket;
366 }
