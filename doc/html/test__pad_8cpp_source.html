22  Test::Result run_one_test(const std::string& header, const VarMap& vars) override
23  {
24  const std::vector<uint8_t> input = vars.get_req_bin("In");
25  const std::vector<uint8_t> expected = vars.get_opt_bin("Out");
26  const size_t block_size = vars.get_req_sz("Blocksize");
27 
28  std::string algo = header;
29 
30  auto underscore = algo.find('_');
31  if(underscore != std::string::npos)
32  {
33  if(algo.substr(underscore + 1, std::string::npos) != "Invalid")
34  {
35  throw Test_Error("Unexpected padding header " + header);
36  }
37  algo = algo.substr(0, underscore);
38  }
39 
40  Test::Result result(algo);
41 
42  std::unique_ptr<Botan::BlockCipherModePaddingMethod> pad(Botan::get_bc_pad(algo));
43 
44  if(!pad)
45  {
46  result.test_failure("Invalid padding method: " + algo);
47  return result;
48  }
49 
50  if(expected.empty())
51  {
52  // This is an unpad an invalid input and ensure we reject
53  if(pad->unpad(input.data(), block_size) != block_size)
54  {
55  result.test_failure("Did not reject invalid padding", Botan::hex_encode(input));
56  }
57  else
58  {
59  result.test_success("Rejected invalid padding");
60  }
61  }
62  else
63  {
64  // This is a pad plaintext and unpad valid padding round trip test
65  Botan::secure_vector<uint8_t> buf(input.begin(), input.end());
66  pad->add_padding(buf, input.size() % block_size, block_size);
67  result.test_eq("pad", buf, expected);
68 
69  buf.assign(expected.begin(), expected.end());
70 
71  const size_t last_block = (buf.size() < block_size) ? 0 : buf.size() - block_size;
72  const size_t pad_bytes = block_size - pad->unpad(&buf[last_block], block_size);
73  buf.resize(buf.size() - pad_bytes); // remove padding
74  result.test_eq("unpad", buf, input);
75  }
76 
77  return result;
78  }
