69  static Botan::TLS::Protocol_Version::Version_Code tls_version_from_str(const std::string& str)
70  {
71  if(str == "tls1.2" || str == "TLS1.2" || str == "TLS-1.2")
72  {
73  return Botan::TLS::Protocol_Version::TLS_V12;
74  }
75  else if(str == "tls1.1" || str == "TLS1.1" || str == "TLS-1.1")
76  {
77  return Botan::TLS::Protocol_Version::TLS_V11;
78  }
79  else if(str == "tls1.0" || str == "TLS1.1" || str == "TLS-1.1")
80  {
81  return Botan::TLS::Protocol_Version::TLS_V10;
82  }
83  if(str == "dtls1.2" || str == "DTLS1.2" || str == "DTLS-1.2")
84  {
85  return Botan::TLS::Protocol_Version::DTLS_V12;
86  }
87  else if(str == "dtls1.0" || str == "DTLS1.0" || str == "DTLS-1.0")
88  {
89  return Botan::TLS::Protocol_Version::DTLS_V10;
90  }
91  else
92  {
93  throw CLI_Error("Unknown TLS version '" + str + "'");
94  }
95  }

107  void go() override
108  {
109  const std::string policy_type = get_arg("policy");
110  const Botan::TLS::Protocol_Version version(tls_version_from_str(get_arg("version")));
111  const bool with_srp = false; // fixme
112 
113  std::unique_ptr<Botan::TLS::Policy> policy;
114 
115  if(policy_type == "default")
116  {
117  policy.reset(new Botan::TLS::Policy);
118  }
119  else if(policy_type == "suiteb_128")
120  {
121  policy.reset(new Botan::TLS::NSA_Suite_B_128);
122  }
123  else if(policy_type == "suiteb_192")
124  {
125  policy.reset(new Botan::TLS::NSA_Suite_B_192);
126  }
127  else if(policy_type == "strict")
128  {
129  policy.reset(new Botan::TLS::Strict_Policy);
130  }
131  else if(policy_type == "all")
132  {
133  policy.reset(new TLS_All_Policy);
134  }
135  else
136  {
137  const std::string policy_txt = slurp_file_as_str(policy_type);
138  policy.reset(new Botan::TLS::Text_Policy(policy_txt));
139  }
140 
141  if(policy->acceptable_protocol_version(version) == false)
142  {
143  error_output() << "Error: the policy specified does not allow the given TLS version\n";
144  return;
145  }
146 
147  for(uint16_t suite_id : policy->ciphersuite_list(version, with_srp))
148  {

157 class TLS_Client_Hello_Reader final : public Command
158  {
159  public:
160  TLS_Client_Hello_Reader()
161  : Command("tls_client_hello --hex input") {}
162 
163  std::string group() const override
164  {
165  return "tls";
166  }

168  std::string description() const override
169  {
170  return "Parse a TLS client hello message";
171  }
172 
173  void go() override
174  {
175  const std::string input_file = get_arg("input");
176  std::vector<uint8_t> input;
177 
178  if(flag_set("hex"))
179  {
180  input = Botan::hex_decode(slurp_file_as_str(input_file));
181  }
182  else
183  {
184  input = slurp_file(input_file);
185  }
186 
187  if(input.size() < 45)
188  {
189  error_output() << "Input too short to be valid\n";
190  return;
191  }
192 
193  // Input also contains the record layer header, strip it
194  if(input[0] == 22)
195  {
196  const size_t len = Botan::make_uint16(input[3], input[4]);
197 
198  if(input.size() != len + 5)
199  {
200  error_output() << "Record layer length invalid\n";
201  return;
202  }
203 
204  input = std::vector<uint8_t>(input.begin() + 5, input.end());
205  }
206 
207  // Assume the handshake header is there, strip it
208  if(input[0] == 1)
209  {
210  const size_t hs_len = Botan::make_uint32(0, input[1], input[2], input[3]);
211 
212  if(input.size() != hs_len + 4)
213  {
214  error_output() << "Handshake layer length invalid\n";
215  return;
216  }
217 
218  input = std::vector<uint8_t>(input.begin() + 4, input.end());
219  }
220 
221  try
222  {
223  Botan::TLS::Client_Hello hello(input);
224 
225  output() << format_hello(hello);
226  }
227  catch(std::exception& e)
228  {
229  error_output() << "Parsing client hello failed: " << e.what() << "\n";
230  }
231  }
232 
233  private:
234  std::string format_hello(const Botan::TLS::Client_Hello& hello)
235  {
236  std::ostringstream oss;
237  oss << "Version: " << hello.version().to_string() << "\n"
238  << "Random: " << Botan::hex_encode(hello.random()) << "\n";
239 
240  if(!hello.session_id().empty())
241  oss << "SessionID: " << Botan::hex_encode(hello.session_id()) << "\n";
242  for(uint16_t csuite_id : hello.ciphersuites())
243  oss << "Cipher: " << Botan::TLS::Ciphersuite::by_id(csuite_id).to_string() << "\n";
244 
245  oss << "Supported signature schemes: ";
246 
247  if(hello.signature_schemes().empty())
248  {
249  oss << "Did not send signature_algorithms extension\n";
250  }
251  else
252  {
253  for(Botan::TLS::Signature_Scheme scheme : hello.signature_schemes())
254  oss << sig_scheme_to_string(scheme) << " ";
255  oss << "\n";
256  }
257 
258  std::map<std::string, bool> hello_flags;
259  hello_flags["ALPN"] = hello.supports_alpn();
260  hello_flags["Encrypt Then Mac"] = hello.supports_encrypt_then_mac();
261  hello_flags["Extended Master Secret"] = hello.supports_extended_master_secret();
262  hello_flags["Session Ticket"] = hello.supports_session_ticket();
263 
264  for(auto&& i : hello_flags)
265  oss << "Supports " << i.first << "? " << (i.second ? "yes" : "no") << "\n";
266 
267  return oss.str();
268  }
269  };
