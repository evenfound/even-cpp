37  libp2p_logger_debug("swarm", "Attempting to get read lock.\n");

39  libp2p_logger_debug("swarm", "Got read lock.\n");
40  if (!stream->read(stream->stream_context, &results, 1)) {
41  libp2p_logger_debug("swarm", "Releasing read lock\n");
42  pthread_mutex_unlock(stream->socket_mutex);

45  return -1;

48  return retVal;
49  }
50  libp2p_logger_debug("swarm", "Releasing read lock.\n");
51  pthread_mutex_unlock(stream->socket_mutex);
52  if (results != NULL) {
53  libp2p_logger_debug("swarm", "Attempting to marshal %d bytes from network.\n", results->data_size);
54  retVal = libp2p_protocol_marshal(results, stream, protocol_handlers);
55  libp2p_logger_debug("swarm", "The return value from the attempt to marshal %d bytes was %d.\n", results->data_size, retVal);
56  libp2p_stream_message_free(results);
