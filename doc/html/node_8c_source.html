172 int ipfs_node_link_protobuf_decode(unsigned char* buffer, size_t buffer_length, struct NodeLink** node_link) {
173  size_t pos = 0;
174  int retVal = 0;
175  struct NodeLink* link = NULL;
176 
177  // allocate memory for object
178  if (ipfs_node_link_new(node_link) == 0)
179  goto exit;
180 
181  link = *node_link;
182 
183  while(pos < buffer_length) {
184  size_t bytes_read = 0;
185  int field_no;
186  enum WireType field_type;
187  if (protobuf_decode_field_and_type(&buffer[pos], buffer_length, &field_no, &field_type, &bytes_read) == 0) {
188  goto exit;
189  }
190  pos += bytes_read;
191  switch(field_no) {
192  case (1): { // hash
193  size_t hash_size = 0;
194  unsigned char* hash;
195  if (protobuf_decode_length_delimited(&buffer[pos], buffer_length - pos, (char**)&hash, &hash_size, &bytes_read) == 0)
196  goto exit;
197  if (hash_size < 2) {
198  goto exit;
199  }
200  link->hash_size = hash_size - 2;
201  link->hash = (unsigned char*)malloc(link->hash_size);
202  if (link->hash == NULL) {
203  goto exit;
204  }
205  memcpy((char*)link->hash, (char*)&hash[2], link->hash_size);
206  free(hash);
207  pos += bytes_read;
208  break;
209  }
210  case (2): { // name
211  if (protobuf_decode_string(&buffer[pos], buffer_length - pos, &link->name, &bytes_read) == 0)
212  goto exit;
213  pos += bytes_read;
214  break;
215  }
216  case (3): { // t_size
217  if (protobuf_decode_varint(&buffer[pos], buffer_length - pos, (unsigned long long*)&link->t_size, &bytes_read) == 0)
218  goto exit;
219  pos += bytes_read;
220  break;
221  }
222  }
223  }
224 
225  retVal = 1;
226 
227 exit:
228  if (retVal == 0) {
229  if (link != NULL)
230  ipfs_node_link_free(link);
231  }
232 
233  return retVal;
234 }

291 /***
292  * Decode a stream of bytes into a Node structure
293  * @param buffer where to get the bytes from
294  * @param buffer_length the length of buffer
295  * @param node pointer to the Node to be created
296  * @returns true(1) on success
297  */
298 int ipfs_hashtable_node_protobuf_decode(unsigned char* buffer, size_t buffer_length, struct HashtableNode** node) {
299  /*
300  * Field 1: data
301  * Field 2: link
302  */
303  size_t pos = 0;
304  int retVal = 0;
305  unsigned char* temp_buffer = NULL;
306  size_t temp_size;
307  struct NodeLink* temp_link = NULL;
308 
309  if (buffer_length == 0)
310  goto exit;
311 
312  if (ipfs_hashtable_node_new(node) == 0)
313  goto exit;
314 
315  while(pos < buffer_length) {
316  size_t bytes_read = 0;
317  int field_no;
318  enum WireType field_type;
319  if (protobuf_decode_field_and_type(&buffer[pos], buffer_length, &field_no, &field_type, &bytes_read) == 0) {
320  goto exit;
321  }
322  pos += bytes_read;
323  switch(field_no) {
324  case (1): { // data
325  if (protobuf_decode_length_delimited(&buffer[pos], buffer_length - pos, (char**)&((*node)->data), &((*node)->data_size), &bytes_read) == 0)
326  goto exit;
327  pos += bytes_read;
328  break;
329  }
330  case (2): {// links
331  if (protobuf_decode_length_delimited(&buffer[pos], buffer_length - pos, (char**)&temp_buffer, &temp_size, &bytes_read) == 0)
332  goto exit;
333  pos += bytes_read;
334  if (ipfs_node_link_protobuf_decode(temp_buffer, temp_size, &temp_link)) {
335  ipfs_hashtable_node_add_link(*node, temp_link);
336  }
337  free(temp_buffer);
338  temp_buffer = NULL;
339  break;
340  }
341  }
342  }
343 
344  retVal = 1;
345 
346 exit:
347  if (retVal == 0) {
348  ipfs_hashtable_node_free(*node);
349  }
350  if (temp_buffer != NULL)
351  free(temp_buffer);
352 
353  return retVal;
354 }
