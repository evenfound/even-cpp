410 void Blowfish::generate_sbox(secure_vector<uint32_t>& box,
411  uint32_t& L, uint32_t& R,
412  const uint8_t salt[],
413  size_t salt_length,
414  size_t salt_off) const

417  {
418  if(salt_length > 0)

420  L ^= load_be<uint32_t>(salt, (i + salt_off) % (salt_length / 4));
421  R ^= load_be<uint32_t>(salt, (i + salt_off + 1) % (salt_length / 4));
