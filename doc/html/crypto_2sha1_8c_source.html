213 void SHA1_Update(SHA1_CTX* context, const uint8_t* data, const size_t len)
214 {
215  size_t i, j;
216 
217 #ifdef VERBOSE
218  SHAPrintContext(context, "before");
219 #endif
220 
221  j = (context->count[0] >> 3) & 63;
222  if ((context->count[0] += len << 3) < (len << 3)) context->count[1]++;
223  context->count[1] += (len >> 29);
224  if ((j + len) > 63) {
225  memcpy(&context->buffer[j], data, (i = 64-j));
226  SHA1_Transform(context->state, context->buffer);
227  for ( ; i + 63 < len; i += 64) {
228  SHA1_Transform(context->state, data + i);
229  }
230  j = 0;
231  }
232  else i = 0;
233  memcpy(&context->buffer[j], &data[i], len - i);
234 
235 #ifdef VERBOSE
236  SHAPrintContext(context, "after ");
237 #endif
238 }

241 /* Add padding and return the message digest. */
242 void SHA1_Final(SHA1_CTX* context, uint8_t digest[SHA1_DIGEST_SIZE])
243 {
244  uint32_t i;
245  uint8_t finalcount[8];
246 
247  for (i = 0; i < 8; i++) {
248  finalcount[i] = (unsigned char)((context->count[(i >= 4 ? 0 : 1)]
249  >> ((3-(i & 3)) * 8) ) & 255); /* Endian independent */
250  }
251  SHA1_Update(context, (uint8_t *)"\200", 1);
252  while ((context->count[0] & 504) != 448) {
253  SHA1_Update(context, (uint8_t *)"\0", 1);
254  }
255  SHA1_Update(context, finalcount, 8); /* Should cause a SHA1_Transform() */
256  for (i = 0; i < SHA1_DIGEST_SIZE; i++) {
257  digest[i] = (uint8_t)
258  ((context->state[i>>2] >> ((3-(i & 3)) * 8) ) & 255);
259  }
260 
261  /* Wipe variables */
262  i = 0;
263  memset(context->buffer, 0, 64);
264  memset(context->state, 0, 20);
265  memset(context->count, 0, 8);
266  memset(finalcount, 0, 8); /* SWR */
267 
268 #ifdef SHA1HANDSOFF /* make SHA1Transform overwrite its own static vars */
269  SHA1_Transform(context->state, context->buffer);
270 #endif
271 }
