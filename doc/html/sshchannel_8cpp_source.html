115 void AbstractSshChannel::flushSendBuffer()
116 {
117  while (true) {
118  const quint32 bytesToSend = qMin(m_remoteMaxPacketSize,
119  qMin<quint32>(m_remoteWindowSize, m_sendBuffer.size()));
120  if (bytesToSend == 0)
121  break;
122  const QByteArray &data = m_sendBuffer.left(bytesToSend);
123  m_sendFacility.sendChannelDataPacket(m_remoteChannel, data);
124  m_sendBuffer.remove(0, bytesToSend);
125  m_remoteWindowSize -= bytesToSend;
126  }
127 }

129 void AbstractSshChannel::handleOpenSuccess(quint32 remoteChannelId,
130  quint32 remoteWindowSize, quint32 remoteMaxPacketSize)
131 {
132  if (m_state != SessionRequested) {
133  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_PROTOCOL_ERROR,
134  "Invalid SSH_MSG_CHANNEL_OPEN_CONFIRMATION packet.");
135  }
136  m_timeoutTimer->stop();
137 
138  if (remoteMaxPacketSize < MinMaxPacketSize) {
139  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_PROTOCOL_ERROR,
140  "Maximum packet size too low.");
141  }
142 
143 #ifdef CREATOR_SSH_DEBUG
144  qDebug("Channel opened. remote channel id: %u, remote window size: %u, "
145  "remote max packet size: %u",
146  remoteChannelId, remoteWindowSize, remoteMaxPacketSize);
147 #endif
148  m_remoteChannel = remoteChannelId;
149  m_remoteWindowSize = remoteWindowSize;
150  m_remoteMaxPacketSize = remoteMaxPacketSize - sizeof(quint32) - sizeof m_remoteChannel - 1;
151  // Original value includes packet type, channel number and length field for string.
152  setChannelState(SessionEstablished);
153  handleOpenSuccessInternal();
154 }

206 void AbstractSshChannel::handleChannelRequest(const SshIncomingPacket &packet)
207 {
208  checkChannelActive();

210  if (requestType == SshIncomingPacket::ExitStatusType)
211  handleExitStatus(packet.extractChannelExitStatus());
212  else if (requestType == SshIncomingPacket::ExitSignalType)
213  handleExitSignal(packet.extractChannelExitSignal());
214  else if (requestType != "eow@openssh.com") // Suppress warning for this one, as it's sent all the time.

218 int AbstractSshChannel::handleChannelOrExtendedChannelData(const QByteArray &data)
219 {
220  checkChannelActive();
221 
222  const int bytesToDeliver = qMin<quint32>(data.size(), maxDataSize());
223  if (bytesToDeliver != data.size())
224  qWarning("Misbehaving server does not respect local window, clipping.");
225 
226  m_localWindowSize -= bytesToDeliver;
227  if (m_localWindowSize < MaxPacketSize) {
228  m_localWindowSize += MaxPacketSize;
229  m_sendFacility.sendWindowAdjustPacket(m_remoteChannel,
230  MaxPacketSize);
231  }
232  return bytesToDeliver;
233 }

235 void AbstractSshChannel::closeChannel()
236 {
237  if (m_state == CloseRequested) {
238  m_timeoutTimer->stop();
239  } else if (m_state != Closed) {
240  if (m_state == Inactive) {
241  setChannelState(Closed);
242  } else {
243  setChannelState(CloseRequested);
244  m_sendFacility.sendChannelEofPacket(m_remoteChannel);
245  m_sendFacility.sendChannelClosePacket(m_remoteChannel);
246  }
247  }
