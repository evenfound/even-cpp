239 int SftpFileSystemModel::rowCount(const QModelIndex &parent) const
240 {
241  if (!d->rootNode)
242  return 0;
243  if (!parent.isValid())
244  return 1;
245  if (parent.column() != 0)
246  return 0;
247  SftpDirNode * const dirNode = indexToDirNode(parent);
248  if (!dirNode)
249  return 0;
250  if (dirNode->lsState != SftpDirNode::LsNotYetCalled)
251  return dirNode->children.count();

253  dirNode->lsState = SftpDirNode::LsRunning;
254  return 0;

313 void SftpFileSystemModel::handleFileInfo(SftpJobId jobId, const QList<SftpFileInfo> &fileInfoList)
314 {
315  if (jobId == d->statJobId) {
316  QSSH_ASSERT_AND_RETURN(!d->rootNode);
317  beginInsertRows(QModelIndex(), 0, 0);
318  d->rootNode = new SftpDirNode;
319  d->rootNode->path = d->rootDirectory;
320  d->rootNode->fileInfo = fileInfoList.first();
321  d->rootNode->fileInfo.name = d->rootDirectory == QLatin1String("/")
322  ? d->rootDirectory : QFileInfo(d->rootDirectory).fileName();
323  endInsertRows();
324  return;
325  }
326  SftpDirNode * const parentNode = d->lsOps.value(jobId);
327  QSSH_ASSERT_AND_RETURN(parentNode);
328  QList<SftpFileInfo> filteredList;
329  foreach (const SftpFileInfo &fi, fileInfoList) {
330  if (fi.name != QLatin1String(".") && fi.name != QLatin1String(".."))
331  filteredList << fi;
332  }
333  if (filteredList.isEmpty())
334  return;
335 
336  // In theory beginInsertRows() should suffice, but that fails to have an effect
337  // if rowCount() returned 0 earlier.
338  emit layoutAboutToBeChanged();
339 
340  foreach (const SftpFileInfo &fileInfo, filteredList) {
341  SftpFileNode *childNode;
342  if (fileInfo.type == FileTypeDirectory)
343  childNode = new SftpDirNode;
344  else
345  childNode = new SftpFileNode;
346  childNode->path = parentNode->path;
347  if (!childNode->path.endsWith(QLatin1Char('/')))
348  childNode->path += QLatin1Char('/');
349  childNode->path += fileInfo.name;
350  childNode->fileInfo = fileInfo;
351  childNode->parent = parentNode;
352  parentNode->children << childNode;
353  }
354  emit layoutChanged(); // Should be endInsertRows(), see above.
355 }

357 void SftpFileSystemModel::handleSftpJobFinished(SftpJobId jobId, const QString &errorMessage)
358 {
359  if (jobId == d->statJobId) {
360  d->statJobId = SftpInvalidJob;
361  if (!errorMessage.isEmpty())
362  emit sftpOperationFailed(tr("Error getting 'stat' info about '%1': %2")
363  .arg(rootDirectory(), errorMessage));
364  return;
365  }
366 
367  DirNodeHash::Iterator it = d->lsOps.find(jobId);
368  if (it != d->lsOps.end()) {
369  QSSH_ASSERT(it.value()->lsState == SftpDirNode::LsRunning);
370  it.value()->lsState = SftpDirNode::LsFinished;
371  if (!errorMessage.isEmpty())
372  emit sftpOperationFailed(tr("Error listing contents of directory '%1': %2")
373  .arg(it.value()->path, errorMessage));
374  d->lsOps.erase(it);
375  return;
376  }
377 
378  const int jobIndex = d->externalJobs.indexOf(jobId);
379  QSSH_ASSERT_AND_RETURN(jobIndex != -1);
380  d->externalJobs.removeAt(jobIndex);
381  emit sftpOperationFinished(jobId, errorMessage);
382 }
