86 QByteArray SshCapabilities::findBestMatch(const QList<QByteArray> &myCapabilities,
87  const QList<QByteArray> &serverCapabilities)
88 {
89  foreach (const QByteArray &myCapability, myCapabilities) {
90  if (serverCapabilities.contains(myCapability))
91  return myCapability;
92  }
93 
94  throw SshServerException(SSH_DISCONNECT_KEY_EXCHANGE_FAILED,
95  "Server and client capabilities do not match.",
96  QCoreApplication::translate("SshConnection",
97  "Server and client capabilities don't match. "
98  "Client list was: %1.\nServer list was %2.")
99  .arg(QString::fromLocal8Bit(listAsByteArray(myCapabilities).data()))
100  .arg(QString::fromLocal8Bit(listAsByteArray(serverCapabilities).data())));
