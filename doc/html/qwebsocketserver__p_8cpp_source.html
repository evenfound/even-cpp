86 void QWebSocketServerPrivate::init()
87 {
88  if (m_secureMode == NonSecureMode) {
89  m_pTcpServer = new QTcpServer(q_ptr);
90  if (Q_LIKELY(m_pTcpServer))
91  QObjectPrivate::connect(m_pTcpServer, &QTcpServer::newConnection,
92  this, &QWebSocketServerPrivate::onNewConnection);
93  else
94  qFatal("Could not allocate memory for tcp server.");
95  } else {
96 #ifndef QT_NO_SSL
97  QSslServer *pSslServer = new QSslServer(q_ptr);
98  m_pTcpServer = pSslServer;
99  if (Q_LIKELY(m_pTcpServer)) {
100  QObjectPrivate::connect(pSslServer, &QSslServer::newEncryptedConnection,
101  this, &QWebSocketServerPrivate::onNewConnection,
102  Qt::QueuedConnection);
103  QObject::connect(pSslServer, &QSslServer::peerVerifyError,
104  q_ptr, &QWebSocketServer::peerVerifyError);
105  QObject::connect(pSslServer, &QSslServer::sslErrors,
106  q_ptr, &QWebSocketServer::sslErrors);
107  QObject::connect(pSslServer, &QSslServer::preSharedKeyAuthenticationRequired,
108  q_ptr, &QWebSocketServer::preSharedKeyAuthenticationRequired);
109  }
110 #else
111  qFatal("SSL not supported on this platform.");
112 #endif
113  }
114  QObject::connect(m_pTcpServer, &QTcpServer::acceptError, q_ptr, &QWebSocketServer::acceptError);
115 }

127 void QWebSocketServerPrivate::close(bool aboutToDestroy)
128 {
129  Q_Q(QWebSocketServer);
130  m_pTcpServer->close();
131  while (!m_pendingConnections.isEmpty()) {
132  QWebSocket *pWebSocket = m_pendingConnections.dequeue();
133  pWebSocket->close(QWebSocketProtocol::CloseCodeGoingAway,
134  QWebSocketServer::tr("Server closed."));
135  pWebSocket->deleteLater();
136  }
137  if (!aboutToDestroy) {
138  //emit signal via the event queue, so the server gets time
139  //to process any hanging events, like flushing buffers aso
140  QMetaObject::invokeMethod(q, "closed", Qt::QueuedConnection);
141  }
142 }

385 void QWebSocketServerPrivate::onNewConnection()
386 {
387  while (m_pTcpServer->hasPendingConnections()) {
388  QTcpSocket *pTcpSocket = m_pTcpServer->nextPendingConnection();
389  //use a queued connection because a QSslSocket
390  //needs the event loop to process incoming data
391  //if not queued, data is incomplete when handshakeReceived is called
392  QObjectPrivate::connect(pTcpSocket, &QTcpSocket::readyRead,
393  this, &QWebSocketServerPrivate::handshakeReceived,
394  Qt::QueuedConnection);
395  }
396 }

413 void QWebSocketServerPrivate::handshakeReceived()
414 {
415  if (Q_UNLIKELY(!currentSender)) {
416  return;
417  }
418  QTcpSocket *pTcpSocket = qobject_cast<QTcpSocket*>(currentSender->sender);
419  if (Q_UNLIKELY(!pTcpSocket)) {
420  return;
421  }
422  //When using Google Chrome the handshake in received in two parts.
423  //Therefore, the readyRead signal is emitted twice.
424  //This is a guard against the BEAST attack.
425  //See: https://www.imperialviolet.org/2012/01/15/beastfollowup.html
426  //For Safari, the handshake is delivered at once
427  //FIXME: For FireFox, the readyRead signal is never emitted
428  //This is a bug in FireFox (see https://bugzilla.mozilla.org/show_bug.cgi?id=594502)
429  if (!pTcpSocket->canReadLine()) {
430  return;
431  }
432  disconnect(pTcpSocket, &QTcpSocket::readyRead,
433  this, &QWebSocketServerPrivate::handshakeReceived);
434  Q_Q(QWebSocketServer);
435  bool success = false;
436  bool isSecure = false;
437 
438  if (m_pendingConnections.length() >= maxPendingConnections()) {
439  pTcpSocket->close();
440  pTcpSocket->deleteLater();
441  setError(QWebSocketProtocol::CloseCodeAbnormalDisconnection,
442  QWebSocketServer::tr("Too many pending connections."));
443  return;
444  }
445 
446  QWebSocketHandshakeRequest request(pTcpSocket->peerPort(), isSecure);
447  QTextStream textStream(pTcpSocket);
448  request.readHandshake(textStream, MAX_HEADERLINE_LENGTH, MAX_HEADERLINES);
449 
450  if (request.isValid()) {
451  QWebSocketCorsAuthenticator corsAuthenticator(request.origin());
452  Q_EMIT q->originAuthenticationRequired(&corsAuthenticator);
453 
454  QWebSocketHandshakeResponse response(request,
455  m_serverName,
456  corsAuthenticator.allowed(),
457  supportedVersions(),
458  supportedProtocols(),
459  supportedExtensions());
460 
461  if (response.isValid()) {
462  QTextStream httpStream(pTcpSocket);
463  httpStream << response;
464  httpStream.flush();
465 
466  if (response.canUpgrade()) {
467  QWebSocket *pWebSocket = QWebSocketPrivate::upgradeFrom(pTcpSocket,
468  request,
469  response);
470  if (pWebSocket) {
471  addPendingConnection(pWebSocket);
472  Q_EMIT q->newConnection();
473  success = true;
474  } else {
475  setError(QWebSocketProtocol::CloseCodeAbnormalDisconnection,
476  QWebSocketServer::tr("Upgrade to WebSocket failed."));
477  }
478  }
479  else {
480  setError(response.error(), response.errorString());
481  }
482  } else {
483  setError(QWebSocketProtocol::CloseCodeProtocolError,
484  QWebSocketServer::tr("Invalid response received."));
485  }
486  }
487  if (!success) {
488  pTcpSocket->close();
489  }
490 }
