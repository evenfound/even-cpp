196 void SshRemoteProcess::sendSignal(Signal signal)
197 {
198  try {
199  if (isRunning()) {
200  const char *signalString = 0;
201  for (size_t i = 0; i < sizeof signalMap/sizeof *signalMap && !signalString; ++i) {
202  if (signalMap[i].signalEnum == signal)
203  signalString = signalMap[i].signalString;
204  }
205  QSSH_ASSERT_AND_RETURN(signalString);
206  d->m_sendFacility.sendChannelSignalPacket(d->remoteChannel(), signalString);
207  }
208  } catch (Botan::Exception &e) {
209  setErrorString(QString::fromAscii(e.what()));
210  d->closeChannel();
211  }
212 }

362 void SshRemoteProcessPrivate::handleExitSignal(const SshChannelExitSignal &signal)
363 {
364 #ifdef CREATOR_SSH_DEBUG
365  qDebug("Exit due to signal %s", signal.signal.data());
366 #endif
367 
368  for (size_t i = 0; i < sizeof signalMap/sizeof *signalMap; ++i) {
369  if (signalMap[i].signalString == signal.signal) {
370  m_signal = signalMap[i].signalEnum;
371  m_procState = Exited;
372  m_proc->setErrorString(tr("Process killed by signal"));
373  return;
374  }
375  }
376 
377  throw SshServerException(SSH_DISCONNECT_PROTOCOL_ERROR, "Invalid signal",
378  tr("Server sent invalid signal '%1'").arg(QString::fromUtf8(signal.signal)));
379 }
