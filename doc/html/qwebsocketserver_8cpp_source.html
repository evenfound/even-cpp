526 QUrl QWebSocketServer::serverUrl() const
527 {
528  QUrl url;
529 
530  if (!isListening()) {
531  return url;
532  }
533 
534  switch (secureMode()) {
535  case NonSecureMode:
536  url.setScheme(QStringLiteral("ws"));
537  break;
538  #ifndef QT_NO_SSL
539  case SecureMode:
540  url.setScheme(QStringLiteral("wss"));
541  break;
542  #endif
543  }
544 
545  url.setPort(serverPort());
546 
547  if (serverAddress() == QHostAddress(QHostAddress::Any)) {
548  // NOTE: On Windows at least, clients cannot connect to QHostAddress::Any
549  // so in that case we always return LocalHost instead.
550  url.setHost(QHostAddress(QHostAddress::LocalHost).toString());
551  } else {
552  url.setHost(serverAddress().toString());
553  }
554 
555  return url;
556 }
