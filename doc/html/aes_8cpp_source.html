15 /*
16 * This implementation is based on table lookups which are known to be
17 * vulnerable to timing and cache based side channel attacks. Some
18 * countermeasures are used which may be helpful in some situations:
19 *
20 * - Only a single 256-word T-table is used, with rotations applied.
21 * Most implementations use 4 (or sometimes 5) T-tables, which leaks
22 * much more information via cache usage.
23 *
24 * - The TE and TD tables are computed at runtime to avoid flush+reload
25 * attacks using clflush. As different processes will not share the
26 * same underlying table data, an attacker can't manipulate another
27 * processes cache lines via their shared reference to the library
28 * read only segment. (However, prime+probe attacks are still possible.)
29 *
30 * - Each cache line of the lookup tables is accessed at the beginning
31 * of each call to encrypt or decrypt. (See the Z variable below)
32 *
33 * If available SSSE3 or AES-NI are used instead of this version, as both
34 * are faster and immune to side channel attacks.
35 *
36 * Some AES cache timing papers for reference:
37 *
38 * "Software mitigations to hedge AES against cache-based software side
39 * channel vulnerabilities" https://eprint.iacr.org/2006/052.pdf
40 *
41 * "Cache Games - Bringing Access-Based Cache Attacks on AES to Practice"
42 * http://www.ieee-security.org/TC/SP2011/PAPERS/2011/paper031.pdf
43 *
44 * "Cache-Collision Timing Attacks Against AES" Bonneau, Mironov
45 * http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.88.4753
46 */

102 inline constexpr uint8_t xtime(uint8_t s) { return static_cast<uint8_t>(s << 1) ^ ((s >> 7) * 0x1B); }
103 inline constexpr uint8_t xtime4(uint8_t s) { return xtime(xtime(s)); }
104 inline constexpr uint8_t xtime8(uint8_t s) { return xtime(xtime(xtime(s))); }

106 inline constexpr uint8_t xtime3(uint8_t s) { return xtime(s) ^ s; }
107 inline constexpr uint8_t xtime9(uint8_t s) { return xtime8(s) ^ s; }

180 /*
181 * AES Encryption
182 */
183 void aes_encrypt_n(const uint8_t in[], uint8_t out[],
184  size_t blocks,
185  const secure_vector<uint32_t>& EK,
186  const secure_vector<uint8_t>& ME)
187  {
188  BOTAN_ASSERT(EK.size() && ME.size() == 16, "Key was set");
189 
190  const size_t cache_line_size = CPUID::cache_line_size();
191  const uint32_t* TE = AES_TE();
192 
193  // Hit every cache line of TE
194  volatile uint32_t Z = 0;
195  for(size_t i = 0; i < 256; i += cache_line_size / sizeof(uint32_t))
196  {
197  Z |= TE[i];
198  }
199  Z &= TE[82]; // this is zero, which hopefully the compiler cannot deduce
200 
201  for(size_t i = 0; i < blocks; ++i)
202  {
203  uint32_t T0, T1, T2, T3;
204  load_be(in + 16*i, T0, T1, T2, T3);
205 
206  T0 ^= EK[0];
207  T1 ^= EK[1];
208  T2 ^= EK[2];
209  T3 ^= EK[3];
210 
211  T0 ^= Z;
212 
213  uint32_t B0 = AES_T(TE, EK[4], T0, T1, T2, T3);
214  uint32_t B1 = AES_T(TE, EK[5], T1, T2, T3, T0);
215  uint32_t B2 = AES_T(TE, EK[6], T2, T3, T0, T1);
216  uint32_t B3 = AES_T(TE, EK[7], T3, T0, T1, T2);
217 
218  for(size_t r = 2*4; r < EK.size(); r += 2*4)
219  {
220  T0 = AES_T(TE, EK[r ], B0, B1, B2, B3);
221  T1 = AES_T(TE, EK[r+1], B1, B2, B3, B0);
222  T2 = AES_T(TE, EK[r+2], B2, B3, B0, B1);
223  T3 = AES_T(TE, EK[r+3], B3, B0, B1, B2);
224 
225  B0 = AES_T(TE, EK[r+4], T0, T1, T2, T3);
226  B1 = AES_T(TE, EK[r+5], T1, T2, T3, T0);
227  B2 = AES_T(TE, EK[r+6], T2, T3, T0, T1);
228  B3 = AES_T(TE, EK[r+7], T3, T0, T1, T2);
229  }
230 
231  /*
232  * Use TE[x] >> 8 instead of SE[] so encryption only references a single
233  * lookup table.
234  */
235  out[16*i+ 0] = static_cast<uint8_t>(TE[get_byte(0, B0)] >> 8) ^ ME[0];
236  out[16*i+ 1] = static_cast<uint8_t>(TE[get_byte(1, B1)] >> 8) ^ ME[1];
237  out[16*i+ 2] = static_cast<uint8_t>(TE[get_byte(2, B2)] >> 8) ^ ME[2];
238  out[16*i+ 3] = static_cast<uint8_t>(TE[get_byte(3, B3)] >> 8) ^ ME[3];
239  out[16*i+ 4] = static_cast<uint8_t>(TE[get_byte(0, B1)] >> 8) ^ ME[4];
240  out[16*i+ 5] = static_cast<uint8_t>(TE[get_byte(1, B2)] >> 8) ^ ME[5];
241  out[16*i+ 6] = static_cast<uint8_t>(TE[get_byte(2, B3)] >> 8) ^ ME[6];
242  out[16*i+ 7] = static_cast<uint8_t>(TE[get_byte(3, B0)] >> 8) ^ ME[7];
243  out[16*i+ 8] = static_cast<uint8_t>(TE[get_byte(0, B2)] >> 8) ^ ME[8];
244  out[16*i+ 9] = static_cast<uint8_t>(TE[get_byte(1, B3)] >> 8) ^ ME[9];
245  out[16*i+10] = static_cast<uint8_t>(TE[get_byte(2, B0)] >> 8) ^ ME[10];
246  out[16*i+11] = static_cast<uint8_t>(TE[get_byte(3, B1)] >> 8) ^ ME[11];
247  out[16*i+12] = static_cast<uint8_t>(TE[get_byte(0, B3)] >> 8) ^ ME[12];
248  out[16*i+13] = static_cast<uint8_t>(TE[get_byte(1, B0)] >> 8) ^ ME[13];
249  out[16*i+14] = static_cast<uint8_t>(TE[get_byte(2, B1)] >> 8) ^ ME[14];
250  out[16*i+15] = static_cast<uint8_t>(TE[get_byte(3, B2)] >> 8) ^ ME[15];
251  }
252  }

257 void aes_decrypt_n(const uint8_t in[], uint8_t out[], size_t blocks,
258  const secure_vector<uint32_t>& DK,
259  const secure_vector<uint8_t>& MD)
260  {
261  BOTAN_ASSERT(DK.size() && MD.size() == 16, "Key was set");
262 
263  const size_t cache_line_size = CPUID::cache_line_size();
264  const uint32_t* TD = AES_TD();
265 
266  volatile uint32_t Z = 0;
267  for(size_t i = 0; i < 256; i += cache_line_size / sizeof(uint32_t))
268  {
269  Z |= TD[i];
270  }
271  for(size_t i = 0; i < 256; i += cache_line_size)
272  {
273  Z |= SD[i];
274  }
275  Z &= TD[99]; // this is zero, which hopefully the compiler cannot deduce
276 
277  for(size_t i = 0; i != blocks; ++i)
278  {
279  uint32_t T0 = load_be<uint32_t>(in, 0) ^ DK[0];
280  uint32_t T1 = load_be<uint32_t>(in, 1) ^ DK[1];
281  uint32_t T2 = load_be<uint32_t>(in, 2) ^ DK[2];
282  uint32_t T3 = load_be<uint32_t>(in, 3) ^ DK[3];
283 
284  T0 ^= Z;
285 
286  uint32_t B0 = AES_T(TD, DK[4], T0, T3, T2, T1);
287  uint32_t B1 = AES_T(TD, DK[5], T1, T0, T3, T2);
288  uint32_t B2 = AES_T(TD, DK[6], T2, T1, T0, T3);
289  uint32_t B3 = AES_T(TD, DK[7], T3, T2, T1, T0);
290 
291  for(size_t r = 2*4; r < DK.size(); r += 2*4)
292  {
293  T0 = AES_T(TD, DK[r ], B0, B3, B2, B1);
294  T1 = AES_T(TD, DK[r+1], B1, B0, B3, B2);
295  T2 = AES_T(TD, DK[r+2], B2, B1, B0, B3);
296  T3 = AES_T(TD, DK[r+3], B3, B2, B1, B0);
297 
298  B0 = AES_T(TD, DK[r+4], T0, T3, T2, T1);
299  B1 = AES_T(TD, DK[r+5], T1, T0, T3, T2);
300  B2 = AES_T(TD, DK[r+6], T2, T1, T0, T3);
301  B3 = AES_T(TD, DK[r+7], T3, T2, T1, T0);
302  }
303 
304  out[ 0] = SD[get_byte(0, B0)] ^ MD[0];
305  out[ 1] = SD[get_byte(1, B3)] ^ MD[1];
306  out[ 2] = SD[get_byte(2, B2)] ^ MD[2];
307  out[ 3] = SD[get_byte(3, B1)] ^ MD[3];
308  out[ 4] = SD[get_byte(0, B1)] ^ MD[4];
309  out[ 5] = SD[get_byte(1, B0)] ^ MD[5];
310  out[ 6] = SD[get_byte(2, B3)] ^ MD[6];
311  out[ 7] = SD[get_byte(3, B2)] ^ MD[7];
312  out[ 8] = SD[get_byte(0, B2)] ^ MD[8];
313  out[ 9] = SD[get_byte(1, B1)] ^ MD[9];
314  out[10] = SD[get_byte(2, B0)] ^ MD[10];
315  out[11] = SD[get_byte(3, B3)] ^ MD[11];
316  out[12] = SD[get_byte(0, B3)] ^ MD[12];
317  out[13] = SD[get_byte(1, B2)] ^ MD[13];
318  out[14] = SD[get_byte(2, B1)] ^ MD[14];
319  out[15] = SD[get_byte(3, B0)] ^ MD[15];
320 
321  in += 16;
322  out += 16;
323  }
324  }

326 void aes_key_schedule(const uint8_t key[], size_t length,
327  secure_vector<uint32_t>& EK,
328  secure_vector<uint32_t>& DK,
329  secure_vector<uint8_t>& ME,
330  secure_vector<uint8_t>& MD)
331  {
332  static const uint32_t RC[10] = {
333  0x01000000, 0x02000000, 0x04000000, 0x08000000, 0x10000000,
334  0x20000000, 0x40000000, 0x80000000, 0x1B000000, 0x36000000 };
335 
336  const size_t rounds = (length / 4) + 6;
337 
338  secure_vector<uint32_t> XEK(length + 32), XDK(length + 32);
339 
340  const size_t X = length / 4;
341 
342  // Can't happen, but make static analyzers happy
343  BOTAN_ARG_CHECK(X == 4 || X == 6 || X == 8, "Invalid AES key size");
344 
345  const uint32_t* TD = AES_TD();
346 
347  // Prefetch TD and SE which are used later on in this function
348  volatile uint32_t Z = 0;
349  const size_t cache_line_size = CPUID::cache_line_size();
350 
351  for(size_t i = 0; i < 256; i += cache_line_size / sizeof(uint32_t))
352  {
353  Z |= TD[i];
354  }
355  for(size_t i = 0; i < 256; i += cache_line_size)
356  {
357  Z |= SE[i];
358  }
359  Z &= TD[99]; // this is zero, which hopefully the compiler cannot deduce
360 
361  for(size_t i = 0; i != X; ++i)
362  XEK[i] = Z ^ load_be<uint32_t>(key, i);
363 
364  for(size_t i = X; i < 4*(rounds+1); i += X)
365  {
366  XEK[i] = XEK[i-X] ^ RC[(i-X)/X] ^ SE_word(rotl<8>(XEK[i-1]));
367 
368  for(size_t j = 1; j != X; ++j)
369  {
370  XEK[i+j] = XEK[i+j-X];
371 
372  if(X == 8 && j == 4)
373  XEK[i+j] ^= SE_word(XEK[i+j-1]);
374  else
375  XEK[i+j] ^= XEK[i+j-1];
376  }
377  }
378 
379  for(size_t i = 0; i != 4*(rounds+1); i += 4)
380  {
381  XDK[i ] = XEK[4*rounds-i ];
382  XDK[i+1] = XEK[4*rounds-i+1];
383  XDK[i+2] = XEK[4*rounds-i+2];
384  XDK[i+3] = XEK[4*rounds-i+3];
385  }
386 
387  for(size_t i = 4; i != length + 24; ++i)
388  {
389  XDK[i] = Z ^ SE_word(XDK[i]);
390  XDK[i] = AES_T(TD, 0, XDK[i], XDK[i], XDK[i], XDK[i]);
391  }
392 
393  ME.resize(16);
394  MD.resize(16);
395 
396  for(size_t i = 0; i != 4; ++i)
397  {
398  store_be(XEK[i+4*rounds], &ME[4*i]);
399  store_be(XEK[i], &MD[4*i]);
400  }
401 
402  EK.resize(length + 24);
403  DK.resize(length + 24);
