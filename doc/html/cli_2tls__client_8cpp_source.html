280  int connect_to_host(const std::string& host, uint16_t port, bool tcp)
281  {
282  addrinfo hints;
283  std::memset(&hints, 0, sizeof(hints));
284  hints.ai_family = AF_UNSPEC;
285  hints.ai_socktype = tcp ? SOCK_STREAM : SOCK_DGRAM;
286  addrinfo* res, *rp = nullptr;
287 
288  if(::getaddrinfo(host.c_str(), std::to_string(port).c_str(), &hints, &res) != 0)
289  {
290  throw CLI_Error("getaddrinfo failed for " + host);
291  }
292 
293  int fd = 0;
294 
295  for(rp = res; rp != nullptr; rp = rp->ai_next)
296  {
297  fd = ::socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
298 
299  if(fd == -1)
300  {
301  continue;
302  }
303 
304  if(::connect(fd, rp->ai_addr, rp->ai_addrlen) != 0)
305  {
306  ::close(fd);
307  continue;
308  }
309 
310  break;
311  }
312 
313  ::freeaddrinfo(res);
314 
315  if(rp == nullptr) // no address succeeded
316  {
317  throw CLI_Error("connect failed");
318  }
319 
320  return fd;
321  }
