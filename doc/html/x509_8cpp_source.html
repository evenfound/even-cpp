28 class Sign_Cert final : public Command
29  {
30  public:
31  Sign_Cert()
32  : Command("sign_cert --ca-key-pass= --hash=SHA-256 "
33  "--duration=365 --emsa= ca_cert ca_key pkcs10_req") {}
34 
35  std::string group() const override
36  {
37  return "x509";
38  }
39 
40  std::string description() const override
41  {
42  return "Create a CA-signed X.509 certificate from a PKCS #10 CSR";
43  }
44 
45  void go() override
46  {
47  Botan::X509_Certificate ca_cert(get_arg("ca_cert"));
48  std::unique_ptr<Botan::Private_Key> key;
49  const std::string pass = get_arg("ca-key-pass");
50  const std::string emsa = get_arg("emsa");
51  const std::string hash = get_arg("hash");
52 
53  if(!pass.empty())
54  {
55  key.reset(Botan::PKCS8::load_key(get_arg("ca_key"), rng(), pass));
56  }
57  else
58  {
59  key.reset(Botan::PKCS8::load_key(get_arg("ca_key"), rng()));
60  }
61 
62  if(!key)
63  {
64  throw CLI_Error("Failed to load key from " + get_arg("ca_key"));
65  }
66 
67  std::map<std::string, std::string> options;
68  if(emsa.empty() == false)
69  options["padding"] = emsa;
70 
71  Botan::X509_CA ca(ca_cert, *key, options, hash, rng());
72 
73  Botan::PKCS10_Request req(get_arg("pkcs10_req"));
74 
75  auto now = std::chrono::system_clock::now();
76 
77  Botan::X509_Time start_time(now);
78 
79  typedef std::chrono::duration<int, std::ratio<86400>> days;
80 
81  Botan::X509_Time end_time(now + days(get_arg_sz("duration")));
82 
83  Botan::X509_Certificate new_cert = ca.sign_request(req, rng(), start_time, end_time);
84 
85  output() << new_cert.PEM_encode();
86  }
87  };
