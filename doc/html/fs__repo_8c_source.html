360 int fs_repo_open_config(struct FSRepo* repo) {
361  int retVal;
362  char* data;
363  size_t full_filename_length = strlen(repo->path) + 8;
364  char full_filename[full_filename_length];
365  retVal = os_utils_filepath_join(repo->path, "config", full_filename, full_filename_length);
366  if (retVal == 0)
367  return 0;
368  retVal = _read_file(full_filename, &data);
369  // parse the data
370  jsmn_parser parser;
371  jsmn_init(&parser);
372  int num_tokens = 256;
373  jsmntok_t tokens[num_tokens];
374  num_tokens = jsmn_parse(&parser, data, strlen(data), tokens, 256);
375  if (num_tokens <= 0) {
376  free(data);
377  return 0;
378  }
379  // fill FSRepo struct
380  // allocation done by fsrepo_new... repo->config = malloc(sizeof(struct RepoConfig));
381  // Identity
382  int curr_pos = _find_token(data, tokens, num_tokens, 0, "Identity");
383  if (curr_pos < 0) {
384  free(data);
385  return 0;
386  }
387  // the next should be the array, then string "PeerID"
388  //NOTE: the code below compares the peer id of the file with the peer id generated
389  // by the key. If they don't match, we fail.
390  unsigned char* test_peer_id = NULL;
391  _get_json_string_value(data, tokens, num_tokens, curr_pos, "PeerID", (char**)&test_peer_id);
392  char* priv_key_base64;
393  // then PrivKey
394  _get_json_string_value(data, tokens, num_tokens, curr_pos, "PrivKey", &priv_key_base64);
395  retVal = repo_config_identity_build_private_key(repo->config->identity, priv_key_base64);
396  if (retVal == 0
397  || strlen((char*)test_peer_id) != repo->config->identity->peer->id_size
398  || strcmp((char*)test_peer_id, repo->config->identity->peer->id) != 0) {
399  free(data);
400  free(priv_key_base64);
401  free(test_peer_id);
402  return 0;
403  }
404  repo->config->identity->peer->is_local = 1;
405  free(test_peer_id);
406 
407  // now the datastore
408  //int datastore_position = _find_token(data, tokens, num_tokens, 0, "Datastore");
409  _get_json_string_value(data, tokens, num_tokens, curr_pos, "Type", &repo->config->datastore->type);
410  _get_json_string_value(data, tokens, num_tokens, curr_pos, "Path", &repo->config->datastore->path);
411  _get_json_string_value(data, tokens, num_tokens, curr_pos, "StorageMax", &repo->config->datastore->storage_max);
412  _get_json_int_value(data, tokens, num_tokens, curr_pos, "StorageGCWatermark", &repo->config->datastore->storage_gc_watermark);
413  _get_json_string_value(data, tokens, num_tokens, curr_pos, "GCPeriod", &repo->config->datastore->gc_period);
414  _get_json_string_value(data, tokens, num_tokens, curr_pos, "Params", &repo->config->datastore->params);
415  _get_json_int_value(data, tokens, num_tokens, curr_pos, "NoSync", &repo->config->datastore->no_sync);
416  _get_json_int_value(data, tokens, num_tokens, curr_pos, "HashOnRead", &repo->config->datastore->hash_on_read);
417  _get_json_int_value(data, tokens, num_tokens, curr_pos, "BloomFilterSize", &repo->config->datastore->bloom_filter_size);
418 
419  // get addresses. First is Swarm array, then Api, then Gateway
420  curr_pos = _find_token(data, tokens, num_tokens, curr_pos, "Addresses");
421  if (curr_pos < 0) {
422  free(data);
423  return 0;
424  }
425  // get swarm addresses
426  int swarm_pos = _find_token(data, tokens, num_tokens, curr_pos, "Swarm") + 1;
427  if (tokens[swarm_pos].type != JSMN_ARRAY)
428  return 0;
429  int swarm_size = tokens[swarm_pos].size;
430  swarm_pos++;
431  repo->config->addresses->swarm_head = NULL;
432  struct Libp2pLinkedList* last = NULL;
433  struct Libp2pLinkedList* current_ma_pos = repo->config->identity->peer->addr_head;
434  for(int i = 0; i < swarm_size; i++) {
435  struct Libp2pLinkedList* current = libp2p_utils_linked_list_new();
436  if (!_get_json_string_value(data, tokens, num_tokens, swarm_pos + i, NULL, (char**)&current->item))
437  break;
438  if (repo->config->addresses->swarm_head == NULL) {
439  repo->config->addresses->swarm_head = current;
440  } else {
441  last->next = current;
442  }
443  last = current;
444  // add current to peer too
445  struct MultiAddress* ma = multiaddress_new_from_string(current->item);
446  if (ma != NULL) {
447  if (current_ma_pos == NULL) {
448  repo->config->identity->peer->addr_head = libp2p_utils_linked_list_new();
449  current_ma_pos = repo->config->identity->peer->addr_head;
450  } else {
451  struct Libp2pLinkedList* next_ma_pos = libp2p_utils_linked_list_new();
452  current_ma_pos->next = next_ma_pos;
453  current_ma_pos = next_ma_pos;
454  }
455  current_ma_pos->item = ma;
456  }
457  }
458  _get_json_string_value(data, tokens, num_tokens, curr_pos, "API", &repo->config->addresses->api);
459  _get_json_string_value(data, tokens, num_tokens, curr_pos, "Gateway", &repo->config->addresses->gateway);
460 
461  // bootstrap peers
462  swarm_pos = _find_token(data, tokens, num_tokens, curr_pos, "Bootstrap");
463  if (swarm_pos >= 0) {
464  swarm_pos++;
465  if (tokens[swarm_pos].type != JSMN_ARRAY) {
466  free(data);
467  return 0;
468  }
469  swarm_size = tokens[swarm_pos].size;
470  repo->config->bootstrap_peers = libp2p_utils_vector_new(swarm_size);
471  swarm_pos++;
472  for(int i = 0; i < swarm_size; i++) {
473  char* val = NULL;
474  if (!_get_json_string_value(data, tokens, num_tokens, swarm_pos + i, NULL, &val))
475  break;
476  struct MultiAddress* cur = multiaddress_new_from_string(val);
477  if (cur == NULL)
478  continue;
479  libp2p_utils_vector_add(repo->config->bootstrap_peers, cur);
480  free(val);
481  }
482  }
483 
484  // replication
485  curr_pos = _find_token(data, tokens, num_tokens, curr_pos, "Replication");
486  if (curr_pos >= 0) {
487  // announce minutes
488  curr_pos++;
489  _get_json_int_value(data, tokens, num_tokens, curr_pos, "AnnounceMinutes", &repo->config->replication->announce_minutes);
490  _get_json_int_value(data, tokens, num_tokens, curr_pos, "Announce", &repo->config->replication->announce);
491  // nodes list
492  int nodes_pos = _find_token(data, tokens, num_tokens, curr_pos, "Peers");
493  if (nodes_pos >= 0) {
494  nodes_pos++;
495  if (tokens[nodes_pos].type == JSMN_ARRAY) {
496  int nodes_size = tokens[nodes_pos].size;
497  repo->config->replication->replication_peers = libp2p_utils_vector_new(nodes_size);
498  nodes_pos++;
499  for(int i = 0; i < nodes_size; i++) {
500  char* val = NULL;
501  if (!_get_json_string_value(data, tokens, num_tokens, nodes_pos, NULL, &val))
502  break;
503  struct MultiAddress* cur = multiaddress_new_from_string(val);
504  if (cur == NULL)
505  continue;
506  // make multiaddress a peer
507  struct Libp2pPeer* peer = libp2p_peer_new_from_multiaddress(cur);
508  multiaddress_free(cur);
509  struct ReplicationPeer* rp = repo_config_replication_peer_new();
510  rp->peer = peer;
511  libp2p_logger_debug("fs_repo", "Adding %s to replication_peers.\n", libp2p_peer_id_to_string(rp->peer));
512  libp2p_utils_vector_add(repo->config->replication->replication_peers, rp);
513  free(val);
514  }
515  } else {
516  libp2p_logger_debug("fs_repo", "Replication|Peers is not an array.\n");
517  }
518  } else {
519  libp2p_logger_debug("fs_repo", "No replication peers found.\n");
520  }
521  }
522  // free the memory used reading the json file
523  free(data);
524  free(priv_key_base64);
525  return 1;
526 }
