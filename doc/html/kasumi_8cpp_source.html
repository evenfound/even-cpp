93 uint16_t FI(uint16_t I, uint16_t K)
94  {
95  uint16_t D9 = (I >> 7);
96  uint8_t D7 = (I & 0x7F);
97  D9 = KASUMI_SBOX_S9[D9] ^ D7;
98  D7 = KASUMI_SBOX_S7[D7] ^ (D9 & 0x7F);
99 
100  D7 ^= (K >> 9);
101  D9 = KASUMI_SBOX_S9[D9 ^ (K & 0x1FF)] ^ D7;
102  D7 = KASUMI_SBOX_S7[D7] ^ (D9 & 0x7F);
103  return static_cast<uint16_t>(D7 << 9) | D9;
104  }
