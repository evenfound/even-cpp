98 bool SshKeyExchange::sendDhInitPacket(const SshIncomingPacket &serverKexInit)
99 {
100 #ifdef CREATOR_SSH_DEBUG
101  qDebug("server requests key exchange");
102 #endif
103  serverKexInit.printRawBytes();
104  SshKeyExchangeInit kexInitParams
105  = serverKexInit.extractKeyExchangeInitData();
106 
107  printNameList("Key Algorithms", kexInitParams.keyAlgorithms);
108  printNameList("Server Host Key Algorithms", kexInitParams.serverHostKeyAlgorithms);
109  printNameList("Encryption algorithms client to server", kexInitParams.encryptionAlgorithmsClientToServer);
110  printNameList("Encryption algorithms server to client", kexInitParams.encryptionAlgorithmsServerToClient);
111  printNameList("MAC algorithms client to server", kexInitParams.macAlgorithmsClientToServer);
112  printNameList("MAC algorithms server to client", kexInitParams.macAlgorithmsServerToClient);
113  printNameList("Compression algorithms client to server", kexInitParams.compressionAlgorithmsClientToServer);
114  printNameList("Compression algorithms client to server", kexInitParams.compressionAlgorithmsClientToServer);
115  printNameList("Languages client to server", kexInitParams.languagesClientToServer);
116  printNameList("Languages server to client", kexInitParams.languagesServerToClient);
117 #ifdef CREATOR_SSH_DEBUG
118  qDebug("First packet follows: %d", kexInitParams.firstKexPacketFollows);
119 #endif
120 
121  const QByteArray &keyAlgo
122  = SshCapabilities::findBestMatch(SshCapabilities::KeyExchangeMethods,
123  kexInitParams.keyAlgorithms.names);
124  m_serverHostKeyAlgo
125  = SshCapabilities::findBestMatch(SshCapabilities::PublicKeyAlgorithms,
126  kexInitParams.serverHostKeyAlgorithms.names);
127  m_encryptionAlgo
128  = SshCapabilities::findBestMatch(SshCapabilities::EncryptionAlgorithms,
129  kexInitParams.encryptionAlgorithmsClientToServer.names);
130  m_decryptionAlgo
131  = SshCapabilities::findBestMatch(SshCapabilities::EncryptionAlgorithms,
132  kexInitParams.encryptionAlgorithmsServerToClient.names);
133  m_c2sHMacAlgo
134  = SshCapabilities::findBestMatch(SshCapabilities::MacAlgorithms,
135  kexInitParams.macAlgorithmsClientToServer.names);
136  m_s2cHMacAlgo
137  = SshCapabilities::findBestMatch(SshCapabilities::MacAlgorithms,
138  kexInitParams.macAlgorithmsServerToClient.names);
139  SshCapabilities::findBestMatch(SshCapabilities::CompressionAlgorithms,
140  kexInitParams.compressionAlgorithmsClientToServer.names);
141  SshCapabilities::findBestMatch(SshCapabilities::CompressionAlgorithms,
142  kexInitParams.compressionAlgorithmsServerToClient.names);
143 
144  AutoSeeded_RNG rng;
145  m_dhKey.reset(new DH_PrivateKey(rng,
146  DL_Group(botanKeyExchangeAlgoName(keyAlgo))));
147 
148  m_serverKexInitPayload = serverKexInit.payLoad();
149  m_sendFacility.sendKeyDhInitPacket(m_dhKey->get_y());
150  return kexInitParams.firstKexPacketFollows;
151 }

153 void SshKeyExchange::sendNewKeysPacket(const SshIncomingPacket &dhReply,
154  const QByteArray &clientId)
155 {
156  const SshKeyExchangeReply &reply
157  = dhReply.extractKeyExchangeReply(m_serverHostKeyAlgo);
158  if (reply.f <= 0 || reply.f >= m_dhKey->group_p()) {
159  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_KEY_EXCHANGE_FAILED,
160  "Server sent invalid f.");
161  }
162 
163  QByteArray concatenatedData = AbstractSshPacket::encodeString(clientId);
164  concatenatedData += AbstractSshPacket::encodeString(m_serverId);
165  concatenatedData += AbstractSshPacket::encodeString(m_clientKexInitPayload);
166  concatenatedData += AbstractSshPacket::encodeString(m_serverKexInitPayload);
167  concatenatedData += reply.k_s;
168  concatenatedData += AbstractSshPacket::encodeMpInt(m_dhKey->get_y());
169  concatenatedData += AbstractSshPacket::encodeMpInt(reply.f);
170  const BigInt k = power_mod(reply.f, m_dhKey->get_x(), m_dhKey->get_domain().get_p());
171  m_k = AbstractSshPacket::encodeMpInt(k);
172  concatenatedData += m_k;
173 
174  m_hash.reset(HashFunction::create_or_throw(botanSha1Name())->clone());
175  const SecureVector<byte> &hashResult
176  = m_hash->process(convertByteArray(concatenatedData),
177  concatenatedData.size());
178  m_h = convertByteArray(hashResult);
179 
180 #ifdef CREATOR_SSH_DEBUG
181  printData("Client Id", AbstractSshPacket::encodeString(clientId));
182  printData("Server Id", AbstractSshPacket::encodeString(m_serverId));
183  printData("Client Payload", AbstractSshPacket::encodeString(m_clientKexInitPayload));
184  printData("Server payload", AbstractSshPacket::encodeString(m_serverKexInitPayload));
185  printData("K_S", reply.k_s);
186  printData("y", AbstractSshPacket::encodeMpInt(m_dhKey->get_y()));
187  printData("f", AbstractSshPacket::encodeMpInt(reply.f));
188  printData("K", m_k);
189  printData("Concatenated data", concatenatedData);
190  printData("H", m_h);
191 #endif // CREATOR_SSH_DEBUG
192 
193  QScopedPointer<Public_Key> sigKey;
194  QScopedPointer<PK_Verifier> verifier;
195  if (m_serverHostKeyAlgo == SshCapabilities::PubKeyDss) {
196  const DL_Group group(reply.parameters.at(0), reply.parameters.at(1),
197  reply.parameters.at(2));
198  DSA_PublicKey * const dsaKey
199  = new DSA_PublicKey(group, reply.parameters.at(3));
200  sigKey.reset(dsaKey);
201  verifier.reset(new PK_Verifier(*dsaKey, botanEmsaAlgoName(SshCapabilities::PubKeyDss)));
202  } else if (m_serverHostKeyAlgo == SshCapabilities::PubKeyRsa) {
203  RSA_PublicKey * const rsaKey
204  = new RSA_PublicKey(reply.parameters.at(1), reply.parameters.at(0));
205  sigKey.reset(rsaKey);
206  verifier.reset(new PK_Verifier(*rsaKey, botanEmsaAlgoName(SshCapabilities::PubKeyRsa)));
207  } else {
208  Q_ASSERT(!"Impossible: Neither DSS nor RSA!");
209  }
210  const byte * const botanH = convertByteArray(m_h);
211  const Botan::byte * const botanSig
212  = convertByteArray(reply.signatureBlob);
213  if (!verifier->verify_message(botanH, m_h.size(), botanSig,
214  reply.signatureBlob.size())) {
215  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_KEY_EXCHANGE_FAILED,
216  "Invalid signature in SSH_MSG_KEXDH_REPLY packet.");
217  }
218 
219  m_sendFacility.sendNewKeysPacket();
220 }
