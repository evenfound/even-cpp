28 int ffi_guard_thunk(const char* func_name, std::function<int ()> thunk)
29  {
30  try
31  {
32  return thunk();
33  }
34  catch(std::bad_alloc&)
35  {
36  return ffi_error_exception_thrown(func_name, "bad_alloc", BOTAN_FFI_ERROR_OUT_OF_MEMORY);
37  }
38  catch(Botan_FFI::FFI_Error& e)
39  {
40  return ffi_error_exception_thrown(func_name, e.what(), e.error_code());
41  }
42  catch(Botan::Lookup_Error& e)
43  {
44  return ffi_error_exception_thrown(func_name, e.what(), BOTAN_FFI_ERROR_NOT_IMPLEMENTED);
45  }
46  catch(Botan::Invalid_Key_Length& e)
47  {
48  return ffi_error_exception_thrown(func_name, e.what(), BOTAN_FFI_ERROR_INVALID_KEY_LENGTH);
49  }
50  catch(Botan::Key_Not_Set& e)
51  {
52  return ffi_error_exception_thrown(func_name, e.what(), BOTAN_FFI_ERROR_KEY_NOT_SET);
53  }
54  catch(Botan::Invalid_Argument& e)
55  {
56  return ffi_error_exception_thrown(func_name, e.what(), BOTAN_FFI_ERROR_BAD_PARAMETER);
57  }
58  catch(Botan::Not_Implemented& e)
59  {
60  return ffi_error_exception_thrown(func_name, e.what(), BOTAN_FFI_ERROR_NOT_IMPLEMENTED);
61  }
62  catch(std::exception& e)
63  {
64  return ffi_error_exception_thrown(func_name, e.what());
65  }
66  catch(...)
67  {
68  return ffi_error_exception_thrown(func_name, "unknown exception");
69  }
70 
71  return BOTAN_FFI_ERROR_UNKNOWN_ERROR;
72  }
