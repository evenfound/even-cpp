84

91  const size_t c9_z = run_compression(result, 9, *c, *d, all_zeros);
92  const size_t c1_r = run_compression(result, 1, *c, *d, random_binary);
93  const size_t c9_r = run_compression(result, 9, *c, *d, random_binary);
94  const size_t c1_t = run_compression(result, 1, *c, *d, text);
95  const size_t c9_t = run_compression(result, 9, *c, *d, text);
96  const size_t c1_s = run_compression(result, 1, *c, *d, short_text);
97  const size_t c9_s = run_compression(result, 9, *c, *d, short_text);
98 
99  result.test_gte("Empty input L1 compresses to non-empty output", c1_e, 1);
100  result.test_gte("Empty input L9 compresses to non-empty output", c9_e, 1);
101 
102  result.test_gte("Level 9 compresses empty at least as well as level 1", c1_e, c9_e);
103  result.test_gte("Level 9 compresses zeros at least as well as level 1", c1_z, c9_z);
104  result.test_gte("Level 9 compresses random at least as well as level 1", c1_r, c9_r);
105  result.test_gte("Level 9 compresses text at least as well as level 1", c1_t, c9_t);
106  result.test_gte("Level 9 compresses short text at least as well as level 1", c1_s, c9_s);
107 
108  result.test_lt("Zeros compresses much better than text", c1_z / 8, c1_t);
109  result.test_lt("Text compresses much better than random", c1_t / 2, c1_r);
110 
111  results.emplace_back(result);
112  }
113  catch(std::exception& e)

124  // Returns # of bytes of compressed message
125  size_t run_compression(Test::Result& result,
126  size_t level,
127  Botan::Compression_Algorithm& c,
128  Botan::Decompression_Algorithm& d,
129  const Botan::secure_vector<uint8_t>& msg)
130  {
131  Botan::secure_vector<uint8_t> compressed(2*msg.size());
132 
133  for(bool with_flush : { true, false })
134  {
135  try
136  {
137  compressed = msg;
138 
139  c.start(level);
140  c.update(compressed, 0, false);
141 
142  if(with_flush)
143  {
144  Botan::secure_vector<uint8_t> flush_bits;
145  c.update(flush_bits, 0, true);
146  compressed += flush_bits;
147  }
148 
149  Botan::secure_vector<uint8_t> final_bits;
150  c.finish(final_bits);
151  compressed += final_bits;
152 
153  Botan::secure_vector<uint8_t> decompressed = compressed;
154  d.start();
155  d.update(decompressed);
156 
157  Botan::secure_vector<uint8_t> final_outputs;
158  d.finish(final_outputs);
159 
160  decompressed += final_outputs;
161 
162  result.test_eq("compression round tripped", msg, decompressed);
163  }
164  catch(Botan::Exception& e)
165  {
166  result.test_failure(e.what());
167  }
168  }
169 
170  return compressed.size();
171  }
