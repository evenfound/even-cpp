76 void SshAbstractCryptoFacility::recreateKeys(const SshKeyExchange &kex)
77 {
78  checkInvariant();
79 
80  if (m_sessionId.isEmpty())
81  m_sessionId = kex.h();
82  const std::string &cryptAlgo = botanCryptAlgoName(cryptAlgoName(kex));
83  BlockCipher *const cipher = BlockCipher::create_or_throw(cryptAlgo)->clone();
84 
85  m_cipherBlockSize = cipher->block_size();
86  const QByteArray ivData = generateHash(kex, ivChar(), m_cipherBlockSize);
87  const InitializationVector iv(convertByteArray(ivData), m_cipherBlockSize);
88 
89  const quint32 keySize = cipher->key_spec().maximum_keylength();
90  const QByteArray cryptKeyData = generateHash(kex, keyChar(), keySize);
91  SymmetricKey cryptKey(convertByteArray(cryptKeyData), keySize);
92 
93  Keyed_Filter * const cipherMode = makeCipherMode(cipher, new Null_Padding, iv, cryptKey);
94  m_pipe.reset(new Pipe(cipherMode));
95 
96  m_macLength = botanHMacKeyLen(hMacAlgoName(kex));
97  const QByteArray hMacKeyData = generateHash(kex, macChar(), macLength());
98  SymmetricKey hMacKey(convertByteArray(hMacKeyData), macLength());
99  const HashFunction * const hMacProto = HashFunction::create(botanHMacAlgoName(hMacAlgoName(kex)))->clone();
100  m_hMac.reset(new HMAC(hMacProto->clone()));
101  m_hMac->set_key(hMacKey);
102 }

104 void SshAbstractCryptoFacility::convert(QByteArray &data, quint32 offset,
105  quint32 dataSize) const
106 {
107  Q_ASSERT(offset + dataSize <= static_cast<quint32>(data.size()));
108  checkInvariant();
109 
110  // Session id empty => No key exchange has happened yet.
111  if (dataSize == 0 || m_sessionId.isEmpty())
112  return;
113 
114  if (dataSize % cipherBlockSize() != 0) {
115  throw SSH_SERVER_EXCEPTION(SSH_DISCONNECT_PROTOCOL_ERROR,
116  "Invalid packet size");
117  }
118  m_pipe->process_msg(reinterpret_cast<const byte *>(data.constData()) + offset,
119  dataSize);
120  quint32 bytesRead = m_pipe->read(reinterpret_cast<byte *>(data.data()) + offset,
121  dataSize, m_pipe->message_count() - 1); // Can't use Pipe::LAST_MESSAGE because of a VC bug.
122  Q_ASSERT(bytesRead == dataSize);
123 }

134 QByteArray SshAbstractCryptoFacility::generateHash(const SshKeyExchange &kex,
135  char c, quint32 length)
136 {
137  const QByteArray &k = kex.k();
138  const QByteArray &h = kex.h();
139  QByteArray data(k);
140  data.append(h).append(c).append(m_sessionId);
141  SecureVector<byte> key
142  = kex.hash()->process(convertByteArray(data), data.size());
143  while (key.size() < length) {
144  secure_vector<byte> tmpKey;
145  tmpKey += secure_vector<byte>(k.begin(), k.end());
146  tmpKey += secure_vector<byte>(h.begin(), h.end());
147  tmpKey += key;
148  key += kex.hash()->process(tmpKey);
149  }
150  return QByteArray(reinterpret_cast<const char *>(key.data()), length);
151 }

190 void SshEncryptionFacility::createAuthenticationKey(const QByteArray &privKeyFileContents)
191 {
192  if (privKeyFileContents == m_cachedPrivKeyContents)
193  return;
194 
195 #ifdef CREATOR_SSH_DEBUG
196  qDebug("%s: Key not cached, reading", Q_FUNC_INFO);
197 #endif
198  QList<BigInt> pubKeyParams;
199  QList<BigInt> allKeyParams;
200  QString error1;
201  QString error2;
202  if (!createAuthenticationKeyFromPKCS8(privKeyFileContents, pubKeyParams, allKeyParams, error1)
203  && !createAuthenticationKeyFromOpenSSL(privKeyFileContents, pubKeyParams, allKeyParams,
204  error2)) {
205 #ifdef CREATOR_SSH_DEBUG
206  qDebug("%s: %s\n\t%s\n", Q_FUNC_INFO, qPrintable(error1), qPrintable(error2));
207 #endif
208  throw SshClientException(SshKeyFileError, SSH_TR("Decoding of private key file failed: "
209  "Format not understood."));
210  }
211 
212  foreach (const BigInt &b, allKeyParams) {
213  if (b.is_zero()) {
214  throw SshClientException(SshKeyFileError,
215  SSH_TR("Decoding of private key file failed: Invalid zero parameter."));
216  }
217  }
218 
219  m_authPubKeyBlob = AbstractSshPacket::encodeString(m_authKeyAlgoName);
220  foreach (const BigInt &b, pubKeyParams)
221  m_authPubKeyBlob += AbstractSshPacket::encodeMpInt(b);
222  m_cachedPrivKeyContents = privKeyFileContents;
223 }

225 bool SshEncryptionFacility::createAuthenticationKeyFromPKCS8(const QByteArray &privKeyFileContents,
226  QList<BigInt> &pubKeyParams, QList<BigInt> &allKeyParams, QString &error)
227 {
228  try {
229  Pipe pipe;
230  pipe.process_msg(convertByteArray(privKeyFileContents), privKeyFileContents.size());
231  Private_Key * const key = PKCS8::load_key(pipe, m_rng, SshKeyPasswordRetriever::get_passphrase);
232  if (DSA_PrivateKey * const dsaKey = dynamic_cast<DSA_PrivateKey *>(key)) {
233  m_authKeyAlgoName = SshCapabilities::PubKeyDss;
234  m_authKey.reset(dsaKey);
235  pubKeyParams << dsaKey->group_p() << dsaKey->group_q()
236  << dsaKey->group_g() << dsaKey->get_y();
237  allKeyParams << pubKeyParams << dsaKey->get_x();
238  } else if (RSA_PrivateKey * const rsaKey = dynamic_cast<RSA_PrivateKey *>(key)) {
239  m_authKeyAlgoName = SshCapabilities::PubKeyRsa;
240  m_authKey.reset(rsaKey);
241  pubKeyParams << rsaKey->get_e() << rsaKey->get_n();
242  allKeyParams << pubKeyParams << rsaKey->get_p() << rsaKey->get_q()
243  << rsaKey->get_d();
244  } else {
245  qWarning("%s: Unexpected code flow, expected success or exception.", Q_FUNC_INFO);
246  return false;
247  }
248  } catch (const Botan::Decoding_Error &ex) {
249  error = QLatin1String(ex.what());
250  return false;
251  } catch (const Botan::Exception &ex) {
252  error = QLatin1String(ex.what());
253  return false;
254  }
255 
256  return true;
257 }

259 bool SshEncryptionFacility::createAuthenticationKeyFromOpenSSL(const QByteArray &privKeyFileContents,
260  QList<BigInt> &pubKeyParams, QList<BigInt> &allKeyParams, QString &error)
261 {
262  try {
263  bool syntaxOk = true;
264  QList<QByteArray> lines = privKeyFileContents.split('\n');
265  while (lines.last().isEmpty())
266  lines.removeLast();
267  if (lines.count() < 3) {
268  syntaxOk = false;
269  } else if (lines.first() == PrivKeyFileStartLineRsa) {
270  if (lines.last() != PrivKeyFileEndLineRsa)
271  syntaxOk = false;
272  else
273  m_authKeyAlgoName = SshCapabilities::PubKeyRsa;
274  } else if (lines.first() == PrivKeyFileStartLineDsa) {
275  if (lines.last() != PrivKeyFileEndLineDsa)
276  syntaxOk = false;
277  else
278  m_authKeyAlgoName = SshCapabilities::PubKeyDss;
279  } else {
280  syntaxOk = false;
281  }
282  if (!syntaxOk) {
283  error = SSH_TR("Unexpected format.");
284  return false;
285  }
286 
287  QByteArray privateKeyBlob;
288  for (int i = 1; i < lines.size() - 1; ++i)
289  privateKeyBlob += lines.at(i);
290  privateKeyBlob = QByteArray::fromBase64(privateKeyBlob);
291 
292  BER_Decoder decoder(convertByteArray(privateKeyBlob), privateKeyBlob.size());
293  BER_Decoder sequence = decoder.start_cons(SEQUENCE);
294  size_t version;
295  sequence.decode (version);
296  if (version != 0) {
297  error = SSH_TR("Key encoding has version %1, expected 0.").arg(version);
298  return false;
299  }
300 
301  if (m_authKeyAlgoName == SshCapabilities::PubKeyDss) {
302  BigInt p, q, g, y, x;
303  sequence.decode (p).decode (q).decode (g).decode (y).decode (x);
304  DSA_PrivateKey * const dsaKey = new DSA_PrivateKey(m_rng, DL_Group(p, q, g), x);
305  m_authKey.reset(dsaKey);
306  pubKeyParams << p << q << g << y;
307  allKeyParams << pubKeyParams << x;
308  } else {
309  BigInt p, q, e, d, n;
310  sequence.decode(n).decode(e).decode(d).decode(p).decode(q);
311  RSA_PrivateKey * const rsaKey = new RSA_PrivateKey(p, q, e, d, n);
312  m_authKey.reset(rsaKey);
313  pubKeyParams << e << n;
314  allKeyParams << pubKeyParams << p << q << d;
315  }
316 
317  sequence.discard_remaining();
318  sequence.verify_end();
319  } catch (const Botan::Decoding_Error &ex) {
320  error = QLatin1String(ex.what());
321  return false;
322  } catch (const Botan::Exception &ex) {
323  error = QLatin1String(ex.what());
324  return false;
325  }
326  return true;
327 }
