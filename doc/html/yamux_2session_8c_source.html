195  return 0;

199  // retrieve the yamux context

201  struct yamux_session* yamux_session = yamuxContext->session;
202  struct yamux_stream* s = NULL;

216  if (f.version != YAMUX_VERSION) {

218  return 0;
219  }
220 
221  if (!f.streamid) {// we're not dealing with a stream, we're dealing with something at the yamux protocol level

223  switch (f.type)
224  {
225  case yamux_frame_ping: {
226  // ping
227  libp2p_logger_debug("yamux", "Received a ping.\n");
228  if (f.flags & yamux_frame_syn)
229  {

236  {

238  if (!timespec_get(&now, TIME_UTC))

241  dt.tv_sec = now.tv_sec - last.tv_sec;
242  if (now.tv_nsec < last.tv_nsec)
243  {
244  dt.tv_sec--;
245  dt.tv_nsec = last.tv_nsec - now.tv_nsec;
246  }
247  else
248  dt.tv_nsec = now.tv_nsec - last.tv_nsec;
249 
250  yamux_session->pong_fn(yamux_session, f.length, dt);
251  }
252  else
253  return -EPROTO;
254  break;
255  }
256  case yamux_frame_go_away: {
257  // go away (hanging up)
258  libp2p_logger_debug("yamux", "Received a \"go away\".\n");
259  yamux_session->closed = 1;
260  if (yamux_session->go_away_fn)
261  yamux_session->go_away_fn(yamux_session, (enum yamux_error)f.length);
262  break;
263  }
264  default: {

266  return -EPROTO;
267  }
268  }

271  // we're handling a stream, not something at the yamux protocol level
272  for (size_t i = 0; i < yamux_session->cap_streams; ++i)
273  {

275  s = ss->stream;
276 
277  if (!ss->alive || s->state == yamux_stream_closed) // skip dead or closed streams
278  continue;
279 
280  if (s->id == f.streamid) // we have a match between the stored stream and the current stream
281  {
282  libp2p_logger_debug("yamux", "We found our stream id of %d.\n", f.streamid);
283  if (f.flags & yamux_frame_rst)
284  {

286  // close the stream
287  s->state = yamux_stream_closed;
288 
289  if (s->rst_fn)
290  s->rst_fn(s);
291  }
292  else if (f.flags & yamux_frame_fin)
293  {
294  libp2p_logger_debug("yamux", "They are asking that stream %d be closed.\n", f.streamid);
295  // local stream didn't initiate FIN
296  if (s->state != yamux_stream_closing)
297  yamux_stream_close(libp2p_yamux_get_parent_channel_context(s->stream));
298 
299  s->state = yamux_stream_closed;
300 
301  if (s->fin_fn)
302  s->fin_fn(s);
303  }
304  else if (f.flags & yamux_frame_ack)
305  {
306  libp2p_logger_debug("yamux", "They sent an ack for stream %d.\n", f.streamid);
307  // acknowldegement
308  if (s->state != yamux_stream_syn_sent) {
309  libp2p_logger_debug("yamux", "We received an ack, but it seems we never sent anything!\n");
310  return -EPROTO;
311  }
312 
313  s->state = yamux_stream_est;
314  }
315  else if (f.flags) {
316  libp2p_logger_debug("yamux", "They sent no flags. I don't know what to do. Erroring out.\n");
317  return -EPROTO;
318  }
319 
320  libp2p_logger_debug("yamux", "Processing the data after the frame for stream %d, which is %d bytes.\n", f.streamid, incoming_size - frame_size);
321  ssize_t re = yamux_stream_process(s, &f, &incoming[frame_size], incoming_size - frame_size);
322  libp2p_logger_debug("yamux", "decode: yamux_stream_process for stream %d returned %d.\n", f.streamid, (int)re);
323  return (re < 0) ? re : (re + incoming_size);
324  //yamux_pull_message_from_frame(incoming, incoming_size, return_message);
325  } // stream id matches
326  }
327

329  // It must not exist yet, so let's try to make it
330  if (f.flags & yamux_frame_syn)
331  {
332  libp2p_logger_debug("yamux", "Looks like we have a new stream coming in. Stream %d.\n", f.streamid);
333  yamux_pull_message_from_frame(incoming, incoming_size, return_message);
334
