37  const BigInt& g,
38  size_t window_bits,
39  bool const_time) :
40  m_params(params),
41  m_window_bits(window_bits == 0 ? 4 : window_bits),
42  m_const_time(const_time)
43  {
44  if(m_window_bits < 1 || m_window_bits > 12) // really even 8 is too large ...
45  throw Invalid_Argument("Invalid window bits for Montgomery exponentiation");
46 
47  const size_t window_size = (static_cast<size_t>(1) << m_window_bits);

50 
51  m_g.push_back(Montgomery_Int(m_params, m_params->R1(), false));
52 
53  m_g.push_back(Montgomery_Int(m_params, g));
54 
55  const Montgomery_Int& monty_g = m_g[1];
56 
57  for(size_t i = 2; i != window_size; ++i)
58  {
59  m_g.push_back(monty_g * m_g[i - 1]);
60  }
61 
62  // Resize each element to exactly p words
63  for(size_t i = 0; i != window_size; ++i)
64  {
65  m_g[i].fix_size();

73 void const_time_lookup(secure_vector<word>& output,
74  const std::vector<Montgomery_Int>& g,
75  size_t nibble)
76  {
77  const size_t words = output.size();
78 
79  clear_mem(output.data(), output.size());
80 
81  for(size_t i = 0; i != g.size(); ++i)
82  {
83  const secure_vector<word>& vec = g[i].repr().get_word_vector();
84 
85  BOTAN_ASSERT(vec.size() >= words,
86  "Word size as expected in const_time_lookup");
87 
88  const word mask = CT::is_equal<word>(i, nibble);
89 
90  for(size_t w = 0; w != words; ++w)
91  output[w] |= (mask & vec[w]);
92  }
93  }

97 BigInt Montgomery_Exponentation_State::exponentiation(const BigInt& scalar, size_t max_k_bits) const
98  {
99  BOTAN_DEBUG_ASSERT(scalar.bits() <= max_k_bits);
100  // TODO add a const-time implementation of above assert and use it in release builds
101 
102  const size_t exp_nibbles = (max_k_bits + m_window_bits - 1) / m_window_bits;
103 
104  if(exp_nibbles == 0)
105  return 1;
106 
107  secure_vector<word> e_bits(m_params->p_words());
108  secure_vector<word> ws;
109 
110  const_time_lookup(e_bits, m_g, scalar.get_substring(m_window_bits*(exp_nibbles-1), m_window_bits));
111  Montgomery_Int x(m_params, e_bits.data(), e_bits.size(), false);
112 
113  for(size_t i = exp_nibbles - 1; i > 0; --i)
114  {
115  x.square_this_n_times(ws, m_window_bits);
116  const_time_lookup(e_bits, m_g, scalar.get_substring(m_window_bits*(i-1), m_window_bits));
117  x.mul_by(e_bits, ws);
118  }
119 
120  x.const_time_unpoison();
121  return x.value();
122  }

124 BigInt Montgomery_Exponentation_State::exponentiation_vartime(const BigInt& scalar) const
125  {
126  BOTAN_ASSERT_NOMSG(m_const_time == false);
127 
128  const size_t exp_nibbles = (scalar.bits() + m_window_bits - 1) / m_window_bits;
129 
130  secure_vector<word> ws;
131 
132  if(exp_nibbles == 0)
133  return 1;
134 
135  Montgomery_Int x = m_g[scalar.get_substring(m_window_bits*(exp_nibbles-1), m_window_bits)];
136 
137  for(size_t i = exp_nibbles - 1; i > 0; --i)
138  {
139  x.square_this_n_times(ws, m_window_bits);
140 
141  const uint32_t nibble = scalar.get_substring(m_window_bits*(i-1), m_window_bits);
142  if(nibble > 0)
143  x.mul_by(m_g[nibble], ws);
144  }
145 
146  x.const_time_unpoison();
147  return x.value();
148  }

202 
203  const Montgomery_Int y3x1 = y3.mul(x1, ws);
204  const Montgomery_Int y3x2 = y3.mul(x2, ws);
205  const Montgomery_Int y3x3 = y3.mul(x3, ws);

207  const Montgomery_Int* M[16] = {
208  &one,
209  &x1, // 0001

213  &y1x1,
