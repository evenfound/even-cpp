22 void check_ecc_math(const Botan::EC_Group& group,
23  const uint8_t in[], size_t len)
24  {

28  // This is shared across runs to reduce overhead

30 
31  const size_t hlen = len / 2;
32  const Botan::BigInt a = Botan::BigInt::decode(in, hlen);
33  const Botan::BigInt b = Botan::BigInt::decode(in + hlen, len - hlen);
34  const Botan::BigInt c = a + b;
35 
36  const Botan::PointGFp P1 = base_point * a;
37  const Botan::PointGFp Q1 = base_point * b;
38  const Botan::PointGFp R1 = base_point * c;
39

41  const Botan::PointGFp T1 = Q1 + P1;
42 
43  FUZZER_ASSERT_EQUAL(S1, R1);
44  FUZZER_ASSERT_EQUAL(T1, R1);
45 
46  const Botan::PointGFp P2 = group.blinded_base_point_multiply(a, fuzzer_rng(), ws);
47  const Botan::PointGFp Q2 = group.blinded_base_point_multiply(b, fuzzer_rng(), ws);
48  const Botan::PointGFp R2 = group.blinded_base_point_multiply(c, fuzzer_rng(), ws);
49  const Botan::PointGFp S2 = P2 + Q2;
50  const Botan::PointGFp T2 = Q2 + P2;
51

54 
55  const Botan::PointGFp P3 = group.blinded_var_point_multiply(base_point, a, fuzzer_rng(), ws);

58  const Botan::PointGFp S3 = P3 + Q3;
