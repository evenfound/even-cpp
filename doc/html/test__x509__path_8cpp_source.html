491 std::vector<Test::Result> BSI_Path_Validation_Tests::run()
492  {
493  if(Botan::has_filesystem_impl() == false)
494  {
495  return {Test::Result::Note("BSI path validation",
496  "Skipping due to missing filesystem access")};
497  }
498 
499  std::vector<Test::Result> results;
500 
501  const std::string bsi_test_dir = Test::data_dir() + "/x509/bsi";
502 
503  const std::map<std::string, std::string> expected = read_results(
504  Test::data_file("/x509/bsi/expected.txt"), '$');
505 
506  for(auto& i : expected)
507  {
508  const std::string test_name = i.first;
509  std::string expected_result = i.second;
510 
511 #if !defined(BOTAN_HAS_MD5)
512  if(expected_result == "Hash function used is considered too weak for security")
513  expected_result = "Certificate signed with unknown/unavailable algorithm";
514 #endif
515 
516  const std::string test_dir = bsi_test_dir + "/" + test_name;
517 
518  Test::Result result("BSI path validation");
519  result.start_timer();
520 
521  const std::vector<std::string> all_files =
522  Botan::get_files_recursive(test_dir);
523 
524  if(all_files.empty())
525  {
526  result.test_failure("No test files found in " + test_dir);
527  results.push_back(result);
528  continue;
529  }
530 
531  Botan::Certificate_Store_In_Memory trusted;
532  std::vector<Botan::X509_Certificate> certs;
533 
534  auto validation_time =
535  Botan::calendar_point(2017, 8, 19, 12, 0, 0).to_std_timepoint();
536 
537  // By convention: if CRL is a substring if the directory name,
538  // we need to check the CRLs
539  bool use_crl = false;
540  if(test_dir.find("CRL") != std::string::npos)
541  {
542  use_crl = true;
543  }
544 
545  try
546  {
547  for(auto const& file : all_files)
548  {
549  // found a trust anchor
550  if(file.find("TA") != std::string::npos)
551  {
552  trusted.add_certificate(Botan::X509_Certificate(file));
553  }
554  // found the target certificate. It needs to be at the front of certs
555  else if(file.find("TC") != std::string::npos)
556  {
557  certs.insert(certs.begin(), Botan::X509_Certificate(file));
558  }
559  // found a certificate that might be part of a valid certificate chain to the trust anchor
560  else if(file.find(".crt") != std::string::npos)
561  {
562  certs.push_back(Botan::X509_Certificate(file));
563  }
564  else if(file.find(".crl") != std::string::npos)
565  {
566  trusted.add_crl(Botan::X509_CRL(file));
567  }
568  }
569 
570  Botan::Path_Validation_Restrictions restrictions(use_crl, 79,
571  use_crl);
572 
573  /*
574  * Following the test document, the test are executed 16 times with
575  * randomly chosen order of the available certificates. However, the target
576  * certificate needs to stay in front.
577  * For certain test, the order in which the certificates are given to
578  * the validation function may be relevant, i.e. if issuer DNs are
579  * ambiguous.
580  */
581  struct random_bit_generator {
582  using result_type = size_t;
583  static constexpr result_type min() { return 0; }
584  static constexpr result_type max() { return std::numeric_limits<size_t>::max(); }
585  result_type operator()()
586  {
587  size_t s;
588  Test::rng().randomize(reinterpret_cast<uint8_t*>(&s), sizeof(s));
589  return s;
590  }
591  } rbg;
592 
593  for(size_t r = 0; r < 16; r++)
594  {
595  std::shuffle(++(certs.begin()), certs.end(), rbg);
596 
597  Botan::Path_Validation_Result validation_result =
598  Botan::x509_path_validate(certs, restrictions, trusted, "",
599  Botan::Usage_Type::UNSPECIFIED, validation_time);
600 
601  // We expect to be warned
602  if(expected_result.find("Warning: ") == 0)
603  {
604  std::string stripped = expected_result.substr(std::string("Warning: ").size());
605  bool found_warning = false;
606  for(const auto& warning_set : validation_result.warnings())
607  {
608  for(const auto& warning : warning_set)
609  {
610  std::string warning_str(Botan::to_string(warning));
611  if(stripped == warning_str)
612  {
613  result.test_eq(test_name + " path validation result",
614  warning_str, stripped);
615  found_warning = true;
616  }
617  }
618  }
619  if(!found_warning)
620  {
621  result.test_failure(test_name,"Did not receive the expected warning: " + stripped);
622  }
623  }
624  else
625  {
626  result.test_eq(test_name + " path validation result",
627  validation_result.result_string(), expected_result);
628  }
629 
630 
631  }
632  }
633 
634  /* Some certificates are rejected when executing the X509_Certificate constructor
635  * by throwing a Decoding_Error exception.
636  */
637  catch(const Botan::Decoding_Error& d)
638  {
639  result.test_eq(test_name + " path validation result", d.what(),
640  expected_result);
641  }
642  catch(const Botan::X509_CRL::X509_CRL_Error& e)
643  {
644  result.test_eq(test_name + " path validation result", e.what(),
645  expected_result);
646  }
647  catch(const std::exception& e)
648  {
649  result.test_failure(test_name, e.what());
650  }
651 
652  result.end_timer();
653  results.push_back(result);
654  }
655 
656  return results;
657  }
