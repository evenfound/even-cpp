210  }
211  } else {

213  struct test* current = first_test;
214  while (current != NULL) {
215  if (current->part_of_suite) {
216  tests_ran++;

218  }
219  current = current->next;
220  }
221  }
222  if (tests_ran == 0)
223  fprintf(stderr, "***** No tests found *****\n");
224  else {
225  if (counter > 0) {
226  fprintf(stderr, "***** There were %d failed (out of %d) test(s) *****\n", counter, tests_ran);
227  } else {

229  }
